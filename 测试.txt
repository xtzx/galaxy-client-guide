# 1. 查看操作系统信息
systeminfo | Select-String "OS"

# 2. 查看当前用户
whoami

# 3. 查看当前目录
pwd

# 4. 查看 IP 地址
ipconfig | Select-String "IPv4"

# 5. 查看 Node.js 版本
node --version

# 6. 查看 npm 源配置
npm config get registry

# 7. 创建一个测试文件
echo "Hello from Windows via Mac Cursor! $(Get-Date)" > C:\Users\zhouming\Downloads\remote-test.txt

# 8. 读取测试文件
Get-Content C:\Users\zhouming\Downloads\remote-test.txt
