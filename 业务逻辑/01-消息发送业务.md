## 一、业务概述

### 1.1 支持的消息类型

| 图片消息 | 3 | ✅ | ✅ | jpg/png/gif |

| 链接卡片 | 5 | ✅ | ✅ | 标题+描述+URL |

### 2.2 关键步骤详解

#### 步骤1: MQTT消息接收与过滤

```javascript
// src/msg-center/core/mq/mqttClientBase.js

client.on("message", async function (topic, message) {
    const dataInfo = JSON.parse(message.toString("utf8"));
    
    // 1. 任务过期检查（默认3小时过期）
    const expireTime = apolloConfig.mqttTaskExpireHours * 60 * 60 * 1000;
    if (dataInfo.createTime * 1000 < Date.now() - expireTime) {
        return; // 过期消息直接丢弃
    }
    
    // 2. 去重检查
    if (mqttTaskMapLock.has(dataInfo.id)) {
        return; // 重复消息直接丢弃
    }
    mqttTaskMapLock.set(dataInfo.id, {
        wxId: wxId,
        timestamp: Date.now(),
        processed: true
    });
    
    // 3. 根据type路由到处理器
    const serverTaskBO = { ...ServerTaskBO, ...dataInfo };
    if (dataInfo.type === 100) {
        // 群发消息：使用队列处理，避免过于频繁
        executeWithDelay(wxId, () => {
            execute(wxId, serverTaskBO);
        });
    } else {
        // 其他任务：延迟1秒随机
        await sleep(Math.ceil(Math.random() * 1000));
        execute(wxId, serverTaskBO);
    }
});
```

**关键配置**:
- `mqttTaskExpireHours`: MQTT任务过期时间，默认3小时
- `mqttTaskMapLock`: 任务去重锁，防止同一任务重复处理

```javascript
async function runTask(wxId, serverTaskBO, registry) {
    // 微信任务处理器列表

    const WorkWxConvertServiceList = [

        // ... 更多处理器

    // 根据机器人类型选择处理器列表

        }
    }
}
```

#### 步骤3: 消息处理器 - MqttChatService

```javascript
// src/msg-center/business/task-mqtt/mqttChatService.js

const MqttChatService = {
    // 过滤条件：群发消息(type=1)或私聊消息(type=100)
    filter(serverTask) {
        return serverTask.type === SunTaskType.CHATROOM_SEND_MSG 
            || serverTask.type === SunTaskType.FRIEND_SEND_MSG;
    },
    
    async operate(serverTask, wxId) {
        let { content, weChatMsgType } = serverTask;
        
        // 1. 构建客户端任务
        let task = ClientTaskFactory.getClientTask(serverTask, wxId);
        if (!task) return false;
        
        // 2. 媒体文件处理
        if (weChatMsgType == EMessage.TYPE__VIDEO
            || weChatMsgType == EMessage.TYPE__MSG_CARD_FILE
            || weChatMsgType == EMessage.TYPE__IMAGE) {
            
            // 如果是网络URL，需要下载到本地
            if (content.startsWith('http://') || content.startsWith('https://')) {
                let fileName = serverTask.ext;
                let fileInfo = await FileDownloadUtil.downloadAsyncReturnFileInfo(content, fileName);
                
                if (!fileInfo.filePath) {
                    // 下载失败，上报超时
                    TimeOutResponse.fileDownloadTimeOutResponse(serverTask, wxId);
                    return true;
                }
                
                task.oldFileUrl = content;           // 保存原始URL
                task.data.content = fileInfo.filePath; // 使用本地路径
            }
        }
        
        // 3. 发送到逆向
        cloudFlowInBound(null, wxId, JSON.stringify(task));
        return true;
    }
};
```

### 3.2 企微消息处理器

```javascript
// src/msg-center/business/task-mqtt/wkwx/mqttWorkWxChatService.js

const MqttWorkWxChatService = {
    filter(serverTask) {
        return serverTask.type == SunTaskType.CHATROOM_SEND_MSG 
            || serverTask.type == SunTaskType.FRIEND_SEND_MSG;
    },
    
    async operate(serverTask, wxId) {
        let workWxClientTaskBO = {};
        let { content } = serverTask;
        
        // 1. 构建会话ID
        if (serverTask.type === PrismRecordType.CHAT_MSG) {
            // 群聊格式: R:chatroom_id
            let { chatroom } = serverTask;
            if (!chatroom.startsWith("R:")) {
                chatroom = "R:" + chatroom;
            }
            workWxClientTaskBO.conversationid = chatroom;
        } else {
            // 私聊格式: S:发送者_接收者
            let to_username = serverTask.toUsernames[0];
            workWxClientTaskBO.conversationid = `S:${wxId}_${to_username}`;
        }
        
        // 2. 根据消息类型构建不同格式
        switch (serverTask.weChatMsgType) {
            case CommonConstant.CLIENT_SEND_TEXT_MSG:
                // 文本消息：需要拆分表情和@
                workWxClientTaskBO.msg_items = this.spiltText(content);
                workWxClientTaskBO.type = GalaxyTaskType.SEND_TEXT;
                break;
                
            case PrismRecordType.SEND_LINK:
                // 链接卡片
                workWxClientTaskBO.type = GalaxyTaskType.SEND_LINK;
                workWxClientTaskBO.linkpara = {
                    title: contentList[0],
                    desc: contentList[1],
                    link: contentList[2],
                    img_url: content
                };
                break;
                
            case PrismRecordType.SEND_FILE:
            case PrismRecordType.SEND_IMG:
            case PrismRecordType.SEND_VIDEO:
                // 媒体文件：先下载到本地
                let localPath = content;
                if (content.startsWith('http')) {
                    let fileInfo = await FileDownloadUtil.downloadAsyncReturnFileInfo(content, serverTask.ext);
                    localPath = fileInfo.filePath;
                }
                
                if (serverTask.weChatMsgType == PrismRecordType.SEND_FILE) {
                    workWxClientTaskBO.filefullpath = localPath;
                    workWxClientTaskBO.type = GalaxyTaskType.SEND_FILE;
                } else if (serverTask.weChatMsgType == PrismRecordType.SEND_IMG) {
                    workWxClientTaskBO.imagefullpath = localPath;
                    workWxClientTaskBO.type = GalaxyTaskType.SEND_IMAG;
                } else {
                    // 视频还需要生成缩略图
                    let imagePath = await FileUtils.videoImage(localPath);
                    workWxClientTaskBO.type = GalaxyTaskType.SEND_VIDEO;
                    workWxClientTaskBO.videomsgpara = {
                        videofullpath: localPath,
                        thumbfullpath: imagePath,
                        thumbwidth: dimensions.width,
                        thumbheight: dimensions.height,
                        seconds: mp4Duration
                    };
                }
                break;
        }
        
        // 3. 发送到逆向
        cloudFlowInBound(null, wxId, JSON.stringify(workWxClientTaskBO));
        return true;
    },
    
    // 文本拆分：识别表情和@符号
```

## 四、消息下发到逆向

### 4.1 dispatchInBound - 入站分发

```javascript
// src/msg-center/dispatch-center/dispatchInBound.js

function dispatchInBound(channelId, wxId, message) {
    // 1. 获取机器人注册信息
    let registry = RegistryConfig.getRegistryByKey(wxId, 'wxid');
    if (!registry) {
        logUtil.customLog(`尚未和逆向建立注册信息`);
        return;
    }
    
    // 2. 解析消息类型
    const jsonObject = JSON.parse(message);
    const type = jsonObject.type || jsonObject.body.type;
    
    // 3. 处理任务
    processInBoundTask({channelId, wxId, message, jsonObject, type, workWxFlag: registry.workWx});
}

function processInBoundTask(params) {
    const {channelId, wxId, message, jsonObject, type, workWxFlag} = params;
    
    // 检查是否需要任务状态跟踪
    if (TaskCallBack.taskCallbackMap[type]) {
        // 需要跟踪的任务：创建任务状态缓存
        const taskId = jsonObject.taskId;
        const galaxyTaskStatus = {
            firstMessageStatus: 'NOT_RECEIVE',
            secondMessageStatus: 'NOT_RECEIVE',
            thirdMessageStatus: 'NOT_RECEIVE',
            username: wxId,
            taskId,
            createTime: Date.now()
        };
        
        // 缓存任务状态（用于三段式追踪）
        GalaxyTaskCache.GALAXY_TASK_STATUS_MAP[taskId] = galaxyTaskStatus;
    }
    
    // 4. 发送到逆向
    inBoundAct(channelId, wxId, message);
}

function inBoundAct(channelId, wxId, message) {
    reverseSendService.sendMessage(wxId, channelId, message);
}
```

```javascript
// src/msg-center/dispatch-center/reverseSend.js

const ReverseSendService = {
    sendMessage(wxId, channelId, message) {
        // 1. 获取IPC管道连接
        const registry = RegistryConfig.getRegistryByKey(wxId, 'wxid') 
                      || RegistryConfig.getRegistryByKey(channelId, 'id');
        
        if (!registry || !registry.pipeLineWrapper) {
            logUtil.customLog(`发送失败：未找到IPC连接`);
            return;
        }
        
        // 2. 通过IPC管道发送
        const pipeLineWrapper = registry.pipeLineWrapper;
        pipeLineWrapper.write(message);
        
        logUtil.customLog(`[wxid-${wxId}] 消息已发送到逆向`);
    }
};
```

### 5.1 文件下载工具

```javascript
// src/msg-center/core/utils/fileDownload.js

const FileDownloadUtil = {
    // 下载文件并返回信息
    async downloadAsyncReturnFileInfo(url, fileName) {
        try {
            // 1. 确定文件名
            if (!fileName) {
                fileName = this.getFileNameFromUrl(url);
            }
            
            // 2. 创建本地路径
            const localDir = this.getDownloadDir();
            const filePath = path.join(localDir, fileName);
            
            // 3. 下载文件
            const response = await axios({
                method: 'get',
                url: url,
                responseType: 'stream',
                timeout: 60000
            });
            
            const writer = fs.createWriteStream(filePath);
            response.data.pipe(writer);
            
            await new Promise((resolve, reject) => {
                writer.on('finish', resolve);
                writer.on('error', reject);
            });
            
            // 4. 获取文件大小
            const stats = fs.statSync(filePath);
            
            return {
                filePath: filePath,
                fileSize: stats.size
            };
            
        } catch (error) {
            logUtil.customLog(`文件下载失败: ${error.message}`);
            return { filePath: null, fileSize: 0 };
        }
    },
    
    // 检查文件是否存在
    getIsExistsFile(filePath) {
        return fs.existsSync(filePath);
    }
};
```

### 5.2 视频处理

```javascript
// src/msg-center/core/utils/fileUtils.js

const FileUtils = {
    // 生成视频缩略图
    async videoImage(videoPath, thumbDir) {
        return new Promise((resolve, reject) => {
            const outputFileName = `thumb_${Date.now()}.jpg`;
            const outputPath = path.join(thumbDir, outputFileName);
            
            ffmpeg(videoPath)
                .screenshots({
                    count: 1,
                    folder: thumbDir,
                    filename: outputFileName,
                    size: '320x240'
                })
                .on('end', () => resolve(outputPath))
                .on('error', reject);
        });
    },
    
    // 获取视频时长
    async getMp4Duration(videoPath) {
        return new Promise((resolve, reject) => {
            ffmpeg.ffprobe(videoPath, (err, metadata) => {
                if (err) {
                    reject(err);
                    return;
                }
                const duration = Math.ceil(metadata.format.duration);
                resolve(duration);
            });
        });
    }
};
```

## 八、调试与排查

### 8.1 日志关键字

| 日志关键字 | 含义 |
|-----------|-----|
| `[接收mqtt任务]` | MQTT收到任务 |
| `friend-chat` | 私聊消息处理 |
| `send-file-img` | 媒体文件处理 |
| `消息过期` | 任务过期被丢弃 |
| `消息重复` | 任务被去重 |
| `[fileDownloadTimeOutResponse]` | 文件下载超时 |

### 8.2 常见问题

| 问题 | 可能原因 | 排查方向 |
|-----|---------|---------|
| 消息未发送 | 任务过期/重复 | 检查`mqttTaskMapLock`日志 |
| 图片发送失败 | 下载超时 | 检查`FileDownloadUtil`日志 |
| 企微消息格式错误 | 消息类型不匹配 | 检查`weChatMsgType`值 |
| 任务无回执 | 逆向未返回 | 检查IPC连接状态 |
