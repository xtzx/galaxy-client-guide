## 二、IPC连接扫描

### 2.1 连接扫描机制

```javascript
// src/msg-center/core/reverse/initIpcTask.js

const InitIpcTask = {
    // 定时扫描间隔（5秒）
    scanInterval: 5000,
    
    run() {
        // 启动定时扫描
        setInterval(() => {
            this.scanIpcConnections();
        }, this.scanInterval);
    },
    
    scanIpcConnections() {
        // 1. 获取所有运行中的微信进程
        const wxProcessList = this.getWxProcessList();
        
        for (const process of wxProcessList) {
            const processId = process.pid;
            const pipeName = `\\\\?\\pipe\\galaxy_${processId}`;
            
            // 2. 检查是否已建立连接
            const existingRegistry = RegistryConfig.getRegistryByKey(processId, 'processId');
            if (existingRegistry && existingRegistry.pipeLineWrapper) {
                continue;  // 已连接，跳过
            }
            
            // 3. 尝试建立IPC连接
            this.tryConnect(pipeName, processId);
        }
    },
    
    tryConnect(pipeName, processId) {
        try {
            // 创建IPC连接
            const pipeLineWrapper = new PipeLineWrapper(pipeName, processId);
            
            // 注册到配置
            const registry = {
                id: generateId(),
                processId: processId,
                pipeLineWrapper: pipeLineWrapper,
                scanTime: Date.now(),
                workWx: this.isWorkWx(processId)  // 判断是微信还是企微
            };
            
            RegistryConfig.addRegistry(registry);
            
            // 发送登录检查
            reverseSendService.sendMessage(null, registry.id, '{"type":"login"}');
            
        } catch (error) {
            logUtil.customLog(`IPC连接失败: ${error.message}`);
        }
    }
};
```

## 三、登录处理

### 3.1 登录服务核心逻辑

```javascript
// src/msg-center/business/convert-service/loginService.js

const LoginService = {
    // 过滤条件：登录消息
    filter(clientRecord) {
        return clientRecord.type === "login" 
            && clientRecord.status === 0 
            && !clientRecord.busy
            && (clientRecord.data || clientRecord.login_user);
    },
    
    operate(clientRecord, pipeLineWrapper) {
        const msgObj = JSON.parse(clientRecord);
        const clientOtherMsgBO = { ...ClientOtherMsgBO, ...msgObj };
        
        // 区分微信和企微
        const loginUser = clientOtherMsgBO.login_user || {};  // 企微
        const loginData = clientOtherMsgBO.data || {};         // 微信
        
        if (IpcUtil.processMap[pipeLineWrapper.processId]) {
            // 企业微信登录
            this.handleLogin(pipeLineWrapper, clientOtherMsgBO, loginUser.id, true);
        } else {
            // 微信登录
            this.handleLogin(pipeLineWrapper, clientOtherMsgBO, loginData.wxid, false);
        }
    },
    
    async handleLogin(pipeLineWrapper, clientOtherMsgBO, ownerWxId, workWxFlag) {
        // 使用锁避免重复登录
        await lock.acquire(`loginLock-${ownerWxId}`, async (done) => {
            try {
                // 1. 获取注册表信息
                const registry = RegistryConfig.getRegistryByKey(pipeLineWrapper.id, "id");
                registry.wxid = "" + ownerWxId;
                
                // 2. 判断是否首次登录
                if (registry.sendToCloudFlag) {
                    // 重复登录：3秒内不重复处理
                    if (Date.now() - lastLoginTimeInfo[ownerWxId] < 3000) {
                        return;
                    }
                    lastLoginTimeInfo[ownerWxId] = Date.now();
                    
                    // 重新发送登录通知
                    await this.reLoginInSend(pipeLineWrapper, clientOtherMsgBO, ownerWxId, workWxFlag);
                } else {
                    // 首次登录
                    lastLoginTimeInfo[ownerWxId] = Date.now();
                    
                    // 3. 检查账号切换
                    const wxid = registry.pipeLineWrapper?.wxid;
                    if (wxid && wxid !== ownerWxId) {
                        logUtil.customLog(`账号交换，先登出旧账号: ${wxid}`);
                        logoutMod.handleLogout(wxid, pipeLineWrapper.id);
                    }
                    
                    // 4. 绑定用户信息
                    this.bindWxId(ownerWxId, clientOtherMsgBO, registry, workWxFlag);
                    
                    // 5. 设置连接状态
                    pipeLineWrapper.wxid = "" + ownerWxId;
                    pipeLineWrapper.available = IpcConstant.AVAILABLE;
                    
                    // 6. 发送登录通知到云端
                    const sendToCloudSuccess = await this.loginInSend(
                        pipeLineWrapper, clientOtherMsgBO, ownerWxId, workWxFlag
                    );
                    
                    // 7. 通知前端刷新
                    frontSendService.sendGetAllConfig();
                    
                    // 8. 标记已登录
                    if (sendToCloudSuccess) {
                        registry.sendToCloudFlag = IpcConstant.SEND_LOGIN_TO_CLOUD;
                        registry.loginTime = Date.now();
                    }
                }
            } catch (error) {
                logUtil.customLog(`[handleLoginError] ${error.message}`);
            } finally {
                done();
            }
        });
    }
};
```

### 3.2 登录通知发送

```javascript
async loginInSend(pipeLineWrapper, clientOtherMsgBO, ownerWxId, workWxFlag) {
    // 给微信一个稳定时间
    await sleep(1000);
    
    // 1. 构建登录消息
    const clientMsg = this.getClientMsg(clientOtherMsgBO, ownerWxId, workWxFlag);
    
    // 2. 调用云端注册接口（重试3次）
    for (let i = 0; i < 3; i++) {
        try {
            let dataRes = await httpFetch({
                url: logicRegisteredCloud,
                data: clientMsg.robotLoginRecord
            });
            logUtil.customLog(`调用云端注册成功`);
            break;
        } catch (err) {
            logUtil.customLog(`调用云端注册失败，重试第${i+1}次`);
        }
    }
    
    // 3. 建立MQTT连接
    await MqttHelper.connectMqtt({ wxId: ownerWxId });
    
    // 4. 发送登录通知到云端
    MqttSendService.sendMessage(ownerWxId, clientMsg);
    
    logUtil.customLog(`[首次登陆通知成功] [wxid-${ownerWxId}]`);
    return true;
}
```

### 3.3 构建登录消息

```javascript
getClientMsg(clientOtherMsgBO, ownerWxId, workWxFlag) {
    const robotLoginRecordBO = { ...RobotLoginRecord };
    robotLoginRecordBO.username = ownerWxId;
    
    if (workWxFlag) {
        // 企业微信登录信息
        const loginUser = clientOtherMsgBO.login_user ?? {};
        robotLoginRecordBO.nickname = loginUser.name;
        robotLoginRecordBO.headImg = loginUser.avator_url;
        robotLoginRecordBO.verifyingName = loginUser.verifying_name;
        robotLoginRecordBO.wxVersion = clientOtherMsgBO.ver;
        robotLoginRecordBO.clientMsgSource = 1;  // 1=企微
        robotLoginRecordBO.corpId = loginUser.corp_id;
        robotLoginRecordBO.phone = loginUser.mobile;
    } else {
        // 微信登录信息
        const wxLoginUser = clientOtherMsgBO.data;
        robotLoginRecordBO.alias = wxLoginUser.username;
        robotLoginRecordBO.headImg = wxLoginUser.headimg;
        robotLoginRecordBO.nickname = wxLoginUser.nickname;
        robotLoginRecordBO.wxVersion = clientOtherMsgBO.ver;
        robotLoginRecordBO.clientMsgSource = 0;  // 0=微信
    }
    
    robotLoginRecordBO.clientVersion = clientVersion;
    robotLoginRecordBO.clientType = 8;  // 8=PC客户端
    
    const clientMsg = { ...ClientMsg };
    clientMsg.username = ownerWxId;
    clientMsg.type = PrismRecordType.ROBOT_LOGIN;
    clientMsg.robotLoginRecord = robotLoginRecordBO;
    
    return clientMsg;
}
```

### 3.4 绑定用户信息

```javascript
bindWxId(wxId, clientOtherMsgBO, registry, workWxFlag) {
    if (!wxId) return;
    
    if (workWxFlag) {
        // 企微用户信息
        const { avator_url, name, verifying_name, id } = clientOtherMsgBO.login_user || {};
        const wxInfo = {
            ...WxInfo,
            apppath: clientOtherMsgBO.apppath,
            headimg: avator_url,
            nickname: name,
            verifyingName: verifying_name,
            wxid: "" + id,
            ver: clientOtherMsgBO.ver,
            pid: clientOtherMsgBO.pid
        };
        registry.wxInfo = wxInfo;
        registry.wxid = "" + id;
    } else {
        // 微信用户信息
        const wxInfo = {
            ...WxInfo,
            apppath: clientOtherMsgBO.apppath,
            headimg: clientOtherMsgBO.data.headimg,
            username: clientOtherMsgBO.data.username,
            nickname: clientOtherMsgBO.data.nickname,
            wxid: clientOtherMsgBO.data.wxid,
            ver: clientOtherMsgBO.ver,
            pid: clientOtherMsgBO.pid
        };
        registry.wxInfo = wxInfo;
    }
}
```

## 四、登出处理

### 4.1 登出触发场景

| 场景 | 触发方式 | 说明 |
|-----|---------|-----|
| 用户主动登出 | 前端发送logout | 用户点击退出 |
| 微信被挤下线 | 逆向推送logout | 其他设备登录 |
| 心跳超时 | HeartBeatTimer | 长时间无心跳 |
| IPC连接断开 | 管道关闭 | 微信进程退出 |
| 账号切换 | loginService | 同一进程换号 |

### 4.3 登出处理模块

```javascript
// src/msg-center/business/convert-service/logoutMod.js

const logoutMod = {
    handleLogout(wxId, registryId) {
        try {
            logUtil.customLog(`[wxid-${wxId}] 开始登出处理`);
            
            // 1. 获取注册信息
            const registry = RegistryConfig.getRegistryByKey(wxId, 'wxid') 
                          || RegistryConfig.getRegistryByKey(registryId, 'id');
            
            if (!registry) {
                logUtil.customLog(`[wxid-${wxId}] 未找到注册信息`);
                return;
            }
            
            // 2. 关闭MQTT连接
            if (registry.mqttClient) {
                MqttClientBase.forceClearMqtt(wxId, registry);
                registry.mqttClient = null;
            }
            
            // 3. 发送登出通知到云端
            this.sendLogoutToCloud(wxId);
            
            // 4. 清理注册信息
            registry.wxid = null;
            registry.wxInfo = null;
            registry.sendToCloudFlag = false;
            registry.loginTime = null;
            
            if (registry.pipeLineWrapper) {
                registry.pipeLineWrapper.wxid = null;
                registry.pipeLineWrapper.available = IpcConstant.NOT_AVAILABLE;
            }
            
            // 5. 通知前端刷新
            frontSendService.sendGetAllConfig();
            
            logUtil.customLog(`[wxid-${wxId}] 登出处理完成`);
            
        } catch (error) {
            logUtil.customLog(`[logoutError] ${error.message}`);
        }
    },
    
    sendLogoutToCloud(wxId) {
        const clientMsg = {
            username: wxId,
            type: PrismRecordType.ROBOT_LOGOUT,
            createTime: Date.now()
        };
        
        MqttSendService.sendMessage(wxId, clientMsg);
    }
};
```

### 5.1 MQTT连接初始化

```javascript
// src/msg-center/core/mq/mqttHelper.js

const MqttHelper = {
    async connectMqtt({ wxId }) {
        logUtil.customLog(`[wxid-${wxId}] 开始建立MQTT连接`);
        
        // 1. 获取MQTT配置
        const mqttConfig = await this.getMqttConfig(wxId);
        
        // 2. 构建连接参数
        const params = {
            endPoint: mqttConfig.endpoint,
            clientId: `galaxy_${wxId}_${Date.now()}`,
            accessKey: mqttConfig.accessKey,
            secretKey: mqttConfig.secretKey,
            filterTopics: [`${mqttConfig.topic}@@@${wxId}/`],
            instanceId: mqttConfig.instanceId,
            connectionOptions: {
                username: mqttConfig.username,
                password: mqttConfig.password,
                keepalive: 60,
                clean: true,
                reconnectPeriod: 0
            },
            qos: 1
        };
        
        // 3. 初始化MQTT客户端
        MqttClientBase.initMqttClient(wxId, params, (wxId, msgObj) => {
            // 消息回调
        });
    },
    
    async getMqttConfig(wxId) {
        // 从云端获取MQTT配置
        const response = await httpFetch({
            url: ApplicationEnv.mqttConfigUrl,
            data: { wxId }
        });
        
        return response.data;
    }
};
```

### 6.2 注册表管理

```javascript
const RegistryConfig = {
    registryList: [],
    
    // 根据key获取注册信息
    getRegistryByKey(value, key) {
        return this.registryList.find(reg => reg[key] === value);
    },
    
    // 获取所有注册信息
    getRegistryList() {
        return this.registryList;
    },
    
    // 添加注册信息
    addRegistry(registry) {
        // 先检查是否已存在
        const existing = this.getRegistryByKey(registry.id, 'id');
        if (existing) {
            Object.assign(existing, registry);
        } else {
            this.registryList.push(registry);
        }
    },
    
    // 移除管道
    removePipe(registryId) {
        const registry = this.getRegistryByKey(registryId, 'id');
        if (registry) {
            // 清理MQTT
            if (registry.mqttClient) {
                MqttClientBase.forceClearMqtt(registry.wxid, registry);
            }
            
            // 清理IPC
            if (registry.pipeLineWrapper) {
                registry.pipeLineWrapper.close();
            }
            
            // 从列表移除
            const index = this.registryList.indexOf(registry);
            if (index > -1) {
                this.registryList.splice(index, 1);
            }
        }
    }
};
```

## 八、调试与排查

### 8.1 日志关键字

| 关键字 | 含义 |
|-------|-----|
| `[appInit] loginService` | 登录流程 |
| `首次登陆` | 首次登录 |
| `重发登陆通知` | 重复登录 |
| `调用云端注册` | 云端注册 |

| `账号交换` | 同进程换号 |

### 8.2 常见问题

| 问题 | 可能原因 | 解决方案 |
|-----|---------|---------|
| 登录后无反应 | IPC连接失败 | 检查逆向服务状态 |
| MQTT连接失败 | 配置错误 | 检查MQTT配置 |
| 重复登录通知 | 锁未生效 | 检查loginLock |
| 登出未清理 | 引用未释放 | 检查registry清理 |
| 前端未刷新 | WebSocket断开 | 检查前端连接 |
