> 好友列表、群成员、标签等数据同步流程

## 一、业务概述

### 1.1 同步场景

| 场景 | 触发时机 | 数据内容 | 同步方向 |
|-----|---------|---------|---------|
| 好友列表同步 | 登录后/定时 | 好友信息+群列表 | 逆向 → 本地 → 云端 |
| 群成员同步 | 收到群消息/定时 | 群成员详情 | 逆向 → 本地 → 云端 |
| 标签同步 | 定时任务 | 标签列表 | 逆向 → 云端 |
| 接收消息同步 | 实时 | 消息内容 | 逆向 → 云端 |

### 2.2 核心处理器 wxUserListResponseMsgHandler

**文件位置**: `src/msg-center/dispatch-center/handle/wxUserListResponseMsgHandler.js`

```javascript
const WxUserListResponseMsgHandler = {
    // 处理类型判断
    isHandler(type, isWorkWx) {
        return [
            GalaxyTaskType.GET_USER_LIST,    // getuserlist
            GalaxyTaskType.USER_INFO_LIST,   // userinfolist
        ].includes(type) && !isWorkWx;
    },

    // 主执行入口
    async execute(jsonObject, wrapper) {
        // 1. 收到总的好友列表（只有 wxid 数组）
        if (GalaxyTaskType.GET_USER_LIST === jsonObject.type) {
            await this.handlerGetUserList(jsonObject, wrapper);
            return;
        }

        // 2. 收到分页的详细信息
        if (GalaxyTaskType.USER_INFO_LIST === jsonObject.type) {
            const arrayData = this.getArrayField(jsonObject);
            const { chatroomIdList, friendIdList } = this.formatChatroomOrUserToList(arrayData);

            // 处理群列表
            await this.handleConversationInfos({ ...jsonObject, chatroomIdList }, wrapper);
            // 处理好友列表
            await this.handleRoomMembers({ ...jsonObject, friendIdList }, wrapper);
        }
    },

    // 区分群和好友的核心逻辑
    formatChatroomOrUserToList(userIdList) {
        const chatroomIdList = [];
        const friendIdList = [];

        userIdList.forEach((item) => {
            if (`${item.wxid}`.indexOf("@chatroom") > -1) {
                chatroomIdList.push(item);  // 群
            } else {
                if (item.nickname) {
                    friendIdList.push(item);  // 好友
                }
            }
        });

        return { chatroomIdList, friendIdList };
    }
};
```

**文件位置**: `src/msg-center/core/cache/wx/wxUserAndChatRoomCache.js`

```javascript
const MsgStatus = {
    START: -1,   // 开始请求
    INIT: 0,     // 初始化
    SECOND: 1,   // 分页请求中
    FINISH: 3    // 完成
};

const wxUserAndChatRoomCache = {
    // 状态标记
    userMsgStatus: new Map(),           // wxid → MsgStatus

    // 初始数据
    userInfoInitData: new Map(),        // wxid → { userIds, channelId }

    // 分页请求管理
    userPageMap: new Map(),             // wxid → { pageNum: [wxids] }
    usersRequestMap: new Map(),         // wxid → { successCount, failCount, retryCount }

    // 数据暂存
    conversationList: new Map(),        // wxid → [群信息数组]
    externalUserList: new Map(),        // wxid → [好友信息数组]

    // 初始化数据
    initData(accountId, userInitData) {
        this.updateUserMsgStatus(accountId, MsgStatus.INIT);
        this.userInfoInitData.set(accountId, userInitData);
        this.conversationList.delete(accountId);
        this.externalUserList.delete(accountId);
        this.userPageMap.set(accountId, {});
        this.usersRequestMap.delete(accountId);
    },

    // 更新群列表（累加）
    updateConversationList(accountId, list = []) {
        const oldList = this.conversationList.get(accountId) || [];
        const newList = oldList.concat(list);
        this.conversationList.set(accountId, newList);
    },

    // 更新好友列表（累加）
    updateExternalUserList(accountId, list = []) {
        const oldList = this.externalUserList.get(accountId) || [];
        const newList = oldList.concat(list);
        this.externalUserList.set(accountId, newList);
    }
};
```

## 三、好友列表同步（旧版处理器）

> 以下是 `friendsListResponseService.js` 的处理逻辑，目前主要使用上面的 `wxUserListResponseMsgHandler`

### 3.1 同步流程

```javascript
// src/msg-center/business/convert-service/friendsListResponseService.js

const FriendsListResponseService = {
    filter(clientRecord) {
        return GalaxyTypeExtEnum.filter(clientRecord, 'FRIENDS_LIST');
    },

    async operate(clientMsg, pipeLineWrapper) {
        const ownerWxId = pipeLineWrapper.wxid;
        const clientFriendsListMsgBO = JSON.parse(clientMsg);

        const chatroomArray = [];      // 群聊列表
        const friendArray = [];         // 好友列表
        const chatroomMemberMap = {};   // 群成员映射

        // 1. 解析返回数据
        const array = clientFriendsListMsgBO.objarray || [];

        for (const data of array) {
            if (!data?.wxid) continue;

            // 区分群聊和好友
            if (data.wxid.includes('@chatroom')) {
                // ======= 处理群聊 =======
                await this.processChatroom(data, ownerWxId, chatroomArray, chatroomMemberMap);
            } else {
                // ======= 处理好友 =======
                await this.processFriend(data, ownerWxId, friendArray);
            }
        }

        // 2. 通知前端
        frontSendService.sendFriendList({
            friendList: friendArray,
            chatroomList: chatroomArray
        }, pipeLineWrapper);

        // 3. 更新本地数据库
        await this.batchUpdateFriend(friendInfoBOList, ownerWxId);

        // 4. 上报云端
        await this.uploadToCloud(ownerWxId, friendInfoBOList, workWxFlag);
    },

    // 处理群聊
    async processChatroom(data, ownerWxId, chatroomArray, chatroomMemberMap) {
        const chatroom = data.wxid;
        const chatroomName = data.nickname;
        const roomOwner = data.ownerwxid;

        // 解析群成员列表（^G分隔）
        const membersUsernameList = data.simplelist?.split("^G").filter(Boolean) || [];

        // 机器人不在群内则跳过
        if (!membersUsernameList.includes(ownerWxId)) {
            logUtil.customLog(`跳过群${chatroom}：机器人不在群内`);
            return;
        }

        chatroomArray.push(data);
        chatroomMemberMap[chatroom] = membersUsernameList;

        // 上传群成员到云端
        await this.uploadChatroomMembers(ownerWxId, chatroom, chatroomName, roomOwner, membersUsernameList);
    },

    // 处理好友
    async processFriend(data, ownerWxId, friendArray) {
        // contacttype 判断是否为好友
        const contactType = parseInt(data.contacttype, 10);

        // contacttype % 2 === 1 表示是好友
        if (contactType % 2 !== 1) {
            await FriendRelationService.deleteFriendRelation(ownerWxId, data.wxid);
            return;
        }

        // 黑名单用户 (contacttype & 8 !== 0)
        if ((contactType & 8) !== 0) {
            return;
        }

        // 官方账号过滤
        if (EXCLUDE_WXID_SET.has(data.wxid)) {
            return;
        }

        friendArray.push(data);
    }
};
```

## 四、群成员同步

### 3.1 触发场景

| 场景 | 触发条件 | 处理方式 |
|-----|---------|---------|
| 收到群消息 | 本地无成员昵称 | 主动请求chatuserinfo |
| 登录后 | 首次获取 | 批量获取所有群成员 |
| 定时任务 | 检查新成员 | 增量同步 |
| 成员变动 | 入群/退群通知 | 更新缓存 |

### 3.2 主动获取群成员

```javascript
// src/msg-center/business/convert-service/recvMsgService.js

const RecvMsgService = {
    // 检查并获取群信息
    checkAndFetchChatroomInfo(ownerWxId, chatroom) {
        // 检查本地缓存
        const chatroomInfo = chatroomInfoService.findByChatroom(chatroom);

        if (chatroomInfo) {
            return; // 已有缓存
        }

        logUtil.customLog(`本地缓存无群信息，异步获取: ${chatroom}`);

        // 获取当前已知成员列表
        const members = chatroomMemberInfoService.findAllMembersByChatroom(chatroom, ownerWxId);
        const memberWxIds = members?.map(m => m.wxid) || [];

        // 创建异步任务获取群成员详情
        memberDetailTaskManager.createChatroomCreateTask(
            ownerWxId,
            chatroom,
            memberWxIds,
            (userInfoMap, chatroomExInfo, context) => {
                this.updateChatroomInfoAndNotifyFront(
                    ownerWxId, chatroom, userInfoMap, chatroomExInfo
                );
            }
        );
    },

    // 主动请求群成员昵称
    requestChatuserinfo(ownerWxId, chatroom) {
        // 防重复请求（30秒内）
        const cacheKey = `chatuserinfo_${ownerWxId}_${chatroom}`;
        const lastRequestTime = chatuserinfoRequestCache.get(cacheKey);

        if (lastRequestTime && (Date.now() - lastRequestTime) < 30000) {
            return;
        }

        chatuserinfoRequestCache.set(cacheKey, Date.now());

        // 发送请求
        memberDetailTaskManager.sendChatuserinfoTask(ownerWxId, chatroom);
    }
};
```

### 3.3 群成员响应处理

```javascript
// src/msg-center/business/convert-service/chatroomMembersDetails.js

const ChatroomMembersDetails = {
    filter(clientRecord) {
        return clientRecord.type === 'chatuserinfo' || clientRecord.type === 'chatuserinfolist';
    },

    async operate(clientMsg, pipeLineWrapper) {
        const msgObj = JSON.parse(clientMsg);
        const ownerWxId = pipeLineWrapper.wxid;
        const chatroom = msgObj.chatroom;

        // 解析成员列表
        const memberList = msgObj.members || [];

        // 更新本地缓存
        for (const member of memberList) {
            const memberPO = {
                wxid: member.wxid,
                chatroomNickname: member.nickname || member.display_name,
                headimg: member.headimg,
                chatroom: chatroom,
                ownerWxId: ownerWxId
            };

            await chatroomMemberInfoService.save(memberPO);
        }

        // 更新群信息
        if (msgObj.owner) {
            const chatroomPO = {
                chatroom: chatroom,
                leaderWxid: msgObj.owner,
                memberCount: memberList.length,
                ownerWxId: ownerWxId
            };
            await chatroomInfoService.save(chatroomPO);
        }

        // 上报云端
        await this.uploadChatroomMembersToCloud(ownerWxId, chatroom, memberList);

        // 通知前端
        frontSendService.sendChatroomMembers(chatroom, memberList, pipeLineWrapper);
    }
};
```

### 4.1 接收消息处理

```javascript
// src/msg-center/business/convert-service/recvMsgsService.js

const RecvMsgsService = {
    filter(clientRecord) {
        return clientRecord.type === 'recvmsgs';
    },

    async operate(clientMsg, pipeLineWrapper) {
        const msgObj = JSON.parse(clientMsg);
        const ownerWxId = pipeLineWrapper.wxid;

        // 批量消息处理
        const msgs = msgObj.msgs || [];

        for (const msg of msgs) {
            await this.processMessage(msg, ownerWxId, pipeLineWrapper);
        }
    },

    async processMessage(msg, ownerWxId, pipeLineWrapper) {
        const { content, from, to, msgtype1, msgtype2, timer, source } = msg;

        // 判断群消息还是私聊
        const isGroupMsg = to?.includes('@chatroom');

        // 构建上报消息
        const recordInfo = {
            username: ownerWxId,
            content: content || '',
            wechatMsgType: msgtype1,
            msgId: JavaClientTools.getMsgId(msg)
        };

        if (isGroupMsg) {
            // 群消息
            recordInfo.chatroom = to;
            recordInfo.fromUsername = from;

            // 获取发送者群昵称
            const memberInfo = await chatroomMemberInfoService.getNicknameByChatroomAndWxid(to, from);
            if (memberInfo?.nickname) {
                recordInfo.fromNickname = memberInfo.nickname;
            }

            // 解析@列表
            if (content?.includes('@') && source) {
                recordInfo.toUsernames = MsgsourceUtil.getUsernamesList(source);
            }

            // 上报云端
            const clientMsgCloud = {
                recordInfo,
                createTime: timer,
                type: PrismRecordType.CHATROOM_MSG,
                username: ownerWxId
            };

            MqttSendService.sendMessage(ownerWxId, clientMsgCloud);

        } else {
            // 私聊消息
            recordInfo.fromUsername = from;
            recordInfo.toUsernames = [to];

            const clientMsgCloud = {
                recordInfo,
                createTime: timer,
                type: PrismRecordType.FRIEND_MSG,
                username: ownerWxId
            };

            MqttSendService.sendMessage(ownerWxId, clientMsgCloud);
        }
    }
};
```
