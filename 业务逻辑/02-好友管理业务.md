### 2.2 核心代码

```javascript
// src/msg-center/business/task-mqtt/mqttFriendPassService.js

const MqttFriendPassService = {
    // 过滤条件：任务类型=104 (通过好友申请)
    filter(serverTask) {
        return serverTask.type === SunTaskType.FRIEND_ACCEPT_REQUEST;
    },
    
    operate(serverTask, wxId) {
        try {
            // 1. 构建客户端任务
            let clientTask = ClientTaskFactory.getClientTask(serverTask, wxId);
            if (!clientTask) {
                return false;
            }
            
            // 2. 发送到逆向
            cloudFlowInBound(null, wxId, JSON.stringify(clientTask));
            
            logUtil.customLog(`friend-pass - clientTask=[${JSON.stringify(clientTask)}]`);
        } catch(error) {
            logUtil.customLog(`[MqttFriendPassServiceError] ${error.message}`);
        }
        return true;
    }
};
```

```javascript
const MqttDeleteFriendService = {
    filter(serverTask) {
        return serverTask.type === SunTaskType.DELETE_FRIEND;
    },
    
    async operate(serverTask, wxId) {
        let { toUsernames, id: taskId } = serverTask;
        
        // 1. 参数校验
        if (!toUsernames || toUsernames.length === 0) {
            this.sendFailResponse(taskId, wxId, "目标用户为空");
            return true;
        }
        
        // 2. 构建删除任务
        let clientTask = ClientTaskFactory.getClientTask(serverTask, wxId);
        
        // 3. 发送到逆向
        cloudFlowInBound(null, wxId, JSON.stringify(clientTask));
        
        return true;
    }
};
```

### 3.2 批量删除

```javascript
// src/msg-center/business/task-mqtt/mqttBatchDeleteFriendService.js

const MqttBatchDeleteFriendService = {
    filter(serverTask) {
        return serverTask.type === SunTaskType.DELETE_FRIEND_BATCH;
    },
    
    async operate(serverTask, wxId) {
        let { toUsernames, id: taskId } = serverTask;
        
        // 批量删除：遍历发送
        for (let i = 0; i < toUsernames.length; i++) {
            const targetWxId = toUsernames[i];
            
            let singleTask = {
                ...serverTask,
                toUsernames: [targetWxId],
                serialNo: i  // 序列号，用于区分
            };
            
            let clientTask = ClientTaskFactory.getClientTask(singleTask, wxId);
            cloudFlowInBound(null, wxId, JSON.stringify(clientTask));
            
            // 间隔1秒，避免频繁操作
            await sleep(1000);
        }
        
        return true;
    }
};
```

```
     │                      │                  │                        │

     │                      │                  │                        │
     │                      │                  │  返回好友列表           │
     │                      │                  │  (含群列表)             │
     │                      │                  │                        │
     │                      │  userlist        │                        │
     │                      │◄─────────────────┤                        │

     │                      │                  │                        │
     │                      ├─────────────────────────────────────────►│
     │                      │                  │                        │
     │                      │                  │   1. 解析好友/群列表   │
     │                      │                  │   2. 过滤非好友        │
     │                      │                  │   3. 存入本地数据库    │
     │                      │                  │   4. 上报云端          │
     │                      │                  │   5. 通知前端          │
```

### 5.2 核心代码

```javascript
// src/msg-center/business/convert-service/friendsListResponseService.js

const FriendsListResponseService = {
    // 过滤条件：type=userlist
    filter(clientRecord) {
        return GalaxyTypeExtEnum.filter(clientRecord, 'FRIENDS_LIST');
    },
    
    async operate(clientMsg, pipeLineWrapper) {
        const clientFriendsListMsgBO = JSON.parse(clientMsg);
        const ownWxId = pipeLineWrapper.wxid;
        
        const chatroomArray = [];   // 群聊列表
        const friendArray = [];      // 好友列表
        const chatroomInfoPOList = [];
        const chatroomMemberMap = {};
        
        // 1. 遍历返回的数据
        const array = clientFriendsListMsgBO.objarray || [];
        for (let data of array) {
            if (!data || !data.wxid) continue;
            
            // 判断是群还是好友
            if (data.wxid.includes('@chatroom')) {
                // ===== 群聊处理 =====
                const chatroom = data.wxid;
                const chatroomName = data.nickname;
                const roomOwner = data.ownerwxid;
                
                // 解析群成员列表
                const membersUsernameList = data.simplelist
                    .split("^G")
                    .filter(arr => !!arr);
                
                // 过滤机器人不在群内的群
                if (!membersUsernameList.includes(ownWxId)) {
                    continue;
                }
                
                chatroomArray.push(data);
                chatroomMemberMap[chatroom] = membersUsernameList;
                
                // 上传群成员到云端
                this.uploadChatroomMembers(ownWxId, chatroom, chatroomName, roomOwner, membersUsernameList);
                
            } else {
                // ===== 好友处理 =====
                // 判断是否为好友（contacttype % 2 === 1）
                if (parseInt(data.contacttype, 10) % 2 !== 1) {
                    // 非好友，删除本地关系
                    await FriendRelationService.deleteFriendRelation(ownWxId, data.wxid);
                    continue;
                }
                
                // 过滤黑名单用户
                if ((parseInt(data.contacttype, 10) & 8) !== 0) {
                    continue;
                }
                
                // 过滤官方账号
                if (EXCLUDE_WXID_SET.has(data.wxid)) {
                    continue;
                }
                
                friendArray.push(data);
            }
        }
        
        // 2. 通知前端
        frontSendService.sendFriendList({ friendList: friendArray, chatroomList: chatroomArray }, pipeLineWrapper);
        
        // 3. 更新本地数据库
        await this.batchUpdateFriend(friendInfoBOList, ownWxId);
        
        // 4. 上传到云端
        await httpFetch({
            url: friendsListUrl,
            data: {
                username: ownWxId,
                friendInfos: friendInfoBOList,
                clientMsgSource: workWxFlag ? 1 : 0
            }
        });
    },
    
    // 批量更新好友信息
    async batchUpdateFriend(friendInfoBOList, ownerWxid) {
        await lock.acquire(`friendUpdateLock-${ownerWxid}`, async (done) => {
            try {
                // 获取现有好友
                const allFriendInfo = await FriendService.findAllByUsername(usernameList);
                
                // 构建更新列表
                const friendList = friendInfoBOList.map(friendInfo => ({
                    alias: friendInfo.alias,
                    username: friendInfo.username,
                    nickname: friendInfo.nickname,
                    headUrl: friendInfo.headUrl,
                    // ...
                }));
                
                // 批量保存/更新
                await FriendService.saveOrUpdate(friendList);
                
                // 更新好友关系
                const friendRelations = friendInfoBOList.map(friendInfo => ({
                    ownerWxid,
                    friendWxid: friendInfo.username,
                    friendRemark: friendInfo.remark
                }));
                
                await FriendRelationService.saveOrUpdate(friendRelations);
                
                done();
            } catch(error) {
                done();
            }
        });
    }
};
```

### 8.3 数据服务

```javascript
// src/msg-center/business/dao-service/friendService.js

const FriendService = {
    // 根据用户名查找
    async findAllByUsername(usernameList) {
        return await Friend.findAll({
            where: { username: usernameList }
        });
    },
    
    // 批量保存或更新
    async saveOrUpdate(friendList) {
        for (const friend of friendList) {
            if (friend.id) {
                await Friend.update(friend, { where: { id: friend.id } });
            } else {
                await Friend.create(friend);
            }
        }
    }
};
```

### 9.2 常见问题

| 问题 | 可能原因 | 解决方案 |
|-----|---------|---------|
| 通过好友失败 | ticket过期 | 检查ticket是否有效 |
| 好友列表不全 | 过滤条件 | 检查contacttype判断逻辑 |
| 备注修改失败 | 权限问题 | 检查是否是好友关系 |
| 标签操作失败 | 标签ID不存在 | 先获取标签列表 |
