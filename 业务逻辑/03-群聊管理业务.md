### 2.2 核心代码

```javascript
// src/msg-center/business/task-mqtt/mqttKickOutService.js

const MqttKickOutService = {
    filter(serverTask) {
        return serverTask.type === SunTaskType.CHATROOM_KICK_OUT;
    },
    
    async operate(serverTask, wxId) {
        logUtil.customLog(`kick-out - serverTask=[${JSON.stringify(serverTask)}]`);
        
        let { id: taskId, toUsernames, chatroom } = serverTask;
        
        // 1. 参数校验
        if (!toUsernames || toUsernames.length === 0) {
            let reason = "用户不存在";
            this.send2Cloud(taskId, wxId, reason);
            return true;
        }
        
        // 2. 获取群成员列表（用于校验）
        let memberSetOld = ChatroomMemberInfoService.getMemberList(chatroom);
        let memberOldMap = new Map();
        for (let member of memberSetOld) {
            memberOldMap.set(member, true);
        }
        
        // 3. 过滤出实际在群内的成员
        let memberListKickOut = toUsernames.filter(wxid => memberOldMap.has(wxid));
        
        // 4. 构建踢人任务
        let clientTask = ClientTaskFactory.getClientTask(serverTask, wxId);
        if (!clientTask) {
            return false;
        }
        
        // 5. 发送到逆向
        logUtil.customLog(`kick-out - clientTask=[${JSON.stringify(clientTask)}]`);
        cloudFlowInBound(null, wxId, JSON.stringify(clientTask));
        
        return true;
    },
    
    // 发送失败结果到云端
    send2Cloud(taskId, wxId, reason) {
        let taskInfoBO = {
            ...TaskInfo,
            reason,
            status: 4,  // 失败
            taskId,
            javaClient: clientVersion
        };
        
        let clientMsgCloud = {
            ...ClientMsg,
            username: wxId,
            taskInfo: taskInfoBO,
            type: PrismRecordType.TASK_RESULT_MSG
        };
        
        MqttSendService.sendMessage(wxId, clientMsgCloud);
    }
};
```

// src/msg-center/business/dao-service/chatroomMemberinfoService.js

const ChatroomMemberInfoService = {
    // 获取群成员列表
    getMemberList(chatroom) {
        const members = ChatroomMemberInfo.findAll({
            where: { chatroom: chatroom }
        });
        return members.map(m => m.wxid);
    },
    
    // 根据群和wxid获取昵称
    async getNicknameByChatroomAndWxid(chatroom, wxid) {
        return await ChatroomMemberInfo.findOne({
            where: { chatroom, wxid }
        });
    },
    
    // 保存群成员
    async save(memberPO) {
        const existing = await ChatroomMemberInfo.findOne({
            where: { chatroom: memberPO.chatroom, wxid: memberPO.wxid }
        });
        
        if (existing) {
            await ChatroomMemberInfo.update(memberPO, { where: { id: existing.id } });
        } else {
            await ChatroomMemberInfo.create(memberPO);
        }
    },
    
    // 查询群所有成员
    findAllMembersByChatroom(chatroom, ownerWxId) {
        return ChatroomMemberInfo.findAll({
            where: { chatroom, ownerWxId }
        });
    }
};

## 十、逆向返回处理

### 10.1 踢人结果处理

```javascript
// src/msg-center/business/convert-response/kickOutTaskResponse.js

const KickOutTaskResponse = {
    filter(clientRecord) {
        return clientRecord.type === 'kickout' || clientRecord.type === 'delchatmenber';
    },
    
    async operate(clientMsg, pipeLineWrapper) {
        const msgObj = JSON.parse(clientMsg);
        const { taskId, status, chatroom, wxids } = msgObj;
        
        // 上报任务结果
        const taskInfo = {
            taskId: taskId,
            status: status === 0 ? 0 : 4,  // 0成功，其他失败
            reason: status === 0 ? '' : '踢人失败'
        };
        
        const clientMsgCloud = {
            username: pipeLineWrapper.wxid,
            taskInfo: taskInfo,
            type: PrismRecordType.TASK_RESULT_MSG
        };
        
        MqttSendService.sendMessage(pipeLineWrapper.wxid, clientMsgCloud);
        
        // 更新本地缓存：移除被踢成员
        if (status === 0 && wxids) {
            for (const wxid of wxids) {
                await ChatroomMemberInfoService.delete(chatroom, wxid);
            }
        }
    }
};
```

### 11.2 常见问题

| 问题 | 可能原因 | 解决方案 |
|-----|---------|---------|
| 踢人失败 | 不是群主/管理员 | 检查机器人权限 |
| 邀请失败 | 用户设置了验证 | 使用发送邀请链接方式 |
| 修改群名失败 | 群主禁止修改 | 需要群主权限 |
| 群公告失败 | 只有管理员可修改 | 检查管理员权限 |
| 创建群失败 | 人数不足 | 至少需要2人 |
