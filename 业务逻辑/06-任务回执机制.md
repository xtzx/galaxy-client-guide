# 任务回执机制详解

> 三段式消息追踪机制

---

## 一、机制概述

### 1.1 为什么需要三段式

微信/企微的消息发送是**异步**的，发送一条消息后，会有多个阶段的回调：

```
发送请求 → 微信服务端接收 → 对方收到 → 发送成功/失败
```

为了准确追踪任务状态，项目设计了"三段式消息机制"来关联这些回调消息。

### 1.2 三段式消息流程

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                        三段式消息机制                                         │
└─────────────────────────────────────────────────────────────────────────────┘

┌─────────────────┐    ┌─────────────────┐    ┌─────────────────┐
│    第一条消息    │    │    第二条消息    │    │    第三条消息    │
│   sendmessage   │───►│     recvmsg     │───►│      oplog      │
│                 │    │                 │    │                 │
│   发送请求确认   │    │    消息发出     │    │    最终状态      │
│   (任务已下发)   │    │   (本地确认)    │    │  (成功/失败)    │
└─────────────────┘    └─────────────────┘    └─────────────────┘
        │                      │                      │
        ▼                      ▼                      ▼
   记录 taskId            关联 taskId             汇总上报云端
   存入缓存               校验消息匹配            清理缓存

时间窗口：通常在 1-5 秒内完成三条消息的接收
```

### 1.3 消息类型映射

| 阶段 | 微信类型 | 企微类型 | 说明 |
|-----|---------|---------|-----|
| 第一条 | `sendmessage` | `sendmsg_*` | 发送请求 |
| 第二条 | `recvmsg` | `recvmsg` | 本地确认 |
| 第三条 | `oplog` | `oplog` | 最终结果 |

## 二、核心实现

### 2.1 消息处理基类

```javascript
// src/msg-center/dispatch-center/handle/msgHandleBase.js

const MsgHandlerCenterBase = {
    // 消息节点缓存 { key: MsgResNode }
    msgResNodeMap: {},
    
    // 任务ID缓存 { key: taskId }
    taskIdMap: {},
    
    // 缓存过期时间（毫秒）
    expireTime: 10000,
    
    // 处理出站消息
    async outBoundMsg(jsonObj, wrapper, sendType) {
        // 1. 消息转换处理
        let jsonObject = this.messageHandler(jsonObj, wrapper, sendType);
        
        // 2. 响应消息处理
        jsonObject = await this.responseMsgHandler(jsonObject, sendType, wrapper);
        
        // 3. 发送到对应目标
        this.send(JSON.stringify(jsonObject), null, sendType, jsonObject, wrapper);
    },
    
    // 消息处理核心逻辑
    messageHandler(jsonObj, wrapper, sendType) {
        const wxId = wrapper.wxid;
        const taskId = jsonObj.taskId;
        
        // 生成缓存key
        let key = this.getWxMsgKey(jsonObj, wxId);
        let hashKey = this.getWxHashKey(jsonObj, wxId);
        
        // 缓存taskId
        if (taskId) {
            this.taskIdMap[key] = taskId;
        }
        
        // 从缓存获取消息节点
        let msgResNode = this.get(hashKey) || this.get(key);
        
        // 判断是第几条消息
        if (this.isFirstMsg(msgResNode, sendType)) {
            // 第一条消息：创建节点
            msgResNode = { first: this.getDefaultMsgResNode(jsonObj) };
            this.add(msgResNode, key, jsonObj);
            if (hashKey) {
                this.add(msgResNode, hashKey, jsonObj);
            }
            return jsonObj;
            
        } else if (this.isSecondMsg(msgResNode, sendType)) {
            // 第二条消息：关联节点
            msgResNode.second = this.getDefaultMsgResNode(jsonObj);
            // 切换为使用hash作为key
            if (hashKey) {
                this.add(msgResNode, hashKey, jsonObj);
            }
            this.remove(key);
            return jsonObj;
            
        } else if (this.isThirdMsg(msgResNode, sendType, jsonObj)) {
            // 第三条消息：完成处理
            msgResNode.third = jsonObj;
            return this.thirdMsgHandler(msgResNode, hashKey);
        }
        
        // 异常情况处理
        return this.errorOrder(msgResNode, sendType, jsonObj, wrapper.workWx, key, wxId);
    }
};
```

// 微信消息Key
getWxMsgKey(jsonObj, wxId) {
    const timer = jsonObj.timer;
    const toUser = jsonObj.to || jsonObj?.data?.to;
    const key = `${wxId}_${toUser}_${timer}`;
    return key;
}

// 微信Hash Key
getWxHashKey(jsonObj, wxId) {
    const hashKey = jsonObj.hash || jsonObj?.data?.hash;
    return hashKey;
}

// 企微Key
getQywxKey(jsonObj, wxId) {
    const clientId = jsonObj.client_id;
    return `${wxId}_${clientId}`;
}

// 微信4.0关联Key（新版）
getWx4AssociationKey(jsonObj, wxId) {
    const from = jsonObj.from || wxId;
    const to = jsonObj.to || '';
    let msgType = this.normalizeMsgType(jsonObj.wechatMsgType, jsonObj);
    
    // 根据消息类型选择关联字段
    let associationValue = '';
    switch(parseInt(msgType)) {
        case 3:  // 图片
        case 43: // 视频
        case 49: // 文件
            // 媒体文件使用文件名关联
            associationValue = this.extractFileName(jsonObj, msgType);
            break;
        case 1:  // 文本
        default:
            // 文本使用内容关联
            associationValue = jsonObj.content || '';
            break;
    }
    
    return `wx4-${wxId}-${from}-${to}-${msgType}-${associationValue}`;
}

## 三、微信消息处理

### 3.1 微信消息处理器

```javascript
// src/msg-center/dispatch-center/handle/wxMsgHandle.js

const WxMsgHandlerCenter = {
    ...MsgHandlerCenterBase,
    
    msgResNodeMap: {},
    
    // 第一条消息类型集合
    FIRST_CALLBACKS: new Set([
        'sendmessage',     // 发送消息
        'addfriend',       // 添加好友
        'groupname',       // 修改群名
        // ...
    ]),
    
    // 第二条消息类型集合
    SECOND_CALLBACKS: new Set([
        'recvmsg',         // 接收消息
    ]),
    
    // 第三条消息类型集合
    THIRD_CALLBACKS: new Set([
        'oplog',           // 操作日志
    ]),
    
    // 判断是否是第一条消息
    isFirstMsg(node, sendType) {
        return !node && this.FIRST_CALLBACKS.has(sendType);
    },
    
    // 判断是否是第二条消息
    isSecondMsg(node, sendType) {
        return node && node.first && !node.second && this.SECOND_CALLBACKS.has(sendType);
    },
    
    // 判断是否是第三条消息
    isThirdMsg(node, sendType, jsonObj) {
        if (!node || !node.first) return false;
        
        // oplog类型
        if (this.THIRD_CALLBACKS.has(sendType)) {
            return true;
        }
        
        // 错误消息也作为第三条
        if (jsonObj.status !== 0) {
            return true;
        }
        
        return false;
    },
    
    // 是否需要重置节点
    isReset(sendType, node) {
        // 收到新的第一条消息时，重置旧节点
        if (node && node.first && this.FIRST_CALLBACKS.has(sendType)) {
            const elapsed = Date.now() - node.first.createTime;
            if (elapsed > this.expireTime) {
                return true;
            }
        }
        return false;
    }
};
```

### 3.2 微信4.0特殊处理

微信4.0版本只有两条消息（sendmessage + recvmsg），没有oplog：

```javascript
// 微信4.0：检查recvmsg是否完成
isWx4RecvmsgComplete(msgResNode, sendType, jsonObj) {
    // 只处理recvmsg
    if (sendType !== 'recvmsg') {
        return false;
    }
    
    // 检查是否有第一条消息
    if (!msgResNode || !msgResNode.first || !msgResNode.first.taskId) {
        return false;
    }
    
    // recvmsg的status=2表示成功
    if (jsonObj.status === 2) {
        jsonObj.status = 0;  // 转换为标准成功状态
        return true;
    }
    
    // 其他status表示失败
    jsonObj.status = 1;
    return true;
}
```

## 五、第三条消息处理

### 5.1 完成处理并上报

```javascript
// 第三条消息结果处理
thirdMsgHandler(msgResNode, msgKey) {
    msgResNode.thirdTime = (msgResNode.thirdTime || 0) + 1;
    const { third } = msgResNode;
    
    // 1. 获取taskId
    let taskId = this.getTaskId(msgResNode);
    
    // 2. 缓存msgId到taskId的映射
    if (third.client_id) {
        MsgTaskInfoCache.msgIdToTaskIdInfo[third.client_id] = taskId;
    }
    
    // 3. 设置taskId到第三条消息
    third.taskId = taskId;
    
    // 4. 更新任务状态缓存
    const galaxyTaskStatus = GalaxyTaskCache.GALAXY_TASK_STATUS_MAP[taskId];
    if (galaxyTaskStatus) {
        galaxyTaskStatus.firstMessageStatus = 'RECEIVED';
        galaxyTaskStatus.secondMessageStatus = 'RECEIVED';
        galaxyTaskStatus.thirdMessageStatus = 'RECEIVED';
    }
    
    // 5. 清理缓存
    if (msgResNode.thirdTime >= 2 || this.get(msgKey)) {
        this.remove(msgKey);
    }
    
    // 6. 清理任务消息类型映射
    GalaxyTaskCache.removeTaskMsgTypeMapToTaskId(taskId);
    
    logUtil.customLog(`[第三条消息] taskId：${taskId} msgKey: ${msgKey}`);
    return third;
}
```

### 5.2 获取taskId

```javascript
getTaskId(node) {
    let taskId = null;
    
    if (!node) return null;
    
    // 优先从第一条消息获取
    if (node.first) {
        taskId = node.first.taskId;
    }
    
    // 第一条没有，尝试从第二条获取
    if (!taskId && node.second) {
        taskId = node.second.taskId;
    }
    
    return taskId;
}
```

## 六、缓存管理

### 6.1 缓存添加

```javascript
add(msgResNode, key, jsonObj, wxId) {
    // 设置过期时间
    msgResNode.expireTime = Date.now() + this.expireTime;
    
    // 视频消息特殊处理：使用数组存储
    const videoKey = this.getVideoMsgKey(jsonObj, wxId);
    if (videoKey) {
        if (!this.msgResNodeMap[videoKey]) {
            this.msgResNodeMap[videoKey] = [];
        }
        this.msgResNodeMap[videoKey].push(msgResNode);
    } else {
        this.msgResNodeMap[key] = msgResNode;
    }
    
    // 启动清理定时器
    this.startCleanupTask();
}
```

### 6.2 缓存获取

```javascript
get(key, jsonObj, wxId) {
    // 检查视频消息
    const videoKey = jsonObj ? this.getVideoMsgKey(jsonObj, wxId) : null;
    const isVideoMsg = videoKey && videoKey.startsWith('video_');
    
    const value = this.msgResNodeMap[videoKey || key];
    if (!value) return null;
    
    // 视频消息返回数组中第一个有效节点
    if (isVideoMsg && Array.isArray(value)) {
        const now = Date.now();
        const validNodes = value.filter(node => node.expireTime >= now);
        if (validNodes.length === 0) {
            delete this.msgResNodeMap[key];
            return null;
        }
        return validNodes[0];
    }
    
    // 检查过期
    if (value.expireTime < Date.now()) {
        delete this.msgResNodeMap[key];
        return null;
    }
    
    return value;
}
```

### 6.3 定期清理

```javascript
startCleanupTask() {
    if (this.cleanCacheIntervalId) return;
    
    // 每10秒清理一次
    this.cleanCacheIntervalId = setInterval(() => {
        this.removeExpireNode();
    }, 10000);
}

removeExpireNode() {
    const now = Date.now();
    const keys = Object.keys(this.msgResNodeMap);
    
    keys.forEach((key) => {
        const value = this.msgResNodeMap[key];
        
        if (key.startsWith('video_') && Array.isArray(value)) {
            // 视频消息：清理过期节点
            const validNodes = value.filter(node => node.expireTime >= now);
            if (validNodes.length === 0) {
                delete this.msgResNodeMap[key];
            } else {
                this.msgResNodeMap[key] = validNodes;
            }
        } else {
            // 普通消息：直接删除过期
            if (value.expireTime < now) {
                delete this.msgResNodeMap[key];
            }
        }
    });
}
```

### 7.1 GalaxyTaskCache

```javascript
// src/msg-center/core/cache/galaxyTaskCache.js

const GalaxyTaskCache = {
    // 任务状态Map { taskId: GalaxyTaskStatus }
    GALAXY_TASK_STATUS_MAP: {},
    
    // 任务消息类型Map { wxId_toUsername: [taskId-msgType-sendTime-content] }
    taskMsgTypeMap: {},
    
    // 设置任务消息类型
    setTaskMsgTypeMap(jsonObject) {
        const taskId = jsonObject.taskId;
        const ownerWxId = jsonObject.ownerWxId;
        const toUsername = jsonObject.data?.to;
        const wechatMsgType = jsonObject.data?.wechatMsgType || jsonObject.wechatMsgType;
        const content = jsonObject.data?.content || jsonObject.content || '';
        
        const key = `${ownerWxId}_${toUsername}`;
        if (!this.taskMsgTypeMap[key]) {
            this.taskMsgTypeMap[key] = [];
        }
        
        // 格式：taskId-msgType-sendTime-content
        const value = `${taskId}-${wechatMsgType}-${Date.now()}-${content.substring(0, 100)}`;
        this.taskMsgTypeMap[key].push(value);
    },
    
    // 获取任务消息类型
    getTaskMsgTypeMap(ownerWxId, toUsername) {
        const key = `${ownerWxId}_${toUsername}`;
        return this.taskMsgTypeMap[key] || [];
    },
    
    // 根据taskId移除
    removeTaskMsgTypeMapToTaskId(taskId) {
        for (const key of Object.keys(this.taskMsgTypeMap)) {
            this.taskMsgTypeMap[key] = this.taskMsgTypeMap[key]
                .filter(v => !v.startsWith(`${taskId}-`));
        }
    }
};
```

## 八、内容匹配策略

### 8.1 内容匹配核心逻辑

当第二条或第三条消息没有直接的taskId时，需要通过内容匹配来找到对应的任务：

```javascript
getCacheTaskId(msgObj, isWorkWx, wxId) {
    let ownerWxId = wxId;
    let toUsername = this.wxGetToUsername(msgObj);
    const wechatMsgType = getMsgType(msgObj);
    
    // 获取当前消息内容
    const currentContent = this.extractMessageContent(msgObj);
    const processedContent = this.processContentByType(currentContent, wechatMsgType, msgObj);
    
    // 获取缓存的任务列表
    const taskMsgTypeMap = GalaxyTaskCache.getTaskMsgTypeMap(ownerWxId, toUsername);
    
    let taskId = '';
    
    // 遍历匹配
    taskMsgTypeMap.find((value) => {
        const [taskIdStr, msgType, sendTime, cachedContent] = value.split('-');
        
        // 类型匹配
        if (wechatMsgType && `${wechatMsgType}` === msgType) {
            // 内容匹配
            if (this.isContentMatch(processedContent, cachedContent, wechatMsgType)) {
                taskId = taskIdStr;
                return true;
            }
        }
        return false;
    });
    
    return taskId;
}
```

### 8.2 不同类型的匹配策略

```javascript
isContentMatch(currentContent, cachedContent, wechatMsgType) {
    switch (wechatMsgType) {
        case '1': // 文本消息
            return this.isTextContentMatch(currentContent, cachedContent);
            
        case '3': // 图片消息
        case '43': // 视频消息
        case '49': // 文件消息
            return this.isMediaContentMatch(currentContent, cachedContent);
            
        case '5': // 链接消息
            return this.isLinkContentMatch(currentContent, cachedContent);
            
        default:
            return this.isFuzzyContentMatch(currentContent, cachedContent);
    }
}

// 文本匹配
isTextContentMatch(current, cached) {
    if (current === cached) return true;
    if (current.toLowerCase() === cached.toLowerCase()) return true;
    if (current.includes(cached) || cached.includes(current)) return true;
    return false;
}

// 媒体匹配（提取文件名比较）
isMediaContentMatch(current, cached) {
    if (current === cached) return true;
    
    // 比较文件名
    const fileName1 = current.split('/').pop() || current;
    const fileName2 = cached.split('/').pop() || cached;
    return fileName1 === fileName2;
}
```

## 九、异常处理

### 9.1 乱序消息处理

```javascript
errorOrder(node, sendType, jsonObject, isWorkWx, key, wxId) {
    logUtil.customLog(`消息乱序: ${sendType} node:${JSON.stringify(node)}`);
    
    // 尝试从taskIdMap获取
    let taskId = this.getTaskId(node);
    if (taskId) {
        delete this.taskIdMap[key];
        return this.thirdMsgHandler({ third: jsonObject, first: { taskId } }, key);
    }
    
    // 尝试通过内容匹配
    taskId = this.getCacheTaskId(jsonObject, isWorkWx, wxId);
    if (taskId) {
        return this.thirdMsgHandler({ third: jsonObject, first: { taskId } }, key);
    }
    
    // 无法匹配，返回原消息
    logUtil.customLog(`[没有匹配到taskId] ${JSON.stringify(jsonObject)}`);
    return jsonObject;
}
```

checkStatus(node, wrapper) {

## 十、调试与排查

### 10.1 日志关键字

| 关键字 | 含义 |
|-------|-----|
| `[msgConvert]` | 消息转换处理 |
| `第一条消息` | 收到sendmessage |
| `第二条消息` | 收到recvmsg |
| `第三条消息` | 收到oplog |

| `没有匹配到taskId` | 匹配失败 |
| `内容匹配成功` | 内容匹配 |

### 10.2 常见问题

| 问题 | 可能原因 | 解决方案 |
|-----|---------|---------|
| taskId匹配不上 | 消息乱序 | 检查缓存key生成逻辑 |
| 缓存过多 | 未及时清理 | 检查过期时间配置 |
| 回执丢失 | 第三条消息未收到 | 检查逆向是否推送 |
| 重复上报 | 未正确清理缓存 | 检查thirdMsgHandler |
