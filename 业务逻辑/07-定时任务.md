## 一、定时任务概览

### 1.1 任务列表

| 任务状态检查 | 60秒 | `GalaxyTaskStatusTimer.js` | 检查超时任务 |

## 二、心跳检测

### 2.1 核心逻辑

```javascript
// src/msg-center/business/timer/HeartBeatTimer.js

const HeartBeatTimer = {
    heartBeat() {
        try {
            let removeList = [];
            const registryList = RegistryConfig.getRegistryList();
            
            // 无机器人时通知前端
            if (!registryList?.length) {
                frontSendService.sendGetAllConfig();
                return;
            }
            
            for (const registry of registryList) {
                const currentTime = Math.floor(Date.now() / 1000);
                const lastPongTime = Math.floor(registry.pipeLineWrapper?.lastPongTime / 1000);
                
                if (lastPongTime) {
                    // 检查未登录的机器人
                    if (!registry.wxid) {
                        logUtil.customLog(`发现未登录机器人，触发登录: ${registry.id}`);
                        if (registry.pipeLineWrapper) {
                            reverseSendService.sendMessage(null, registry.id, '{"type":"login"}');
                        }
                    }
                    
                    // 检查心跳超时
                    const timeInterval = currentTime - lastPongTime;
                    logUtil.customLog(`[HeartBeat] [wxid-${registry.wxid}] 心跳间隔: ${timeInterval}秒`);
                    
                    if (timeInterval > heartBeatOverTime) {
                        // 检查是否仍在读取消息
                        const lastReadTime = Math.floor(registry.pipeLineWrapper.lastReadTime / 1000);
                        if (lastReadTime && currentTime - lastReadTime <= heartBeatOverTime) {
                            logUtil.customLog(`心跳超时但仍在读取消息，跳过`);
                            continue;
                        }
                        
                        // 标记为不可用
                        registry.pipeLineWrapper.available = IpcConstant.NOT_AVAILABLE;
                        logUtil.customLog(`[wxid-${registry.wxid}] 心跳超时${timeInterval}秒，移除连接`);
                        
                        removeList.push(registry.id);
                        RegistryConfig.removePipe(registry.id);
                        
                        // 发送超时通知
                        notify.onPongOvertime(registry, timeInterval);
                    }
                }
                // 建立连接后一直没有pong
                else if (!lastPongTime && registry.scanTime) {
                    const timeInterval = currentTime - Math.floor(registry.scanTime / 1000);
                    if (timeInterval > heartBeatOverTime) {
                        registry.pipeLineWrapper.available = IpcConstant.NOT_AVAILABLE;
                        logUtil.customLog(`[wxid-${registry.wxid}] 连接后${timeInterval}秒无心跳，移除`);
                        
                        removeList.push(registry.id);
                        RegistryConfig.removePipe(registry.id);
                    }
                }
                // 没有扫描时间记录
                else if (!registry.scanTime) {
                    logUtil.customLog(`没有扫描时间记录，移除: ${registry.id}`);
                    removeList.push(registry.id);
                    RegistryConfig.removePipe(registry.id);
                }
            }
            
            // 通知前端刷新
            if (removeList?.length) {
                frontSendService.sendGetAllConfig();
            }
            
        } catch(error) {
            logUtil.customLog(`[HeartBeatTimer] 错误: ${error.message}`);
        }
    }
};
```

```javascript
const PingTimer = {
    ping() {
        try {
            const registryList = RegistryConfig.getRegistryList();
            
            for (const registry of registryList) {
                if (!registry.pipeLineWrapper) continue;
                if (!registry.pipeLineWrapper.available) continue;

                    registry.wxid, 
                    registry.id, 
                    JSON.stringify(pingMsg)
                );
                
                logUtil.customLog(`[wxid-${registry.wxid}] 发送Ping`);
            }
        } catch(error) {
            logUtil.customLog(`[PingTimer] 错误: ${error.message}`);
        }
    }
};
```

## 四、任务状态检查

### 4.1 检查超时任务

```javascript
// src/msg-center/business/timer/GalaxyTaskStatusTimer.js

const GalaxyTaskStatusTimer = {
    // 任务超时时间（毫秒）
    taskTimeout: 60000,
    
    checkTaskStatus() {
        try {
            const now = Date.now();
            const statusMap = GalaxyTaskCache.GALAXY_TASK_STATUS_MAP;
            
            for (const taskId of Object.keys(statusMap)) {
                const taskStatus = statusMap[taskId];
                
                // 检查是否超时
                if (now - taskStatus.createTime > this.taskTimeout) {
                    // 检查是否有未收到的消息
                    if (taskStatus.thirdMessageStatus === 'NOT_RECEIVE') {
                        logUtil.customLog(`[任务超时] taskId: ${taskId}, 第三条消息未收到`);
                        
                        // 上报超时状态
                        this.reportTimeoutTask(taskId, taskStatus.username);
                    }
                    
                    // 清理缓存
                    delete statusMap[taskId];
                }
            }
        } catch(error) {
            logUtil.customLog(`[GalaxyTaskStatusTimer] 错误: ${error.message}`);
        }
    },
    
    reportTimeoutTask(taskId, wxId) {
        const taskInfo = {
            taskId: taskId,
            status: 4,  // 失败
            reason: '任务超时未收到回执'
        };
        
        const clientMsg = {
            username: wxId,
            taskInfo: taskInfo,
            type: PrismRecordType.TASK_RESULT_MSG
        };
        
        MqttSendService.sendMessage(wxId, clientMsg);
    }
};
```

```javascript
// src/msg-center/business/timer/GetCloudConfigTimer.js

const GetCloudConfigTimer = {
    async getConfig() {
        try {
            const registryList = RegistryConfig.getRegistryList();
            
            for (const registry of registryList) {
                if (!registry.wxid) continue;
                
                // 获取机器人配置
                const response = await httpFetch({
                    url: ApplicationEnv.getRobotConfigUrl,
                    data: { username: registry.wxid }
                });
                
                if (response.code === 200 && response.data) {
                    // 更新本地配置
                    registry.robotConfig = response.data;
                    
                    logUtil.customLog(`[wxid-${registry.wxid}] 配置同步成功`);
                }
            }
        } catch(error) {
            logUtil.customLog(`[GetCloudConfigTimer] 错误: ${error.message}`);
        }
    }
};
```

```javascript
const GetOssAccessKeyTimer = {
    async refresh() {
        try {
            // 获取OSS临时凭证
            const response = await httpFetch({
                url: ApplicationEnv.getOssCredentialsUrl,
                data: {}
            });
            
            if (response.code === 200 && response.data) {
                // 更新全局OSS配置
                global.ossCredentials = {
                    accessKeyId: response.data.accessKeyId,
                    accessKeySecret: response.data.accessKeySecret,
                    securityToken: response.data.securityToken,
                    expiration: response.data.expiration
                };
                
                logUtil.customLog(`OSS凭证刷新成功，过期时间: ${response.data.expiration}`);
            }
        } catch(error) {
            logUtil.customLog(`[GetOssAccessKeyTimer] 错误: ${error.message}`);
        }
    }
};
```

```javascript
const ReportMonitorInfoTimer = {
    async report() {
        try {
            const registryList = RegistryConfig.getRegistryList();
            
            // 收集系统信息
            const systemInfo = {
                cpuUsage: process.cpuUsage(),
                memoryUsage: process.memoryUsage(),
                uptime: process.uptime(),
                platform: process.platform,
                nodeVersion: process.version
            };
            
            // 收集机器人状态
            const robotStatusList = registryList.map(registry => ({
                wxId: registry.wxid,
                connected: !!registry.pipeLineWrapper?.available,
                mqttConnected: !!registry.mqttClient?.connected,
                loginTime: registry.loginTime,
                workWx: registry.workWx
            }));
            
            // 上报
            await httpFetch({
                url: ApplicationEnv.reportMonitorUrl,
                data: {
                    systemInfo,
                    robotStatusList,
                    timestamp: Date.now()
                }
            });
            
        } catch(error) {
            logUtil.customLog(`[ReportMonitorInfoTimer] 错误: ${error.message}`);
        }
    }
};
```

## 八、补偿任务处理

### 8.1 处理失败任务

```javascript
// src/msg-center/business/timer/ProcessMakeUpTaskTimer.js

const ProcessMakeUpTaskTimer = {
    async process() {
        try {
            const makeUpTasks = MqttMakeUpManager.getMakeUpTasks();
            
            for (const task of makeUpTasks) {
                const { wxId, taskId, retryCount, lastTryTime } = task;
                
                // 检查重试次数
                if (retryCount >= 3) {
                    logUtil.customLog(`任务${taskId}重试次数超限，放弃`);
                    MqttMakeUpManager.removeTask(taskId);
                    continue;
                }
                
                // 检查重试间隔
                if (Date.now() - lastTryTime < 30000) {
                    continue;  // 30秒内不重试
                }
                
                // 获取机器人
                const registry = RegistryConfig.getRegistryByKey(wxId, 'wxid');
                if (!registry?.pipeLineWrapper?.available) {
                    continue;
                }
                
                // 重新执行任务
                logUtil.customLog(`重试任务: ${taskId}, 第${retryCount + 1}次`);
                
                task.retryCount++;
                task.lastTryTime = Date.now();
                
                // 发送任务
                cloudFlowInBound(null, wxId, JSON.stringify(task.taskData));
            }
        } catch(error) {
            logUtil.customLog(`[ProcessMakeUpTaskTimer] 错误: ${error.message}`);
        }
    }
};
```

## 九、前端消息延迟发送

### 9.1 延迟发送机制

```javascript
// src/msg-center/business/timer/delaySendFrontmsgTimer.js

const DelaySendFrontMsgTimer = {
    // 消息队列
    messageQueue: {},
    
    // 添加延迟消息
    addMessage(wxId, message, delay = 1000) {
        const key = `${wxId}_${message.type}`;
        
        // 清除之前的定时器
        if (this.messageQueue[key]) {
            clearTimeout(this.messageQueue[key].timer);
        }
        
        // 设置新定时器
        this.messageQueue[key] = {
            message,
            timer: setTimeout(() => {
                this.sendMessage(wxId, message);
                delete this.messageQueue[key];
            }, delay)
        };
    },
    
    sendMessage(wxId, message) {
        const registry = RegistryConfig.getRegistryByKey(wxId, 'wxid');
        if (registry) {
            frontSendService.sendToFront(JSON.stringify(message), registry);
        }
    },
    
    // 群信息更新通知（合并延迟发送）
    chatroomUpdateSendFront(pipeLineWrapper, ownerWxId, chatroomUpdateVo) {
        this.addMessage(ownerWxId, chatroomUpdateVo, 2000);
    }
};
```

## 十、调试与排查

### 10.1 日志关键字

| 关键字 | 含义 |
|-------|-----|
| `[HeartBeat]` | 心跳检测 |
| `发送Ping` | Ping发送 |
| `心跳超时` | 连接超时 |
| `任务超时` | 任务未完成 |
| `配置同步成功` | 配置获取 |
| `OSS凭证刷新` | 密钥刷新 |

### 10.2 常见问题

| 问题 | 可能原因 | 解决方案 |
|-----|---------|---------|
| 频繁掉线 | 心跳超时阈值太短 | 调整heartBeatOverTime |
| 任务超时过多 | 逆向处理慢 | 检查逆向性能 |
| 配置不同步 | 网络问题 | 检查云端接口 |
| 内存增长 | 定时器泄漏 | 检查clearInterval |
