# 前端交互业务详解

> Web前端与主进程的交互场景

---

## 一、业务概述

### 1.1 交互场景分类

前端（通宝管理后台）与主进程通过 **WebSocket** 进行实时通信，主要业务场景：

| 场景 | 触发方 | 核心文件 | 说明 |
|------|--------|---------|------|
| **获取机器人状态** | 前端 | `getAllconfigTask.js` | 获取所有微信连接状态 |
| **转发任务** | 前端 | `forwardTask.js` | 将前端操作转发给逆向 |
| **前端登录** | 前端 | `frontLoginTask.js` | 前端上线通知 |
| **前端登出** | 前端 | `FrontLogoutTask.js` | 前端下线通知 |
| **文件上传** | 前端 | `UploadTask.js` | 上传文件到OSS |
| **获取MQTT状态** | 前端 | `GetMqttConnectionStatusTask.js` | 获取MQTT连接状态 |
| **杀掉应用** | 前端 | `killAppTask.js` | 停止逆向服务 |
| **上报工作状态** | 前端 | `ReportLogicWorkingTask.js` | 上报逻辑工作状态 |

## 二、核心业务：转发任务

### 2.1 业务场景

`forwardTask` 是最核心的前端交互业务，它将前端发起的操作请求**转发给逆向服务**执行。

支持的操作类型：

| 任务类型 | GalaxyTaskType | 说明 |
|---------|---------------|------|
| 发送消息 | `SEND_MESSAGE` | 发送文字/图片/视频等 |
| 踢人 | `KICK_OUT` | 踢群成员 |
| 修改备注 | `REMARK` | 修改好友备注 |
| 获取好友列表 | `USER_LIST` | 请求刷新好友列表 |
| 获取群成员 | `CHATUSER_INFO_LIST` | 获取群成员详情 |

### 2.2 处理流程

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                        forwardTask 处理流程                                   │
└─────────────────────────────────────────────────────────────────────────────┘

前端                     forwardTask                dispatchInBound      逆向服务
  │                          │                           │                  │
  │  WS发送请求               │                           │                  │
  │  {                       │                           │                  │
  │    cmdId:"forwardTask",  │                           │                  │
  │    body: {               │                           │                  │
  │      type: "SEND_MESSAGE"│                           │                  │
  │      data: {...}         │                           │                  │
  │    }                     │                           │                  │
  │  }                       │                           │                  │
  ├─────────────────────────►│                           │                  │
  │                          │                           │                  │
  │                          │ 1. 查找Registry           │                  │
```

### 2.3 核心代码

```javascript
// src/msg-center/business/task-front/forwardTask.js

async function forwardTask(params) {
    const { request = {}, message } = params ?? {};
    let clientTask = { ...ClientTaskBo, ...(request?.body ?? {}) };
    const { channelId, wxid } = request;

    // ═══════════════════════════════════════════════════════════════════
    // 1. 查找机器人注册信息
    // ═══════════════════════════════════════════════════════════════════
    let registry = null;
    if (wxid) {
        registry = RegistryConfig.getRegistryByKey(wxid, 'wxid');
    } else {
        registry = RegistryConfig.getRegistryByKey(channelId, 'id');
    }
    
    if (!registry) {
        logUtil.customLog(`[wxid-${wxid}] forwardTask 尚未建立注册信息`);
        return;
    }
    
    const workWxFlag = registry.workWx;
    const type = clientTask.type;
    
    // 设置消息来源
    if (!UN_FORWARD_TYPE.has(type)) {
        clientTask.flowSource = FlowSourceEnum.FRONT;
    }

    // ═══════════════════════════════════════════════════════════════════
    // 2. 发送消息：处理媒体文件下载
    // ═══════════════════════════════════════════════════════════════════
    if (type === GalaxyTaskType.SEND_MESSAGE) {
        if (!clientTask.data) return;
        
        const content = clientTask?.data?.content;
        if (content.startsWith('http://') || content.startsWith('https://')) {
            const fileName = clientTask.data.fileName;
            clientTask.data.content = await FileDownloadUtil.downloadAsync(content, fileName);
        }
    }
    
    // ═══════════════════════════════════════════════════════════════════
    // 3. 踢人：校验群成员
    // ═══════════════════════════════════════════════════════════════════
    else if (type === GalaxyTaskType.KICK_OUT) {
        const chatroom = clientTask.data?.to;
        const toUsernames = clientTask.data?.array;
        
        if (!chatroom || !toUsernames) {
            logUtil.customLog("踢人参数为空");
            return;
        }
        
        // 企业微信特殊处理
        if (workWxFlag) {
            const kickList = JSON.parse(toUsernames);
            const workWxTask = {
                type: GalaxyTaskType.WORK_KICK_OUT_MEMBER,
                array: kickList.map(id => +id),
                conversation: chatroom,
            };
            dispatchInBound(registry.id, registry.wxid, JSON.stringify(workWxTask));
            return;
        }
        
        // 校验目标是否在群内
        const memberList = ChatroomMemberInfoService.getMemberList(chatroom);
        if (!memberList) {
            sendKickOutFailFront(clientTask, "群不存在或无成员信息");
            return;
        }
        
        const toUsernameList = JSON.parse(toUsernames);
        const list = toUsernameList.filter(item => memberList.includes(item));
        if (!list.length) {
            sendKickOutFailFront(clientTask, "目标不在群内");
            return;
        }
        
        clientTask.data.array = list;
    }
    
    // ═══════════════════════════════════════════════════════════════════
    // 4. 企业微信改备注：格式转换
    // ═══════════════════════════════════════════════════════════════════
    else if (workWxFlag && type === GalaxyTaskType.REMARK) {
        const data = clientTask.data;
        const workWxFriendBO = {
            userid: data.to,
            type: GalaxyTaskType.WORK_CHANGE_REMARK,
            remarks: data.content,
            corp_remark: data?.corpRemark ?? '',
            decription: data?.decription,
            remark_url: data.remarkUrl,
        };
        dispatchInBound(registry.id, registry.wxid, JSON.stringify(workWxFriendBO));
        return;
    }
    
    // ═══════════════════════════════════════════════════════════════════
    // 5. 获取好友列表：特殊处理
    // ═══════════════════════════════════════════════════════════════════
    else if (GalaxyTaskType.USER_LIST === clientTask.type) {
        const userInfoCache = workWxFlag ? wkUserAndChatRoomCache : wxUserAndChatRoomCache;
        const msgStatus = userInfoCache.userMsgStatus.get(registry.wxid);
        
        // 正在获取中，不重复请求
        if (msgStatus === wkUserAndChatRoomCache.MsgStatus.START || 
            msgStatus === wkUserAndChatRoomCache.MsgStatus.INIT) {
            return;
        }
        
        // 先推送旧数据
        if (workWxFlag) {
            frontSendService.sendWkFrinedList(registry.pipeLineWrapper);
        } else {
            frontSendService.sendWxFrinedList(registry.pipeLineWrapper);
        }
        
        // 更新状态并发送请求
        userInfoCache.updateUserMsgStatus(registry.wxid, wkUserAndChatRoomCache.MsgStatus.START);
        clientTask.type = GalaxyTaskType.GET_USER_LIST;
        
        // 设置超时（180秒）
        const timeoutTimer = setTimeout(() => {
            if (userInfoCache.userMsgStatus.get(registry.wxid) !== wkUserAndChatRoomCache.MsgStatus.FINISH) {
                userInfoCache.updateUserMsgStatus(registry.wxid, null);
                notify.onGetUserListTimeout(registry.wxid);
            }
        }, 180000);
        userInfoCache.timeoutTimerMap.set(registry.wxid, timeoutTimer);
        
        await sleep(1000);
        dispatchInBound(registry.id, registry.wxid, JSON.stringify(clientTask));
        return;
    }

    // ═══════════════════════════════════════════════════════════════════
    // 6. 通用处理：发送到逆向
    // ═══════════════════════════════════════════════════════════════════
    clientTask.channelId = request.channelId;
    clientTask.wxId = request.wxid;
    
    // 清理空data
    if (clientTask.data && Object.keys(clientTask.data).length === 0) {
        clientTask.data = undefined;
    }
    
    dispatchInBound(registry.id, registry.wxid, JSON.stringify(clientTask));
}
```

## 三、获取机器人配置

### 3.1 业务场景

前端需要获取当前所有微信/企微的连接状态，用于界面展示。

### 3.2 处理流程

```javascript
// src/msg-center/business/task-front/getAllconfigTask.js

function getAllConfigTask(params) {
    logUtil.customLog("[appInit] getAllConfigTask 获取所有逆向状态的配置");
    frontSendService.sendGetAllConfig();
}

// src/msg-center/dispatch-center/frontSend.js
sendGetAllConfig() {
    const registryList = RegistryConfig.getRegistryList();
    
    // 构建配置列表
    const configs = registryList.map(registry => ({
        id: registry.id,
        wxid: registry.wxid,
        nickname: registry.wxInfo?.nickname,
        headimg: registry.wxInfo?.headimg,
        workWx: registry.workWx,
        connected: !!registry.pipeLineWrapper?.available,
        mqttConnected: !!registry.mqttClient?.connected,
        loginTime: registry.loginTime,
    }));
    
    // 推送到前端
    const message = JSON.stringify({
        cmdId: 'get-all-config',
        code: 0,
        data: configs
    });
    
    this.send(message);
}
```

## 四、文件上传

### 4.1 业务场景

前端上传文件（图片、视频等）到阿里云 OSS。

### 4.2 处理流程

```javascript
// src/msg-center/business/task-front/UploadTask.js

async function uploadTask(params) {
    const { request } = params;
    const { filePath, fileName, wxId } = request.body;
    
    try {
        // 1. 获取OSS凭证
        const credentials = global.ossCredentials;
        if (!credentials) {
            SendToFrontUtil.sendResponse2Front(wxId, {
                cmdId: 'upload-result',
                code: -1,
                error: 'OSS凭证未初始化'
            });
            return;
        }
        
        // 2. 初始化OSS客户端
        const ossClient = new OSS({
            region: credentials.region,
            accessKeyId: credentials.accessKeyId,
            accessKeySecret: credentials.accessKeySecret,
            securityToken: credentials.securityToken,
            bucket: credentials.bucket
        });
        
        // 3. 生成OSS路径
        const ossPath = generateOssPath(wxId, fileName);
        
        // 4. 上传文件
        const result = await ossClient.put(ossPath, filePath);
        
        // 5. 返回结果
        SendToFrontUtil.sendResponse2Front(wxId, {
            cmdId: 'upload-result',
            code: 0,
            data: {
                url: result.url,
                path: ossPath
            }
        });
        
    } catch (error) {
        SendToFrontUtil.sendResponse2Front(wxId, {
            cmdId: 'upload-result',
            code: -1,
            error: error.message
        });
    }
}
```

## 六、消息推送到前端

### 6.1 推送类型

| 推送类型 | cmdId | 触发时机 |
|---------|-------|---------|
| 配置更新 | `get-all-config` | 机器人登录/登出/状态变化 |
| 登出通知 | `logout-notify` | 微信账号退出 |
| 接收消息 | `recv-msg` | 收到新消息 |
| 群聊更新 | `chatroom-update` | 群信息变化 |
| 好友更新 | `friend-update` | 好友信息变化 |

### 6.2 推送实现

```javascript
// src/msg-center/core/front/sendToFrontUtil.js

const SendToFrontUtil = {
    sendResponse2Front(wxId, response) {
        const frontConn = frontConnection;
        
        if (!frontConn.channel || !frontConn.isLogin) {
            logUtil.customLog('前端未连接，跳过推送');
            return;
        }
        
        try {
            frontConn.channel.send(JSON.stringify(response));
        } catch (error) {
            logUtil.customLog(`推送前端失败: ${error.message}`);
        }
    }
};

// src/msg-center/dispatch-center/dispatch/sendToFront.js

function sendToFront(message, channelId, wxid, cmdId) {
    const response = {
        cmdId: cmdId || 'recv-msg',
        channelId: channelId,
        wxid: wxid,
        body: JSON.parse(message)
    };
    
    SendToFrontUtil.sendResponse2Front(wxid, response);
}
```

## 七、消息协议

### 7.1 前端 → 主进程

```javascript
// 请求格式
{
    "cmdId": "forwardTask",      // 命令ID
    "channelId": 12345,          // 通道ID（进程ID）
    "wxid": "wxid_xxx",          // 微信ID
    "body": {                    // 请求体
        "type": "SEND_MESSAGE",
        "data": {
            "to": "target_wxid",
            "content": "Hello",
            "wechatMsgType": 1
        }
    }
}
```

### 7.2 主进程 → 前端

```javascript
// 响应/推送格式
{
    "cmdId": "get-all-config",   // 命令ID
    "code": 0,                   // 状态码 (0=成功)
    "data": {                    // 数据
        // ...
    },
    "error": null                // 错误信息（失败时）
}
```

### 7.3 命令ID列表

| cmdId | 方向 | 说明 |
|-------|------|------|
| `system` | 前端→主进程 | 系统消息（online/offline）|
| `getAllConfig` | 前端→主进程 | 获取所有配置 |
| `forwardTask` | 前端→主进程 | 转发任务 |
| `uploadFile` | 前端→主进程 | 上传文件 |
