### 1.2 核心业务场景

| 场景 | 描述 | 涉及模块 |
|-----|-----|---------|

### 3.2 登录核心逻辑

```javascript
// src/msg-center/business/convert-service/loginService.js

const LoginService = {
    // 过滤条件：判断是否是登录消息
    filter(clientRecord) {
        return clientRecord.type === "login" 
            && clientRecord.status === 0 
            && !clientRecord.busy
            && (clientRecord.data || clientRecord.login_user);
    },
    
    // 处理登录
    async handleLogin(pipeLineWrapper, clientOtherMsgBO, ownerWxId, workWxFlag) {
        await lock.acquire(`loginLock-${ownerWxId}`, async (done) => {
            try {
                // 1. 获取注册表信息
                const registry = RegistryConfig.getRegistryByKey(pipeLineWrapper.id, "id");
                registry.wxid = "" + ownerWxId;
                
                if (registry.sendToCloudFlag) {
                    // 重复登录：重新发送登录通知
                    await this.reLoginInSend(pipeLineWrapper, clientOtherMsgBO, ownerWxId, workWxFlag);
                } else {
                    // 首次登录
                    // 1. 绑定用户信息
                    this.bindWxId(ownerWxId, clientOtherMsgBO, registry, workWxFlag);
                    
                    // 2. 调用云端注册接口
                    await httpFetch({ url: logicRegisteredCloud, data: clientMsg.robotLoginRecord });
                    
                    // 3. 建立 MQTT 连接
                    await MqttHelper.connectMqtt({ wxId: ownerWxId });
                    
                    // 4. 发送登录通知
                    MqttSendService.sendMessage(ownerWxId, clientMsg);
                    
                    // 5. 通知前端刷新
                    frontSendService.sendGetAllConfig();
                }
            } finally {
                done();
            }
        });
    }
};
```

### 4.2 MQTT 消息处理

```javascript
// src/msg-center/core/mq/mqttClientBase.js

client.on("message", async function (topic, message) {
    // 1. 解析消息
    const dataInfo = JSON.parse(message.toString("utf8"));
    
    // 2. 过期检查（默认3小时过期）
    const expireTime = apolloConfig.mqttTaskExpireHours * 60 * 60 * 1000;
    if (dataInfo.createTime * 1000 < Date.now() - expireTime) {
        return; // 消息过期，跳过处理
    }
    
    // 3. 去重检查
    if (mqttTaskMapLock.has(dataInfo.id)) {
        return; // 重复消息，跳过处理
    }
    mqttTaskMapLock.set(dataInfo.id, { timestamp: Date.now() });
    
    // 4. 分发处理
    const serverTaskBO = { ...ServerTaskBO, ...dataInfo };
    if (dataInfo.type === 100) {
        // 群发消息：使用队列处理
        executeWithDelay(wxId, () => {
            execute(wxId, serverTaskBO);
        });
    } else {
        // 其他任务：直接处理
        execute(wxId, serverTaskBO);
    }
});
```

## 五、任务回执流程（逆向 → 云端）

### 5.1 三段式消息机制

项目使用"三段式消息"追踪任务状态：

```
┌─────────────────────────────────────────────────────────────────────────┐
│                        三段式消息机制                                     │
└─────────────────────────────────────────────────────────────────────────┘

┌─────────────────┐    ┌─────────────────┐    ┌─────────────────┐
│    第一条消息    │    │    第二条消息    │    │    第三条消息    │
│   sendmessage   │───►│     recvmsg     │───►│      oplog      │
│                 │    │                 │    │                 │
│   发送确认      │    │    接收确认      │    │    最终状态      │
│   (任务已下发)  │    │   (微信已接收)   │    │  (成功/失败)    │
└─────────────────┘    └─────────────────┘    └─────────────────┘
        │                      │                      │
        ▼                      ▼                      ▼
   记录 taskId            关联 taskId             汇总上报云端
   存入 msgResNodeMap     校验消息匹配            清理缓存

时间窗口：通常在 1-5 秒内完成三条消息的接收
```

### 5.2 消息处理核心逻辑

```javascript
// src/msg-center/dispatch-center/handle/msgHandleBase.js

const MsgHandlerCenterBase = {
    // 消息节点缓存
    msgResNodeMap: {},
    
    // 处理出站消息
    outBoundMsg(msgObj, pipeLineWrapper, sendType) {
        const wxId = pipeLineWrapper.wxid;
        const key = `${wxId}:${msgObj.firstCallbackType || sendType}`;
        
        let node = this.msgResNodeMap[key];
        
        // 判断是否需要重置节点
        if (this.isReset(sendType, node)) {
            this.msgResNodeMap[key] = null;
            node = null;
        }
        
        // 判断消息阶段
        if (this.isFirstMsg(node, sendType)) {
            // 第一条消息：创建节点
            this.msgResNodeMap[key] = { first: msgObj };
            this.processFirstMsg(msgObj, pipeLineWrapper);
        } 
        else if (this.isSecondMsg(node, sendType)) {
            // 第二条消息：关联节点
            node.second = msgObj;
            this.processSecondMsg(msgObj, node, pipeLineWrapper);
        }
        else if (this.isThirdMsg(node, sendType, msgObj)) {
            // 第三条消息：完成处理
            node.third = msgObj;
            this.processThirdMsg(msgObj, node, pipeLineWrapper);
            
            // 上报云端
            cloudFlowOutBound(wxId, msgObj, node);
            
            // 清理缓存
            delete this.msgResNodeMap[key];
        }
    }
};
```

### 5.3 回执上报流程

```javascript
// src/msg-center/dispatch-center/dispatch/cloudFlowOutBound.js

function cloudFlowOutBound(wxId, msgObj, node) {
    // 1. 构建上报消息
    const reportData = {
        taskId: msgObj.taskId,
        wxId: wxId,
        status: msgObj.status, // 0=成功, 1=失败
        sendTime: Date.now(),
        // ...
    };
    
    // 2. 调用回执处理器
    const responseHandler = getResponseHandler(msgObj.type);
    if (responseHandler) {
        responseHandler.operate(reportData, msgObj, node);
    }
    
    // 3. 发送到云端
    mqttSend.sendMessage(wxId, reportData);
}
```
