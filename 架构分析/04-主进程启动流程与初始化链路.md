## 二、electron.js — 主进程入口

**文件路径**：`galaxy-client/src/electron.js`  
**总行数**：25 行

### 2.1 完整代码解析

```javascript
// 第 1 行：全局性能计时起点
global.readyStartTime = Date.now();

// 第 2-8 行：引入核心模块
const { mode, type } = require('./common/urls');
const initLog = require('./init/initLog');
const addAppEvent = require('./event/app');
const addIpcEvent = require('./event/ipc');
const addStoreEvent = require('./event/store');
const AppStart = require('./msg-center/start/appStart');
const { setupRemoteDebug } = require('./init/remoteDebug');

// 第 10 行：日志系统初始化（必须最先执行）
initLog();

// 第 13 行：远程调试初始化（必须在 app.ready 之前）
setupRemoteDebug();

// 第 15-23 行：引导启动函数
function bootstrap() {
    let isQuit = addAppEvent();   // 注册 app 事件，返回是否需要退出
    if (isQuit) {
        return;                   // 单实例锁获取失败，退出
    }
    addIpcEvent();                // 注册 IPC 事件
    addStoreEvent();              // 注册 store 变更监听
    AppStart.run();               // 启动业务子系统
}

// 第 25 行：执行引导
bootstrap();
```

## 三、initLog() — 日志系统初始化

**文件路径**：`galaxy-client/src/init/initLog.js`

```javascript
const log = require('electron-log');
const slsLogUtil = require('./slsLog.js');

function initLog() {
    log.transports.file.format = '{h}:{i}:{s}:{ms} {text}';
    log.transports.file.level = 'error';
    log.transports.file.maxSize = 1024 * 1024 * 100;  // 100MB
    slsLogUtil.initSlsLog();
}
module.exports = initLog;
```

### 3.1 配置详解

| 配置项 | 值 | 说明 |
|--------|-----|------|
| `format` | `{h}:{i}:{s}:{ms} {text}` | 日志格式：时:分:秒:毫秒 + 内容 |
| `level` | `'error'` | 文件日志只记录 error 级别（控制台会记录所有级别） |
| `maxSize` | `100MB` | 单个日志文件最大 100MB |

### 3.2 SLS 日志初始化

`slsLogUtil.initSlsLog()` 初始化阿里云 SLS（Simple Log Service）远程日志上报。后续所有通过 `logUtil.customLog()` 的日志会同时写入本地文件和 SLS。

### 3.3 自定义日志工具

**文件路径**：`galaxy-client/src/init/log.js`

```javascript
const log = require('electron-log');
const slsLogUtil = require('./slsLog.js');

const logUtil = {
    customLog(message, options) {
        slsLogUtil.customLog(message);          // 上报 SLS
        const {level = 'info', errorKey = ''} = options ?? {};
        if (level === 'info') {
            log[level](errorKey, message);      // electron-log 记录
        } else {
            log[level](message);
        }
    },
};
```

`logUtil.customLog()` 是项目中使用最广泛的日志函数，被 `msg-center/` 下所有模块引用。它同时完成：
1. 本地文件日志（通过 `electron-log`）
2. 远程 SLS 上报（通过 `slsLogUtil`）

## 四、setupRemoteDebug() — 远程调试

**文件路径**：`galaxy-client/src/init/remoteDebug.js`

### 4.1 设计目的

在生产环境的 Windows 机器上，开发者无法直接使用 Chrome DevTools 调试。远程调试模块通过以下方式解决：

1. **主进程调试**：开启 Node.js Inspector（端口 9229），绑定所有网卡
2. **渲染进程调试**：开启 Chrome DevTools Protocol（端口 9222），但 Chromium 强制绑定 127.0.0.1
3. **TCP 代理**：创建 0.0.0.0:9223 → 127.0.0.1:9222 的 TCP 代理，实现远程访问

### 4.2 激活条件

```javascript
function isRemoteDebugEnabled() {
    return process.argv.includes('--remote-debug');
}
```

仅当启动参数包含 `--remote-debug` 时才激活。

### 4.3 初始化时序

```javascript
function setupRemoteDebug() {
    if (!isRemoteDebugEnabled()) return false;

    // 1. 主进程 Inspector（必须在 app.ready 之前）
    const inspector = require('inspector');
    inspector.open(9229, '0.0.0.0');

    // 2. 渲染进程 CDP（Chromium 命令行参数，必须在 app.ready 之前）
    app.commandLine.appendSwitch('remote-debugging-port', '9222');

    // 3. app.ready 后启动 TCP 代理
    app.on('ready', () => {
        setTimeout(() => {
            createTcpProxy(9223, 9222);
        }, 2000);
    });
}
```

### 4.4 调试信息对话框

`showDebugInfoDialog()` 在 `app.on('ready')` 后被调用，向用户弹出包含调试连接信息的对话框，显示本机 IP 和各端口。

function getLoadUrl() {
    let userId = store.getUserId();
    if (!userId) {
        userId = randomString(16);
        store.setUserId(userId);
        return `${loginUrl}?userId=${userId}`;     // → CAS 登录页

async function getLoadUrlAsync() {
    const result = await net.check();               // 网络连通性检查
    if (result) {
        return getLoadUrl();
    }
    return offlineUrl;                              // → 离线错误页 (error.html)
}

## 七、addIpcEvent() — IPC 事件注册

**文件路径**：`galaxy-client/src/event/ipc.js`

使用 `electron-better-ipc` 库注册 IPC 事件。分为两类 API：
- `ipc.on(channel, handler)` — 渲染进程 → 主进程单向通信

- `ipc.handle(channel, handler)` — 标准 Electron invoke/handle 模式

exports.getAppConfig = () => {
    if (!appConfig) {
        const configRoot = `../config/${this.getProduct()}`;
        const configFilePath = path.resolve(__dirname, 
            isDev ? `${configRoot}/runtime.yml` : '../../app-config.yml'
        );
        appConfig = this.getYamlConfig(configFilePath);
        if (isDev) {
            const devConfigFilePath = path.resolve(__dirname, `${configRoot}/runtime.dev.yml`);

### 11.3 常见启动失败场景

| 场景 | 表现 | 原因 |
|------|------|------|
