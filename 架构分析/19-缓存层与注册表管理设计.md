## 2. 注册表管理 registryList.js

**文件路径**：`galaxy-client/src/msg-center/core/registry-config/registryList.js`

### 2.1 完整源码

```javascript
// @ts-check
let registryList = [];

function addRegistry(registry) {
    const registryItem = registryList.find(item => item.id === registry.id);
    if (registryItem) {
        Object.assign(registryItem, registry);
    }
    else {
        registryList.push(registry);
    }
}

function removeRegistry(registry) {
    registryList = registryList.filter(item => item.id !== registry?.id);
}

function getRegistryByKey(value, key) {
    if (value) {
        const registry = registryList.find(item => item[key] == value);
        return registry;
    }
    return null;
}

function getAllConfig() {
    return registryList.map(registry => ({
        channelId: registry.id,
        wxInfo: registry.wxInfo,
        available: registry.pipeLineWrapper.available,
        lastPongTime: registry.pipeLineWrapper.lastPongTime,
        orderId: registry.orderId,
        workWx: registry.workWx,
    }));
}

function getRegistryList() {
    return registryList.map(registry => ({...registry}));
}

const MyObject = (() => {
    let instance;
    function init() {
        return {
            addRegistry,
            removeRegistry,
            getRegistryByKey,
            getAllConfig,
            getRegistryList,
        };
    }
    return {
        getInstance: () => {
            if (!instance) {
                instance = init();
            }
            return instance;
        },
    };
})();

module.exports = MyObject;
```

### 2.2 单例模式实现

使用 IIFE + 闭包实现单例模式，确保整个进程中只有一个注册表实例。所有模块通过 `require('./registryList').getInstance()` 获取同一个实例。

### 2.3 registry 数据结构

每个 `registry` 代表一个已连接的微信/企微实例：

```javascript
{
    id: '12345',              // 进程 ID（如微信.exe 的 PID）
    wxid: 'wxid_abc123',     // 微信号 ID
    wxInfo: {                 // 微信账号信息
        nickname: '张三',
        headimg: 'http://...',
        wxid: 'wxid_abc123',
        // ...
    },
    pipeLineWrapper: {        // Named Pipe 连接信息
        pipeCode: 42,         // IPC 连接句柄
        processId: '12345',   // 进程 ID
        available: true,      // 是否可用
        lastPongTime: 1710312345678,  // 最后心跳时间
    },
    mqttClient: mqttInstance, // MQTT 客户端实例（用于该微信号的独立 MQTT 连接）
    orderId: 'order_001',    // 订单 ID
    workWx: false,           // 是否是企业微信
}
```

### 2.4 核心方法

| 方法 | 说明 |
|------|------|
| `addRegistry(registry)` | 添加或更新注册表条目（按 `id` 去重） |
| `removeRegistry(registry)` | 按 `id` 移除条目 |
| `getRegistryByKey(value, key)` | 按任意字段查找（`key` 为字段名） |
| `getAllConfig()` | 返回精简的配置列表（供前端展示） |
| `getRegistryList()` | 返回完整列表的浅拷贝 |

### 2.5 addRegistry 的幂等设计

```javascript
function addRegistry(registry) {
    const registryItem = registryList.find(item => item.id === registry.id);
    if (registryItem) {
        Object.assign(registryItem, registry);  // 存在则合并更新
    } else {
        registryList.push(registry);            // 不存在则新增
    }
}
```

使用 `Object.assign` 合并更新，支持部分字段更新（如只更新 `wxInfo`）。

### 2.6 getRegistryList 的拷贝保护

```javascript
function getRegistryList() {
    return registryList.map(registry => ({...registry}));
}
```

返回浅拷贝，防止外部直接修改内部数组。但需注意：嵌套对象（如 `pipeLineWrapper`）仍是引用共享。

### 3.1 与 registryList.js 的关系

| 方法 | 功能 |

| `getRegistry(wxid)` | 按 wxid 查找注册表条目 |

### 3.2 removePipe 详解

```javascript
removePipe(processId, pipeCode) {
    try {
        let registry = this.getRegistryByKey(processId, 'id');
        logUtil.customLog(`[removePipe] [关闭IPC及MQTT连接] ...`);
        if (!registry?.pipeLineWrapper?.pipeCode && !pipeCode) {
            return;
        }
        if (!registry?.pipeLineWrapper?.pipeCode && pipeCode) {
            Clibrary.IpcClientClose(pipeCode);
            return;
        }
        Clibrary.IpcClientClose(registry.pipeLineWrapper.pipeCode);
        this.forceClearMqtt(registry?.wxid, registry);
        this.remove(registry);
    } catch(error) {
        logUtil.customLog(`[codeError] ...`, { level: 'error', errorKey: 'removePipeError' });
    }
}
```

`removePipe` 执行三步清理：
1. 关闭 Named Pipe 连接（`Clibrary.IpcClientClose`）
2. 关闭 MQTT 连接（`forceClearMqtt`）
3. 从内存注册表中移除条目

### 3.3 注册表生命周期

```
微信进程被发现         Named Pipe 连接建立        获取到微信信息
    │                      │                       │
    └──▶ initIpcTask ──▶ addRegistry({          ──▶ addRegistry({
          扫描进程           id: pid,                wxInfo: {...},
                            pipeLineWrapper: {      wxid: 'wxid_xxx'
                              pipeCode: handle,   })
                              processId: pid,
                              available: true
                            }
                          })
         │
         ▼
   ┌─────────────────────────────────────────────────┐
   │            registry 存活期间                      │
   │                                                  │
   │  HeartBeatTimer → setPongTime(id) 每10秒更新     │
   │  消息收发 → getRegistryByKey() 查找实例           │
   │  前端展示 → getAllConfig() 获取配置列表           │
   │                                                  │
   └─────────────────────────────────────────────────┘
         │
         ▼
   微信进程消失 / 手动断开
         │
         └──▶ removePipe(processId)
               ├── IpcClientClose(pipeCode)   // 关闭 IPC
               ├── forceClearMqtt(wxId)       // 关闭 MQTT
               └── remove(registry)           // 移除条目
```

## 4. 群信息缓存 chatroomInfoCache.js

**文件路径**：`galaxy-client/src/msg-center/core/cache/chatroomInfoCache.js`

这是最复杂的缓存模块，管理群相关的多种数据。

### 4.1 缓存类型一览

| 缓存名 | 数据结构 | 用途 |
|--------|----------|------|
| `chatroomKeywordInfoCache` | `Map<key, value>` | 群关键信息（群名、头像、验证状态） |
| `newChatroomMembersCache` | `Map<cacheKey, memberList>` | 临时获取的群成员列表 |
| `singleChatroomInfoCache` | `Map<cacheKey, chatroomInfo>` | 临时获取的群详情（群主等） |
| `isGetSingleChatroomInfoCache` | `Map<cacheKey, boolean>` | 是否已触发群详情请求 |
| `newMemberCache` | `Map<accountId, {chatroomId: members}>` | 新增群成员 |
| `chatroomMemberListCache` | `Object<chatroomId, memberList>` | 群成员列表 |
| `memberKickOutMap` | `Object<key, fromUsername>` | 被踢出群记录 |
| `inviteTalkList` | `Map<chatroomId, Array>` | 邀请人进群的匹配话术 |
| `chatroomNewMemberList` | `Array` | 新进群成员列表 |
| `userInfoListRecord` | `Array<Array<string>>` | 已请求的 userInfoList 记录 |

### 4.2 异步重试获取模式

缓存模块大量使用了「请求 → 等待 → 重试」的异步模式：

```javascript
async getChatroomKeywordInfoCache(key, retryInfo = { retryCount: 0, maxRetryCount: 5 }) {
    const chatroomKeywordInfo = this.chatroomKeywordInfoCache.get(key);
    if (!chatroomKeywordInfo && retryInfo.retryCount < retryInfo.maxRetryCount) {
        retryInfo.retryCount++;
        await sleep(300);
        return await this.getChatroomKeywordInfoCache(key, retryInfo);
    }
    return chatroomKeywordInfo;
}
```

这种模式的核心思路：
1. 先发送逆向 IPC 请求获取群信息
2. 数据异步返回后写入缓存
3. 调用方通过轮询+重试的方式等待数据到达

## 11. 线程池管理 pool

### 11.1 reversePoolManager.js — 逆向 IPC 线程池

**文件路径**：`galaxy-client/src/msg-center/core/pool/reversePoolManager.js`

```javascript
const WorkerPool = require('../worker-threads/workerPool');
const os = require('os');

const ReverseThreadPoolManger = {
    pool: null,
    submitIpcInit(task) {
        if (!this.pool) {
            this.pool = new WorkerPool(os.cpus().length);
        }
        this.pool.runTask(task, (err, result) => {
            console.log(err, result);
        });
    },
    shoutDown() {
        this.pool.close();
    }
};
```

| 特性 | 说明 |
|------|------|
| 线程池大小 | `os.cpus().length`（CPU 核心数） |
| 底层实现 | `WorkerPool`（worker-threads 模块） |
| 用途 | 将耗时的 IPC 初始化任务放到 Worker Thread 执行 |
| 懒初始化 | 首次 `submitIpcInit` 时创建线程池 |

### 12.1 职责

### 12.2 主入口 getClientTask

```javascript
function getClientTask(serverTask) {
    switch (serverTask.type) {
        case TaskType.SEND_MESSAGE:
            return getSendMessageTask(serverTask);
        case TaskType.FRIEND_LIST:
            return getFriendListTask(serverTask);
```

### 13.2 store.js 中的持久化接口

`galaxy-client/src/common/store.js` 提供了 `getRegistryList()` / `setRegistryList()` 方法用于持久化注册表数据，但**当前项目中未发现实际调用**。注册表数据完全在内存中管理，应用重启后需要重新扫描微信进程来重建。

### 13.3 缓存清理策略
