# 数据库架构详解

> SQLite 数据库设计与 Sequelize ORM 使用

---

## 一、数据库概述

### 1.1 技术选型

| 技术 | 版本 | 说明 |
|-----|------|-----|
| SQLite | 3.x | 轻量级嵌入式数据库 |
| Sequelize | 7.0.0-alpha.2 | Node.js ORM 框架 |

### 1.2 为什么选择 SQLite

```
┌─────────────────────────────────────────────────────────────────────────┐
│                       SQLite 优势                                        │
└─────────────────────────────────────────────────────────────────────────┘

1. 嵌入式数据库
   └── 无需单独安装数据库服务器
   └── 数据存储在单个 .db 文件中
   └── 随应用一起打包分发

2. 轻量高效
   └── 启动快，无需连接池
   └── 读取性能优秀
   └── 适合客户端本地存储

3. 零配置
   └── 无需数据库管理
   └── 无需用户权限配置
   └── 开箱即用

局限性：
- 不支持高并发写入
- 不适合大数据量（>1GB）
- 无网络访问能力
```

### 1.3 数据库文件位置

```javascript
// 数据库文件存储路径
```

### 3.2 friend 表（好友信息）

```sql
CREATE TABLE friend (
    id          INTEGER PRIMARY KEY,
    username    VARCHAR(128) NOT NULL UNIQUE,  -- 微信ID (wxid_xxx)
    alias       VARCHAR(255),                   -- 微信号
    city        VARCHAR(255),                   -- 城市
    con_remark  VARCHAR(255),                   -- 备注名
    ext         VARCHAR(255),                   -- 扩展信息
    head_url    VARCHAR(255),                   -- 头像URL
    nickname    VARCHAR(128),                   -- 昵称
    province    VARCHAR(255),                   -- 省份
    sex         TINYINT,                        -- 性别 (1男 2女 0未知)
    create_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    update_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 索引
CREATE UNIQUE INDEX friend_wxid_index ON friend (username);
```

**Sequelize 模型：**

```javascript
// src/sqlite/entities/friend.js

module.exports = function(sequelize, DataTypes) {
    return sequelize.define('friend', {
        id: {
            autoIncrement: true,
            type: DataTypes.INTEGER,
            allowNull: true,
            primaryKey: true
        },
        username: {
            type: DataTypes.STRING(128),
            allowNull: false,
            unique: true
        },
        alias: {
            type: DataTypes.STRING(255),
            allowNull: true
        },
        // ... 其他字段
        conRemark: {
            type: DataTypes.STRING(255),
            allowNull: true,
            field: 'con_remark'  // 字段映射
        },
        headUrl: {
            type: DataTypes.STRING(255),
            allowNull: true,
            field: 'head_url'
        },
    }, {
        sequelize,
        tableName: 'friend',
        timestamps: false,  // 禁用自动时间戳
        indexes: [
            {
                name: "friend_wxid_index",
                unique: true,
                fields: [{ name: "username" }]
            }
        ]
    });
};
```

### 3.3 friend_relation 表（好友关系）

```sql
CREATE TABLE friend_relation (
    id                INTEGER PRIMARY KEY,
    friend_remark     VARCHAR(128),              -- 好友备注
    friend_wxid       VARCHAR(128) NOT NULL,     -- 好友微信ID
    owner_wxid        VARCHAR(128) NOT NULL,     -- 机器人微信ID
    report_flag       TINYINT NOT NULL,          -- 上报标记 (0未上报 1已上报)
    friend_record_time INTEGER NOT NULL,         -- 成为好友时间戳
    create_time       TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    update_time       TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    UNIQUE (owner_wxid, friend_wxid)  -- 联合唯一约束
);

-- 索引
CREATE INDEX friend_relation_wxid_friend_index
ON friend_relation (owner_wxid, friend_wxid);
```

**说明：** 这是一个中间表，记录「机器人」与「好友」的关系。一个好友可能被多个机器人添加。

### 4.2 DAO Model 层

```javascript
// src/msg-center/business/dao-model/friendModel.js

const { friend } = require('../../../sqlite/entities/index');

const FriendModel = {
    /**
     * 批量保存（新增或更新）
     */
    async saveAll(entitys) {
        const insertList = [];
        const promiseList = [];

        for (let entity of entitys) {
            if (entity.id) {
                // 有ID则更新
                const promise = friend.update(entity, {
                    where: { id: entity.id }
                });
                promiseList.push(promise);
            } else {
                // 无ID则新增
                insertList.push(entity);
            }
        }

        await Promise.all(promiseList);

        if (insertList.length > 0) {
            await friend.bulkCreate(insertList);
        }

        return entitys.length;
    },

    /**
     * 单条保存
     */
    save(entity) {
        if (entity.id) {
            return friend.update(entity, {
                where: { id: entity.id }
            });
        }
        return friend.create(entity);
    },

    /**
     * 根据ID删除
     */
    deleteFriendById(id) {
        return friend.destroy({
            where: { id }
        });
    },

    /**
     * 根据用户名查找
     */
    findFriendByUsername(wxid) {
        return friend.findOne({
            where: { username: wxid }
        });
    },

    /**
     * 根据用户名删除
     */
    deleteByUsername(username) {
        return friend.destroy({
            where: { username }
        });
    }
};

module.exports = FriendModel;
```

### 4.3 DAO Service 层

```javascript
// src/msg-center/business/dao-service/FriendService.js

const FriendModel = require("../dao-model/friendModel");
const logUtil = require('../../../init/log');

const FriendService = {

     * 删除好友

    },

    /**
     * 根据ID查找
     */
    async findById(id) {
        const result = await FriendModel.findById(id);
        return result ?? {};
    },

    /**
     * 批量保存或更新

    async saveOrUpdate(friendList) {

     * 根据用户名删除

        return await FriendModel.deleteByUsername(username);
```

## 七、注意事项

### 7.1 并发写入

```javascript
// SQLite 不支持高并发写入，使用 async-lock 避免冲突

const AsyncLock = require('async-lock');
const lock = new AsyncLock();

async function saveFriendList(wxId, friendList) {
    // 使用锁保护数据库写入
    await lock.acquire(`db-friend-${wxId}`, async () => {
        await FriendService.saveOrUpdate(friendList);
    });
}
```

### 7.2 事务处理

```javascript
// Sequelize 事务
const { sequelize } = require('../sqlite/entities/index');

async function transferData() {
    const transaction = await sequelize.transaction();

    try {
        await friend.create({ ... }, { transaction });
        await friendRelation.create({ ... }, { transaction });

        await transaction.commit();
    } catch (error) {
        await transaction.rollback();
        throw error;
    }
}
```

### 7.3 数据库迁移

```javascript
// 添加新字段
await sequelize.query(`
    ALTER TABLE friend
    ADD COLUMN new_field VARCHAR(255) DEFAULT NULL
`);

// 创建新表
await sequelize.query(`
    CREATE TABLE IF NOT EXISTS new_table (
        id INTEGER PRIMARY KEY,
        ...
    )
`);
```

## 八、与 React 开发对比

| 概念 | React 前端 | Electron SQLite |
|-----|-----------|-----------------|
| 数据存储 | LocalStorage / IndexedDB | SQLite |
| ORM | 无（直接操作） | Sequelize |
| 查询 | 键值对 / 游标 | SQL / ORM 方法 |
| 关系 | 无 | 外键 / 关联 |
| 事务 | 无 | 支持 |
| 索引 | 键 / 游标 | CREATE INDEX |

```javascript
// React LocalStorage
localStorage.setItem('user', JSON.stringify(user));
const user = JSON.parse(localStorage.getItem('user'));

// Electron SQLite
await friend.create(user);
const user = await friend.findOne({ where: { id: 1 } });
```
