### 2.2 核心操作

```javascript
// src/msg-center/core/registry-config/index.js

const RegistryConfig = {
    // ═══════════════════════════════════════════════════════════════════
    // 添加注册信息
    // ═══════════════════════════════════════════════════════════════════
    add(registry) {
        registryList.addRegistry(registry);
    },

    // ═══════════════════════════════════════════════════════════════════
    // 根据键查找
    // ═══════════════════════════════════════════════════════════════════
    getRegistryByKey(value, key = 'id') {
        return registryList.getRegistryByKey(value, key);
        // key 可以是: 'id', 'wxid', 'pipeCode'
    },

    // ═══════════════════════════════════════════════════════════════════
    // 获取所有配置（用于前端展示）
    // ═══════════════════════════════════════════════════════════════════
    getAllConfig() {
        return registryList.getRegistryList().map(r => ({
            wxid: r.wxid,
            nickname: r.wxInfo?.nickname,
            avatar: r.wxInfo?.avatar,
            workWx: r.workWx,
            available: r.pipeLineWrapper?.available,
            mqttConnected: r.mqttClient?.connected
        }));
    },

    // ═══════════════════════════════════════════════════════════════════
    // 移除注册（断开连接时）
    // ═══════════════════════════════════════════════════════════════════
    removePipe(processId, pipeCode) {
        const registry = this.getRegistryByKey(processId, 'id');

        // 关闭IPC连接
        Clibrary.IpcClientClose(pipeCode || registry?.pipeLineWrapper?.pipeCode);

        // 清除MQTT连接
        this.forceClearMqtt(registry?.wxid, registry);

        // 从列表移除
        registryList.removeRegistry(registry);
    },

    // ═══════════════════════════════════════════════════════════════════
    // 清除MQTT连接
    // ═══════════════════════════════════════════════════════════════════
    forceClearMqtt(wxId, registry) {
        registry?.mqttClient?.end?.(true, {}, () => {
            if (registry) {
                registry.mqttClient = null;
            }
        });
    },

    // ═══════════════════════════════════════════════════════════════════
    // 获取所有微信ID
    // ═══════════════════════════════════════════════════════════════════
    getAllWxId() {
        return registryList.getRegistryList()
            .filter(r => r.wxid)
            .map(r => r.wxid);
    },

    // ═══════════════════════════════════════════════════════════════════
    // 更新心跳时间
    // ═══════════════════════════════════════════════════════════════════
    setPongTime(id) {
        const registry = this.getRegistryByKey(id, 'id');
        if (registry?.pipeLineWrapper) {
            registry.pipeLineWrapper.lastPongTime = Date.now();
        }
    }
};
```

### 2.3 使用场景

```javascript
// 1. 登录时添加注册信息

    id: processId,
    wxid: wxId,
    pipeLineWrapper: { ... },
```

### 3.1 数据结构

```javascript
// src/msg-center/core/cache/galaxyTaskCache.js

const GalaxyTaskCache = {
    // ═══════════════════════════════════════════════════════════════════

    TASK_MSGTYPE_MAP: new Map(),

    TASK_INFO_MAP: new Map(),
};

// GalaxyTaskStatus 结构
const GalaxyTaskStatus = {
    taskId: 'task_123',
    username: 'wxid_xxx',
    type: 'sendmessage',
    createTime: 1642000000000,

    // 三段消息状态
    firstMessageStatus: 'NOT_RECEIVE',   // NOT_RECEIVE / RECEIVED
    secondMessageStatus: 'NOT_RECEIVE',

    // 消息内容（可选）
    firstMessage: {},
    secondMessage: {},
    thirdMessage: {}
```

### 3.2 核心操作

```javascript
const GalaxyTaskCache = {
    // ═══════════════════════════════════════════════════════════════════
    // 设置任务消息类型映射
    // ═══════════════════════════════════════════════════════════════════
    setTaskMsgTypeMap(msgObj) {
        if (!msgObj) return;

        const sendTime = Date.now();
        const sendUsername = msgObj.ownerWxId;
        const taskId = msgObj.taskId;
        const wechatMsgType = msgObj.wechatMsgType;

        // 确定接收者
        let toUsername = '';
        const conversationId = msgObj.conversation_id || msgObj.conversationid;
        if (conversationId) {
            toUsername = conversationId.split('_')[1];  // 企微
        } else {
            toUsername = msgObj.data.to;  // 微信
        }

        // 提取消息内容
        let content = msgObj.data?.content || msgObj.data?.msg || msgObj.content || '';
        content = this.processContentByType(content, wechatMsgType, msgObj);

        // 构建缓存
        const key = `${sendUsername}-${toUsername}`;
        const valueStr = `${taskId}-${wechatMsgType}-${sendTime}-${content}`;

        if (this.TASK_MSGTYPE_MAP.get(key)) {
            this.TASK_MSGTYPE_MAP.get(key).push(valueStr);
        } else {
            this.TASK_MSGTYPE_MAP.set(key, [valueStr]);
        }

        // 反向索引
        this.TASK_INFO_MAP.set(taskId, key);
    },

    // ═══════════════════════════════════════════════════════════════════
    // 根据消息类型处理内容
    // ═══════════════════════════════════════════════════════════════════
    processContentByType(content, wechatMsgType, msgObj) {
        const MAX_CONTENT_LENGTH = 500;

        switch (wechatMsgType) {
            case '1':  // 文本
                return content.length > MAX_CONTENT_LENGTH
                    ? content.substring(0, MAX_CONTENT_LENGTH) + '...'
                    : content;

            case '3':  // 图片
            case '43': // 视频
            case '49001': // 文件
                return msgObj.data?.path || content;

            default:
                return content.length > MAX_CONTENT_LENGTH
                    ? content.substring(0, MAX_CONTENT_LENGTH) + '...'
                    : content;
        }
    },

    // ═══════════════════════════════════════════════════════════════════
    // 获取任务消息类型
    // ═══════════════════════════════════════════════════════════════════
    getTaskMsgTypeMap(sendUsername, toUsername) {
        const key = `${sendUsername}-${toUsername}`;
        return this.TASK_MSGTYPE_MAP.get(key);
    },

    // ═══════════════════════════════════════════════════════════════════
    // 清理任务缓存
    // ═══════════════════════════════════════════════════════════════════
    removeTaskMsgTypeMapToTaskId(taskId) {
        const key = this.TASK_INFO_MAP.get(taskId);
        if (key) {
            const [sendUsername, toUsername] = key.split('-');
            this.removeTaskMsgTypeMapValue(sendUsername, toUsername, taskId);
        }
    },

    removeTaskMsgTypeMapValue(sendUsername, toUsername, taskId) {
        const key = `${sendUsername}-${toUsername}`;
        let valueArr = this.TASK_MSGTYPE_MAP.get(key);

        if (valueArr) {
            // 过滤掉指定taskId
            valueArr = valueArr.filter(str => str.split('-')[0] !== taskId);

            if (valueArr.length === 0) {
                this.TASK_MSGTYPE_MAP.delete(key);
            } else {
                this.TASK_MSGTYPE_MAP.set(key, valueArr);
            }
        }

        this.TASK_INFO_MAP.delete(taskId);
    }
};
```

### 3.3 使用场景

```javascript
// 1. 发送任务时缓存
GalaxyTaskCache.setTaskMsgTypeMap({
    ownerWxId: 'wxid_xxx',
    taskId: 'task_123',
    wechatMsgType: '1',
    data: { to: 'wxid_target', content: 'Hello' }
});

// 2. 消息补偿时查询
const cached = GalaxyTaskCache.getTaskMsgTypeMap('wxid_xxx', 'wxid_target');
if (cached) {
    // 找到缓存的任务，可以进行补偿
}

// 3. 任务完成后清理

```

```javascript
    // value: { first, second, third, taskId, createTime }

    msgResNodeMap: {},

    // ═══════════════════════════════════════════════════════════════════
    // 任务ID映射
    // key: taskId
    // value: associationKey
    // ═══════════════════════════════════════════════════════════════════

// 节点结构
const MsgResNode = {
    first: {                      // 第一条消息（命令发送）
        type: 'sendmessage',
        taskId: 'task_123',
        status: 0
    },
    second: {                     // 第二条消息（服务器接收）
        type: 'recvmsg',
        status: 2
    },
    third: {                      // 第三条消息（最终结果）
        type: 'oplog',
        status: 0
    },
    taskId: 'task_123',
    createTime: 1642000000000
};
```

### 4.2 核心操作

```javascript
const MsgHandlerCenterBase = {
    // ═══════════════════════════════════════════════════════════════════
    // 处理出站消息
    // ═══════════════════════════════════════════════════════════════════
    outBoundMsg(msgObj, pipeLineWrapper, sendType) {
        const wxId = pipeLineWrapper.wxid;
        const key = this.getAssociationKey(msgObj, wxId, sendType);
        let node = this.msgResNodeMap[key];

        // 检查是否需要重置（超时）
        if (this.shouldResetNode(sendType, node)) {
            this.msgResNodeMap[key] = null;
            node = null;
        }

        // ═══════════════════════════════════════════════════════════════
        // 判断消息阶段
        // ═══════════════════════════════════════════════════════════════

        if (this.isFirstMsg(node, sendType)) {
            // 第一条：创建节点
            this.msgResNodeMap[key] = {
                first: msgObj,
                taskId: msgObj.taskId,
                createTime: Date.now()
            };
        }
        else if (this.isSecondMsg(node, sendType)) {
            // 第二条：关联到节点
            node.second = msgObj;
        }
        else if (this.isThirdMsg(node, sendType, msgObj)) {
            // 第三条：完成处理
            node.third = msgObj;

            // 上报云端
            cloudFlowOutBound(wxId, msgObj, node);

            // 清理缓存
            delete this.msgResNodeMap[key];
        }
    },

    // ═══════════════════════════════════════════════════════════════════
    // 生成关联键
    // ═══════════════════════════════════════════════════════════════════
    getAssociationKey(msgObj, wxId, sendType) {
        const firstType = msgObj.firstCallbackType || sendType;
        return `${wxId}:${firstType}`;
    },

    // ═══════════════════════════════════════════════════════════════════
    // 判断是否需要重置
    // ═══════════════════════════════════════════════════════════════════
    shouldResetNode(sendType, node) {
        if (node && CallBackClassify.FIRST_CALLBACKS.has(sendType)) {
            // 超过1分钟的节点需要重置
            return Date.now() - node.createTime > 60000;
        }
        return false;
    },

    // ═══════════════════════════════════════════════════════════════════
    // 消息阶段判断
    // ═══════════════════════════════════════════════════════════════════
    isFirstMsg(node, sendType) {
        return !node && CallBackClassify.FIRST_CALLBACKS.has(sendType);
    },

    isSecondMsg(node, sendType) {
        return node && node.first && !node.second
            && CallBackClassify.SECOND_CALLBACKS.has(sendType);
    },

    isThirdMsg(node, sendType, msgObj) {
        return node && node.first && node.second
            && CallBackClassify.THIRD_CALLBACKS.has(sendType);
    }
};
```

### 4.3 使用场景

```javascript
// 消息流程示例
// 1. 收到第一条消息 (sendmessage)

    first: { type: 'sendmessage', taskId: 'task_123' },
    createTime: Date.now()
};

    type: 'recvmsg',

};

// 清理

```

## 六、mqttTaskMapLock（MQTT去重锁）

### 6.1 数据结构

```javascript
// src/msg-center/core/mq/mqttClientBase.js

// 任务去重锁
let mqttTaskMapLock = new Map();
// key: taskId
// value: { wxId, timestamp, processed }
```

### 6.2 核心操作

```javascript
// 检查并添加锁
if (receiveId && mqttTaskMapLock.has(receiveId)) {
    logUtil.customLog(`taskId=${receiveId} 消息重复，跳过`);
    return;
}

// 添加锁
if (receiveId) {
    mqttTaskMapLock.set(receiveId, {
        wxId: wxId,
        timestamp: Date.now(),
        processed: true
    });
}

// 定期清理过期锁
function cleanupExpiredLocks() {
    const now = Date.now();
    const expireTime = 6000 * 1000;  // 100分钟

    for (const [taskId, lockInfo] of mqttTaskMapLock.entries()) {
        if (now - lockInfo.timestamp > expireTime) {
            mqttTaskMapLock.delete(taskId);
        }
    }
}

// 启动清理定时器
setInterval(cleanupExpiredLocks, 60000);  // 每分钟清理
```

## 七、缓存生命周期管理

### 7.1 缓存创建时机

| 缓存 | 创建时机 |
|-----|---------|
| RegistryConfig | IPC连接建立时 |
| GalaxyTaskCache | 任务发送时 |
| msgResNodeMap | 收到第一条消息时 |
| pipeCodeCache | IPC连接建立时 |
| mqttTaskMapLock | 收到MQTT消息时 |

### 7.2 缓存清理时机

| 缓存 | 清理时机 | 清理方式 |
|-----|---------|---------|
| RegistryConfig | IPC断开/登出 | `removePipe()` |
| GalaxyTaskCache.TASK_MSGTYPE_MAP | 任务完成 | `removeTaskMsgTypeMapToTaskId()` |
| msgResNodeMap | 收到第三条消息/超时 | `delete` |
| pipeCodeCache | IPC断开 | `deleteProcessIdFromCache()` |
| mqttTaskMapLock | 定时清理（100分钟） | `cleanupExpiredLocks()` |

### 7.3 应用退出清理

```javascript
// src/event/app.js

app.on('before-quit', event => {
    event.preventDefault();

    // 清理所有连接
    RegistryConfig.removeAll();

    // 停止逆向服务
    stopBasicService(() => {
        app.exit();
    });
});
```

### 8.3 内存泄漏防护

```javascript
// 定期清理过期数据
setInterval(() => {
    // 清理过期的消息节点（超过5分钟）
    const now = Date.now();
    for (const key of Object.keys(msgResNodeMap)) {
        const node = msgResNodeMap[key];
        if (now - node.createTime > 5 * 60 * 1000) {
            delete msgResNodeMap[key];
        }
    }

    // 清理过期的任务缓存
    cleanupExpiredLocks();

}, 5 * 60 * 1000);  // 每5分钟
```
