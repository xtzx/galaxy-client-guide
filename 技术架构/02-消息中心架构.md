## 一、消息中心概述

### 1.1 模块定位

消息中心（msg-center）是项目的**核心业务模块**，承担以下职责：

```
┌─────────────────────────────────────────────────────────────────────────┐
│                       消息中心核心职责                                   │
└─────────────────────────────────────────────────────────────────────────┘

消息中心 (msg-center)
├── 通信管理
│   ├── MQTT连接（每个机器人独立）
│   ├── WebSocket服务（前端通信）
│   └── IPC连接（逆向服务通信）
│
├── 消息分发
│   ├── 入站消息分发（云端/前端 → 逆向）
│   ├── 出站消息分发（逆向 → 云端/前端）
│   └── 三段式消息追踪
│
├── 任务处理
│   ├── MQTT任务（46个微信 + 25个企微）
│   ├── 前端任务（10个）
│   └── 逆向响应（56个）
│
├── 数据管理
│   ├── 连接注册（RegistryConfig）
│   ├── 任务缓存（GalaxyTaskCache）
│   └── 本地数据库（SQLite）
│
└── 定时任务
    ├── 心跳检测（30秒）
    ├── 任务状态检查（60秒）
    └── 配置同步（5分钟）
```

## 二、启动流程

### 2.1 启动入口

```javascript
// src/msg-center/start/appStart.js

const FrontStart = require('./frontStart');
const ReverseStart = require('./reverseStart');
const scheduleRun = require('./schedual');
const MemoryQueueApplication = require('../core/queue/MemoryQueueApplication');
const { judgeCrashAndReport } = require('../../common/recordActivityInfo');

const AppMain = {
    run() {
        // 记录启动时间
        global.startTime = Date.now();
        
        // 1. 启动WebSocket服务（与前端通信）
        FrontStart.run();
        
        // 2. 启动IPC连接扫描（与逆向通信）
        ReverseStart.run();
        
        // 3. 启动定时任务
        scheduleRun();
        
        // 4. 启动内存队列
        MemoryQueueApplication.run();
        
        // 5. 崩溃检测上报
        judgeCrashAndReport();
    }
};

module.exports = AppMain;
```

### 3.2 入站消息处理（InBound）

```javascript
// src/msg-center/dispatch-center/dispatchInBound.js

const AsyncLock = require('async-lock');
const logUtil = require('../../init/log');
const RegistryConfig = require('../core/registry-config/index');
const GalaxyTaskCache = require('../core/cache/galaxyTaskCache');
const TaskCallBack = require('../core/data-config/taskCallback');
const WkTaskCallBack = require('../core/data-config/workTaskCallback');
const GalaxyTaskStatusConstant = require('../core/data-config/galaxyTaskStatusConstant');
const reverseSend = require('./reverseSend');

const lock = new AsyncLock();
const reverseSendService = reverseSend.getInstance();

/**
 * 入站消息分发 - 将任务发送给逆向服务
 * 
 * @param {string} channelId - 通道ID（进程ID）
 * @param {string} wxId - 微信ID
 * @param {string} message - JSON消息字符串
 */
function dispatchInBound(channelId, wxId, message) {
    try {
        // ═══════════════════════════════════════════════════════════════
        // 1. 获取连接注册信息
        // ═══════════════════════════════════════════════════════════════
        let registry = null;
        if (wxId) {
            registry = RegistryConfig.getRegistryByKey(wxId, 'wxid');
        } else {
            registry = RegistryConfig.getRegistryByKey(channelId, 'id');
        }
        
        if (!registry) {
            logUtil.customLog(`[dispatchInBound] 尚未建立注册信息 wxId=${wxId} channelId=${channelId}`);
            return;
        }
        
        // ═══════════════════════════════════════════════════════════════
        // 2. 解析消息
        // ═══════════════════════════════════════════════════════════════
        const jsonObject = JSON.parse(message);
        const type = typeConvert(jsonObject.type || jsonObject.body?.type);
        const taskId = jsonObject.taskId || jsonObject.body?.taskId;
        
        logUtil.customLog(`[dispatchInBound] [wxid-${wxId}] type=${type} taskId=${taskId}`);
        
        // ═══════════════════════════════════════════════════════════════
        // 3. 处理需要追踪的任务（三段式消息）
        // ═══════════════════════════════════════════════════════════════
        const callbackType = TaskCallBack.taskCallbackMap[type] || WkTaskCallBack.taskCallbackMap[type];
        
        if (type && callbackType) {
            // 使用锁防止并发
            const lockKey = `galaxyTaskLock-${wxId}:${callbackType}`;
            
            lock.acquire(lockKey, (done) => {
                try {
                    // 创建任务状态对象
                    const galaxyTaskStatus = {
                        firstMessageStatus: GalaxyTaskStatusConstant.NOT_RECEIVE_STATUS,
                        secondMessageStatus: GalaxyTaskStatusConstant.NOT_RECEIVE_STATUS,
                        thirdMessageStatus: GalaxyTaskStatusConstant.NOT_RECEIVE_STATUS,
                        username: wxId,
                        taskId: taskId,
                        createTime: Date.now(),
                        type: type
                    };
                    
                    // 缓存任务状态（用于后续回执匹配）
                    if (taskId) {
                        GalaxyTaskCache.GALAXY_TASK_STATUS_MAP[taskId] = galaxyTaskStatus;
                    }
                    
                    // 发送给逆向服务
                    inBoundAct(channelId, wxId, message, registry);
                    
                    done();
                } catch (error) {
                    logUtil.customLog(`[dispatchInBound] 加锁处理错误: ${error.message}`, { level: 'error' });
                    done();
                }
            });
        } else {
            // 不需要追踪的任务，直接发送
            inBoundAct(channelId, wxId, message, registry);
        }
        
    } catch (error) {
        logUtil.customLog(`[dispatchInBound] 错误: ${error.message}`, { level: 'error' });
    }
}

/**
 * 实际发送到逆向服务
 */
function inBoundAct(channelId, wxId, message, registry) {
    reverseSendService.sendMessage(wxId, channelId, message);
}

/**
 * 类型转换（统一格式）
 */
function typeConvert(type) {
    if (typeof type === 'string') {
        return type.toLowerCase();
    }
    return type;
}

module.exports = dispatchInBound;
```

### 3.3 出站消息处理（OutBound）

```javascript
// src/msg-center/dispatch-center/dispatchOutBound.js

const logUtil = require('../../init/log');
const { reportLog } = require('../../init/habo');
const WxMsgHandlerCenter = require('./handle/wxMsgHandle');
const WkMsgHandlerCenter = require('./handle/workWxMsgHandle');
const CallBackClassify = require('../core/data-config/callbackClassify');
const apolloConfig = require('../core/utils/getApolloConfig');
const GalaxyVersionCache = require('../core/cache/galaxyVersionCache');

/**
 * 处理大数字（避免JSON解析精度丢失）
 */
function replaceLargeNumbers(json) {
    const regex = /(?<=:)\s*\d{17,}(?=[,\}])/g;
    return json.replace(regex, match => `"${match}"`);
}

/**
 * 出站消息分发 - 处理逆向返回的消息
 * 
 * @param {string} message - JSON消息字符串
 * @param {Object} pipeLineWrapper - 管道包装对象
 */
function dispatchOutBound(message, pipeLineWrapper) {
    try {
        // 处理大数字
        message = replaceLargeNumbers(message);
        const jsonObject = JSON.parse(message);
        
        // ═══════════════════════════════════════════════════════════════
        // 1. 日志记录（排除高频消息）
        // ═══════════════════════════════════════════════════════════════
        if (jsonObject.type !== 'pong') {
            logUtil.customLog(`[DispatchCenter] [wxid-${pipeLineWrapper.wxid}] 接收逆向 type=${jsonObject.type}`);
        }
        
        // ═══════════════════════════════════════════════════════════════
        // 2. 错误类型上报
        // ═══════════════════════════════════════════════════════════════
        if (jsonObject.type?.indexOf('error') !== -1) {
            reportLog({
                name: 'REVERSE_ERROR',
                wxid: pipeLineWrapper.wxid,
                errorName: jsonObject.type,
                message: JSON.stringify(jsonObject)
            });
        }
        
        // ═══════════════════════════════════════════════════════════════
        // 3. Bug上报类型特殊处理
        // ═══════════════════════════════════════════════════════════════
        if (jsonObject.type === 'bugreport') {
            reportLog({
                name: 'REVERSE_BUG_REPORT',
                wxid: pipeLineWrapper.wxid,
                content: jsonObject.content
            });
            return;
        }
        
        // ═══════════════════════════════════════════════════════════════
        // 4. 版本信息更新
        // ═══════════════════════════════════════════════════════════════
        if (jsonObject.version) {
            GalaxyVersionCache.setVersion(pipeLineWrapper.wxid, jsonObject.version);
        }
        
        // ═══════════════════════════════════════════════════════════════
        // 5. 分发到对应处理器
        // ═══════════════════════════════════════════════════════════════
        const sendType = jsonObject.type;
        
        if (pipeLineWrapper.workWx) {
            // 企业微信消息处理
            WkMsgHandlerCenter.outBoundMsg(jsonObject, pipeLineWrapper, sendType);
        } else {
            // 微信消息处理
            if (CallBackClassify.THIRD_CALLBACKS.has(jsonObject.type) && jsonObject.status != 0) {
                // 失败的第三条消息延迟处理（等待可能的成功消息）
                const waitTime = apolloConfig.taskFailWaitTime || 3000;
                setTimeout(() => {
                    WxMsgHandlerCenter.outBoundMsg(jsonObject, pipeLineWrapper, sendType);
                }, waitTime);
            } else {
                WxMsgHandlerCenter.outBoundMsg(jsonObject, pipeLineWrapper, sendType);
            }
        }
        
    } catch (error) {
        logUtil.customLog(`[dispatchOutBound] 错误: ${error.message}`, { level: 'error' });
    }
}

module.exports = dispatchOutBound;
```

## 四、三段式消息机制

### 4.1 设计原理

```
┌─────────────────────────────────────────────────────────────────────────┐
│                       三段式消息流程                                     │
└─────────────────────────────────────────────────────────────────────────┘

为什么需要三段式？
- 微信操作是异步的，发送命令后不能立即得到最终结果
- 需要追踪：命令已发送 → 消息已到达 → 操作已完成
- 用于判断任务最终成功/失败，并上报云端

    客户端发送命令                微信服务器处理                返回最终结果
        │                           │                           │
        ▼                           │                           │
┌──────────────┐                    │                           │
│ 第一条消息    │                    │                           │
│ sendmessage  │ ──────────────────►│                           │
│ status=?     │                    │                           │
│ (命令已发送)  │                    │                           │
└──────────────┘                    │                           │
        │                           ▼                           │
        │                  ┌──────────────┐                     │
        │                  │ 微信服务器    │                     │
        │                  │ 处理消息     │                     │
        │                  └──────────────┘                     │
        │                           │                           │
        │                           ▼                           │
        │               ┌──────────────────┐                    │
        │ ◄─────────────│ 第二条消息       │                    │
        │               │ recvmsg          │                    │
        │               │ status=2/3       │                    │
        │               │ (消息已到达服务器)│                    │
        │               └──────────────────┘                    │
        │                           │                           │
        │                           ▼                           │
        │               ┌──────────────────┐                    │
        │ ◄─────────────│ 第三条消息       │──────────────────►│
        │               │ oplog/status     │                    │
        │               │ status=0/1       │                    │
        │               │ (最终操作结果)   │                    │
        │               └──────────────────┘                    │
        │                                                       │
        ▼                                                       ▼
   清理缓存节点                                           上报云端结果
```

### 4.3 消息处理器实现

```javascript
// src/msg-center/dispatch-center/handle/msgHandleBase.js

const logUtil = require('../../../init/log');
const cloudFlowOutBound = require('../dispatch/cloudFlowOutBound');
const CallBackClassify = require('../../core/data-config/callbackClassify');
const GalaxyTaskCache = require('../../core/cache/galaxyTaskCache');

const MsgHandlerCenterBase = {
    // ═══════════════════════════════════════════════════════════════════
    // 消息节点缓存
    // key: `${wxId}:${firstCallbackType}`
    // value: { first, second, third, taskId, createTime }
    // ═══════════════════════════════════════════════════════════════════
    msgResNodeMap: {},
    
    // 任务ID映射
    taskIdMap: {},
    
    /**
     * 处理出站消息
     * @param {Object} msgObj - 消息对象
     * @param {Object} pipeLineWrapper - 管道包装
     * @param {string} sendType - 消息类型
     */
    outBoundMsg(msgObj, pipeLineWrapper, sendType) {
        const wxId = pipeLineWrapper.wxid;
        const key = this.getAssociationKey(msgObj, wxId, sendType);
        
        let node = this.msgResNodeMap[key];
        
        // 检查是否需要重置节点
        if (this.shouldResetNode(sendType, node)) {
            this.msgResNodeMap[key] = null;
            node = null;
        }
        
        // ═══════════════════════════════════════════════════════════════
        // 判断消息阶段并处理
        // ═══════════════════════════════════════════════════════════════
        
        if (this.isFirstMsg(node, sendType)) {
            // 第一条消息：创建节点
            logUtil.customLog(`[msgConvert] [wxid-${wxId}] 第一条消息 type=${sendType}`);
            
            this.msgResNodeMap[key] = {
                first: msgObj,
                taskId: msgObj.taskId,
                createTime: Date.now()
            };
            
            // 处理第一条消息
            this.processFirstMsg(msgObj, pipeLineWrapper);
        }
        else if (this.isSecondMsg(node, sendType)) {
            // 第二条消息：关联到节点
            logUtil.customLog(`[msgConvert] [wxid-${wxId}] 第二条消息 type=${sendType}`);
            
            node.second = msgObj;
            
            // 处理第二条消息
            this.processSecondMsg(msgObj, node, pipeLineWrapper);
        }
        else if (this.isThirdMsg(node, sendType, msgObj)) {
            // 第三条消息：完成处理
            logUtil.customLog(`[msgConvert] [wxid-${wxId}] 第三条消息 type=${sendType} status=${msgObj.status}`);
            
            node.third = msgObj;
            
            // 处理第三条消息
            this.processThirdMsg(msgObj, node, pipeLineWrapper);
            
            // 上报云端
            cloudFlowOutBound(wxId, msgObj, node);
            
            // 清理缓存
            delete this.msgResNodeMap[key];
            if (node.taskId) {
                GalaxyTaskCache.clearTask(node.taskId);
            }
        }
        else {
            // 不属于三段式的消息，直接分发
            this.dispatchNormalMsg(msgObj, pipeLineWrapper);
        }
    },
    
    // ═══════════════════════════════════════════════════════════════════
    // 消息阶段判断
    // ═══════════════════════════════════════════════════════════════════
    
    isFirstMsg(node, sendType) {
        return !node && CallBackClassify.FIRST_CALLBACKS.has(sendType);
    },
    
    isSecondMsg(node, sendType) {
        return node && node.first && !node.second 
            && CallBackClassify.SECOND_CALLBACKS.has(sendType);
    },
    
    isThirdMsg(node, sendType, msgObj) {
        return node && node.first && node.second 
            && CallBackClassify.THIRD_CALLBACKS.has(sendType);
    },
    
    shouldResetNode(sendType, node) {
        // 如果收到新的第一条消息但节点已存在且超时，重置
        if (node && CallBackClassify.FIRST_CALLBACKS.has(sendType)) {
            const expired = Date.now() - node.createTime > 60000; // 1分钟超时
            return expired;
        }
        return false;
    },
    
    // ═══════════════════════════════════════════════════════════════════
    // 关联键生成
    // ═══════════════════════════════════════════════════════════════════
    
    getAssociationKey(msgObj, wxId, sendType) {
        // 默认使用 wxId + firstCallbackType
        const firstType = msgObj.firstCallbackType || sendType;
        return `${wxId}:${firstType}`;
    },
    
    // ═══════════════════════════════════════════════════════════════════
    // 消息处理（由子类实现）
    // ═══════════════════════════════════════════════════════════════════
    
    processFirstMsg(msgObj, pipeLineWrapper) {
        // 子类实现
    },
    
    processSecondMsg(msgObj, node, pipeLineWrapper) {
        // 子类实现
    },
    
    processThirdMsg(msgObj, node, pipeLineWrapper) {
        // 子类实现
    },
    
    dispatchNormalMsg(msgObj, pipeLineWrapper) {
        // 分发到对应的业务处理器
    }
};

module.exports = MsgHandlerCenterBase;
```

### 4.4 微信4.0适配（两段式）

```javascript
// src/msg-center/dispatch-center/handle/wxMsgHandle.js

const MsgHandlerCenterBase = require('./msgHandleBase');
const CallBackClassify = require('../../core/data-config/callbackClassify');
const ConvertServiceList = require('../../business/convertServiceList');

const WxMsgHandlerCenter = {
    ...MsgHandlerCenterBase,
    
    /**
     * 微信4.0使用两段式消息（sendmessage + recvmsg）
     * recvmsg的status=2/3可以直接作为最终结果
     */
    isWx4RecvmsgComplete(node, sendType, msgObj) {
        // 检查是否是微信4.0的两段式消息
        if (node && node.first 
            && CallBackClassify.FIRST_CALLBACKS_NEW.has(node.first.type)
            && sendType === 'recvmsg') {
            
            // 状态转换：status=2 → 成功(0), status=3 → 失败(1)
            const originalStatus = msgObj.status || msgObj.data?.status;
            const convertedStatus = originalStatus === 2 ? 0 : 1;
            
            // 继承 taskId
            if (node.first.taskId && !msgObj.taskId) {
                msgObj.taskId = node.first.taskId;
            }
            
            msgObj.status = convertedStatus;
            msgObj.flowSource = 2;  // 标记为云端任务
            
            return true;
        }
        return false;
    },
    
    /**
     * 重写出站消息处理
     */
    outBoundMsg(msgObj, pipeLineWrapper, sendType) {
        const wxId = pipeLineWrapper.wxid;
        const key = this.getAssociationKey(msgObj, wxId, sendType);
        let node = this.msgResNodeMap[key];
        
        // 检查微信4.0两段式完成
        if (this.isWx4RecvmsgComplete(node, sendType, msgObj)) {
            // 直接作为第三条消息处理
            node.third = msgObj;
            this.processThirdMsg(msgObj, node, pipeLineWrapper);
            require('../dispatch/cloudFlowOutBound')(wxId, msgObj, node);
            delete this.msgResNodeMap[key];
            return;
        }
        
        // 否则按正常三段式处理
        MsgHandlerCenterBase.outBoundMsg.call(this, msgObj, pipeLineWrapper, sendType);
    },
    
    /**
     * 分发普通消息到业务处理器
     */
    dispatchNormalMsg(msgObj, pipeLineWrapper) {
        for (const service of ConvertServiceList) {
            if (service.filter(msgObj)) {
                service.operate(msgObj, pipeLineWrapper);
                break;
            }
        }
    }
};

module.exports = WxMsgHandlerCenter;
```

```javascript
            // 2. 处理媒体文件（如果是URL需要下载）

            const mediaTypes = [3, 43, 47, 49];  // 图片、视频、表情、文件
            
            if (mediaTypes.includes(weChatMsgType) && content?.startsWith('http')) {
                // 下载文件到本地

                const fileInfo = await downloadAsyncReturnFileInfo(content, filename);

            const clientTask = ClientTaskFactory.createSendMsgTask({
```

### 5.3 逆向响应处理器

```javascript
// src/msg-center/business/convert-service/loginService.js

const AsyncLock = require('async-lock');
const logUtil = require('../../../init/log');
const RegistryConfig = require('../../core/registry-config/index');
const MqttHelper = require('../../core/mq/mqttHelper');
const MqttSendService = require('../../dispatch-center/mqttSend');
const frontSend = require('../../dispatch-center/frontSend');
const IpcConstant = require('../../core/data-config/ipcConstant');
const { httpFetch } = require('../../../common/fetch');
const ApplicationEnv = require('../../core/data-config/applicationEnv');

const lock = new AsyncLock();
const frontSendService = frontSend.getInstance();

const LoginService = {
    /**
     * 过滤：处理登录成功消息
     */
    filter(clientRecord) {
        return clientRecord.type === "login" 
            && clientRecord.status === 0      // 成功
            && !clientRecord.busy             // 非忙碌
            && (clientRecord.data || clientRecord.login_user);  // 有用户数据
    },
    
    /**
     * 操作：处理登录
     */
    operate(clientRecord, pipeLineWrapper) {
        const workWxFlag = pipeLineWrapper.workWx;
        const userInfo = clientRecord.data || clientRecord.login_user;
        const ownerWxId = userInfo.wxid || userInfo.username;
        
        logUtil.customLog(`[wxid-${ownerWxId}] [loginService] 收到登录消息 workWx=${workWxFlag}`);
        
        this.handleLogin(pipeLineWrapper, userInfo, ownerWxId, workWxFlag);
    },
    
    /**
     * 处理登录逻辑
     */
    async handleLogin(pipeLineWrapper, userInfo, ownerWxId, workWxFlag) {
        // 使用锁防止同一账号并发登录
        await lock.acquire(`loginLock-${ownerWxId}`, async (done) => {
            try {
                const registry = RegistryConfig.getRegistryByKey(pipeLineWrapper.id, "id");
                
                if (!registry) {
                    logUtil.customLog(`[wxid-${ownerWxId}] 找不到注册信息`, { level: 'error' });
                    done();
                    return;
                }
                
                // 绑定wxid
                registry.wxid = String(ownerWxId);
                pipeLineWrapper.wxid = String(ownerWxId);
                
                if (registry.sendToCloudFlag) {
                    // 重复登录（已经登录过）
                    await this.handleReLogin(registry, userInfo, ownerWxId, workWxFlag);
                } else {
                    // 首次登录
                    await this.handleFirstLogin(registry, pipeLineWrapper, userInfo, ownerWxId, workWxFlag);
                }
                
                done();
            } catch (error) {
                logUtil.customLog(`[wxid-${ownerWxId}] 登录处理错误: ${error.message}`, { level: 'error' });
                done();
            }
        });
    },
    
    /**
     * 首次登录
     */
    async handleFirstLogin(registry, pipeLineWrapper, userInfo, ownerWxId, workWxFlag) {
        logUtil.customLog(`[wxid-${ownerWxId}] 首次登录开始`);
        
        // 1. 绑定用户信息
        this.bindUserInfo(registry, userInfo, workWxFlag);
        
        // 2. 设置管道可用
        pipeLineWrapper.available = IpcConstant.AVAILABLE;
        
        // 3. 调用云端注册接口
        const registerData = this.buildRegisterData(ownerWxId, userInfo, workWxFlag);
        await httpFetch({
            url: ApplicationEnv.logicRegisteredCloud,
            data: registerData
        });
        
        logUtil.customLog(`[wxid-${ownerWxId}] 云端注册成功`);
        
        // 4. 建立MQTT连接
        await MqttHelper.connectMqtt({ wxId: ownerWxId, workWx: workWxFlag });
        
        logUtil.customLog(`[wxid-${ownerWxId}] MQTT连接成功`);
        
        // 5. 发送登录消息到云端
        const loginMsg = this.buildLoginMessage(ownerWxId, userInfo, workWxFlag);
        MqttSendService.sendMessage(ownerWxId, loginMsg);
        
        // 6. 通知前端刷新
        frontSendService.sendGetAllConfig();
        
        // 7. 更新登录状态
        registry.sendToCloudFlag = IpcConstant.SEND_LOGIN_TO_CLOUD;
        registry.loginTime = Date.now();
        
        logUtil.customLog(`[wxid-${ownerWxId}] 首次登录完成`);
    },
    
    /**
     * 绑定用户信息
     */
    bindUserInfo(registry, userInfo, workWxFlag) {
        registry.wxInfo = {
            nickname: userInfo.nickname || userInfo.name,
            avatar: userInfo.avatar || userInfo.headimgurl,
            alias: userInfo.alias,
            sex: userInfo.sex,
            signature: userInfo.signature
        };
        registry.workWx = workWxFlag;
    },
    
    buildRegisterData(wxId, userInfo, workWxFlag) {
        return {
            username: wxId,
            nickname: userInfo.nickname || userInfo.name,
            avatar: userInfo.avatar || userInfo.headimgurl,
            workWx: workWxFlag ? 1 : 0,
            // ...其他字段
        };
    },
    
    buildLoginMessage(wxId, userInfo, workWxFlag) {
        return {
```

### 6.1 连接管理（RegistryConfig）

```javascript
// src/msg-center/core/registry-config/index.js

const logUtil = require('../../../init/log');
const Clibrary = require('../reverse/dll/clibrary');
const registryList = require('./registryList').getInstance();

const RegistryConfig = {
    // ═══════════════════════════════════════════════════════════════════
    // 基础操作（委托给registryList）
    // ═══════════════════════════════════════════════════════════════════
    add: registryList.addRegistry,
    remove: registryList.removeRegistry,
    getRegistryByKey: registryList.getRegistryByKey,
    getAllConfig: registryList.getAllConfig,
    getRegistryList: registryList.getRegistryList,
    
    // ═══════════════════════════════════════════════════════════════════
    // 清除MQTT连接
    // ═══════════════════════════════════════════════════════════════════
    forceClearMqtt(wxId, registry) {
        registry?.mqttClient?.end?.(true, {}, () => {
            if (registry) {
                registry.mqttClient = null;
            }
            logUtil.customLog(`[wxid-${wxId}] MQTT连接已清除`);
        });
    },
    
    // ═══════════════════════════════════════════════════════════════════
    // 移除管道（清理所有连接）
    // ═══════════════════════════════════════════════════════════════════
    removePipe(processId, pipeCode) {
        try {
            let registry = this.getRegistryByKey(processId, 'id');
            
            logUtil.customLog(`[removePipe] [关闭IPC及MQTT连接] [wxid-${registry?.wxid}] processId:[${processId}]`);
            
            if (!registry?.pipeLineWrapper?.pipeCode && !pipeCode) {
                return;
            }
            
            // 仅关闭IPC连接
            if (!registry?.pipeLineWrapper?.pipeCode && pipeCode) {
                Clibrary.IpcClientClose(pipeCode);
                return;
            }
            
            // 关闭IPC连接
            Clibrary.IpcClientClose(registry.pipeLineWrapper.pipeCode);
            
            // 清除MQTT连接
            this.forceClearMqtt(registry?.wxid, registry);
            
            // 移除注册信息
            this.remove(registry);
            
        } catch (error) {
            logUtil.customLog(`[removePipe] 错误: ${error.message}`, { level: 'error' });
        }
    },
    
    // ═══════════════════════════════════════════════════════════════════
    // 移除所有连接
    // ═══════════════════════════════════════════════════════════════════
    removeAll() {
        const registryList = this.getRegistryList();
        for (let registry of registryList) {
            this.removePipe(registry.id);
        }
    },
    
    // ═══════════════════════════════════════════════════════════════════
    // 更新心跳时间
    // ═══════════════════════════════════════════════════════════════════
    setPongTime(id) {
        const registry = this.getRegistryByKey(id, 'id');
        if (registry?.pipeLineWrapper) {
            registry.pipeLineWrapper.lastPongTime = Date.now();
        }
    },
    
    // ═══════════════════════════════════════════════════════════════════
    // 获取所有微信ID
    // ═══════════════════════════════════════════════════════════════════
    getAllWxId() {
        return this.getRegistryList()
            .filter(r => r.wxid)
            .map(r => r.wxid);
    },
    
    // ═══════════════════════════════════════════════════════════════════
    // 获取当前所有进程ID
    // ═══════════════════════════════════════════════════════════════════
    getCurrProcessIds() {
        return this.getRegistryList()
            .map(r => r.pipeLineWrapper?.processId)
            .filter(Boolean);
    }
};

module.exports = RegistryConfig;
```

### 7.1 队列设计

```javascript
// src/msg-center/core/queue/mqTask.js

const logUtil = require('../../../init/log');

// 按wxId分组的任务队列
const wxTaskQueues = new Map();

// 每个wxId的处理状态
const processingStatus = new Map();

/**
 * 延迟执行任务（按wxId排队）
 * @param {string} wxId - 微信ID
 * @param {Function} taskFn - 任务函数
 */
function executeWithDelay(wxId, taskFn) {
    // 获取或创建队列
    if (!wxTaskQueues.has(wxId)) {
        wxTaskQueues.set(wxId, []);
    }
    
    const queue = wxTaskQueues.get(wxId);
    queue.push(taskFn);
    
    // 如果是第一个任务，开始处理
    if (!processingStatus.get(wxId)) {
        processQueue(wxId);
    }
}

/**
 * 处理队列
 */
async function processQueue(wxId) {
    const queue = wxTaskQueues.get(wxId);
    
    if (!queue || queue.length === 0) {
        processingStatus.set(wxId, false);
        return;
    }
    
    processingStatus.set(wxId, true);
    
    while (queue.length > 0) {
        const taskFn = queue.shift();
        
        try {
            await taskFn();
        } catch (error) {
            logUtil.customLog(`[wxid-${wxId}] 队列任务执行错误: ${error.message}`, { level: 'error' });
        }
        
        // 任务间隔100ms
        await new Promise(resolve => setTimeout(resolve, 100));
    }
    
    processingStatus.set(wxId, false);
}

module.exports = executeWithDelay;
```
