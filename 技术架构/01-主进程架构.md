## 一、主进程概述

### 1.1 主进程职责


### 1.2 主进程入口

```javascript
// src/electron.js - 应用入口文件

// 记录启动时间（用于计算启动耗时）
global.readyStartTime = Date.now();

// 引入核心模块
const { mode, type } = require('./common/urls');        // 环境配置
const initLog = require('./init/initLog');              // 日志初始化
const addAppEvent = require('./event/app');             // 应用事件
const addIpcEvent = require('./event/ipc');             // IPC事件
const addStoreEvent = require('./event/store');         // 存储事件
const AppStart = require('./msg-center/start/appStart'); // 消息中心

// 初始化日志系统（必须最先执行）
initLog();

function bootstrap() {
    // 1. 添加应用事件（单实例检查在这里）
    let isQuit = addAppEvent();
    if (isQuit) {
        return;  // 已有实例在运行，直接退出
    }
    
    // 2. 注册IPC事件监听（与渲染进程通信）
    addIpcEvent();
    
    // 3. 注册Store事件（持久化存储）
    addStoreEvent();
    
    // 4. 启动消息中心（核心业务）
    AppStart.run();
}

bootstrap();
```

## 三、应用事件处理

### 3.1 完整事件注册

```javascript
// src/event/app.js

const path = require('path');
const { app, dialog } = require('electron');
const log = require('electron-log');

const { judgeProcessExist, getAppConfig, sendToRenderMsg } = require('../utils');
const { registerGlobalShortcut, unregisterGlobalShortcut } = require('../common/shortcut');
const { crashOrErrorReport, processErrorMonitor } = require('../common/monitor');
const store = require('../common/store');
const initWindow = require('../init/window');
const { initRegedit } = require('./regedit');
const RegistryConfig = require('../msg-center/core/registry-config/index');
const { stopBasicService, stopBasicService64, runInjectWly } = require('../common/inject');
const { judgeHostsAndRunBat } = require('../utils');

const appConfig = getAppConfig();
const isDev = !app.isPackaged;
const isWin = process.platform === 'win32';

module.exports = () => {
    // ═══════════════════════════════════════════════════════════════════
    // 单实例锁 - 确保只有一个应用实例运行
    // ═══════════════════════════════════════════════════════════════════
    const gotTheLock = app.requestSingleInstanceLock();
    if (!gotTheLock) {
        app.quit();
        return true;  // 返回true表示需要退出
    }
    
    // 处理第二个实例启动（用户双击快捷方式时）
    app.on('second-instance', (event, argv) => {
        log.info('SECOND INSTANCE');
        
        // 聚焦到已存在的窗口
        const { mainWindow } = app;
        if (mainWindow) {
            if (mainWindow.isMinimized()) {
                mainWindow.restore();
            }
            mainWindow.focus();
            mainWindow.show();
        }
        
        // Windows下处理URL协议唤醒
        if (isWin) {
            const commands = argv.slice();
            const activeUrl = decodeURI(commands.pop());
            log.info('AWAKEN', activeUrl);
            sendToRenderMsg('web-awaken', activeUrl);
        }
    });
    
    // ═══════════════════════════════════════════════════════════════════
    // 命令行参数配置
    // ═══════════════════════════════════════════════════════════════════
    // 增加V8堆内存限制到6GB
    app.commandLine.appendSwitch('js-flags', '--max-old-space-size=6144');
    // 禁用HTTP缓存
    app.commandLine.appendSwitch('--disable-http-cache');
    
    // ═══════════════════════════════════════════════════════════════════
    // 应用就绪事件 - 核心初始化
    // ═══════════════════════════════════════════════════════════════════
    app.on('ready', async () => {
        log.info('READY');
        
        // 1. 注册URL协议（wxzsclient://）
        if (appConfig.protocol) {
            setProtocol(appConfig.protocol);
        }
        
        // 2. 初始化主窗口
        initWindow();
        
        // 3. 初始化注册表配置
        initRegedit();
        
        // 4. 检查hosts文件（网络配置）
        judgeHostsAndRunBat();
    });
    
    // ═══════════════════════════════════════════════════════════════════
    // macOS 应用激活事件
    // ═══════════════════════════════════════════════════════════════════
    app.on('activate', () => {
        if (app.mainWindow === null) {
            initWindow();
        }
    });
    
    // ═══════════════════════════════════════════════════════════════════
    // 应用退出事件 - 清理资源
    // ═══════════════════════════════════════════════════════════════════
    app.on('will-quit', () => {
        // 注销全局快捷键
        unregisterGlobalShortcut();
    });
    
    app.on('before-quit', event => {
        event.preventDefault();
        
        // 停止32位逆向服务
        stopBasicService(() => {
            RegistryConfig.removeAll();
            app.exit();
        });
        
        // 停止64位逆向服务
        stopBasicService64(() => {
            RegistryConfig.removeAll();
            app.exit();
        });
    });
    
    // ═══════════════════════════════════════════════════════════════════
    // 窗口焦点事件 - 快捷键管理
    // ═══════════════════════════════════════════════════════════════════
    app.on('browser-window-blur', (event, win) => {
        if (win === app.mainWindow) {
            unregisterGlobalShortcut();
        }
    });
    
    app.on('browser-window-focus', (event, win) => {
        if (win === app.mainWindow) {
            registerGlobalShortcut();
        }
    });
    
    // ═══════════════════════════════════════════════════════════════════
    // 渲染进程崩溃事件
    // ═══════════════════════════════════════════════════════════════════
    app.on('render-process-gone', event => {
        crashOrErrorReport('RendererProcessCrashed', event);
    });
    
    // ═══════════════════════════════════════════════════════════════════
    // URL协议唤醒事件（macOS）
    // ═══════════════════════════════════════════════════════════════════
    app.on('open-url', (event, url) => {
        log.info('OPEN WITH', url);
        sendToRenderMsg('web-awaken', url);
    });
    
    // ═══════════════════════════════════════════════════════════════════
    // 启动进程错误监控
    // ═══════════════════════════════════════════════════════════════════
    processErrorMonitor();
};
```

### 3.2 URL协议处理详解

```javascript
// URL协议注册
function setProtocol(protocol) {
    // 先处理启动参数中的协议URL
    handleArgv(process.argv);
    
    // 构建注册参数
    const args = [];
    if (isDev) {
        // 开发模式需要指定脚本路径
        args.push(path.resolve(process.argv[1]));
    }
    args.push('--');
    
    // 注册为默认协议处理程序
    // 例如：wxzsclient://open?emailPrefix=xxx
    app.setAsDefaultProtocolClient(protocol, process.execPath, args);
}

// 解析协议URL参数
function handleArgv(argv) {
    if (argv.length) {
        const reg = new RegExp(`^${appConfig.protocol}://`);
        for (let i = argv.length; i--;) {
            const command = argv[i];
            if (reg.test(command)) {
                log.info('PROTOCOL', command);
                handleProtocol(command);
                break;
            }
        }
    }
}

// 处理协议参数
function handleProtocol(url) {
    try {
        // 解析URL：wxzsclient://action?emailPrefix=xxx&other=yyy
        const urlobj = new URL(url);
        const keyValueArr = urlobj.search.substr(1).split('&');
        
        // 提取emailPrefix用于自动填充登录用户名
        for (let keyValue of keyValueArr) {
            if (keyValue.indexOf('emailPrefix') !== -1) {
                const [key, value] = keyValue.split('=');
                if (key === 'emailPrefix' && value) {
                    // 存储用于登录自动完成
                    store.setAutoCompleteInfo({ username: value });
                }
                break;
            }
        }
    } catch (error) {
        log.error('PROTOCOL ERROR', error);
    }
}
```

## 四、窗口管理

### 4.1 窗口创建详解

```javascript
// src/init/window.js

const { app, session, dialog, Menu, MenuItem } = require('electron');
const log = require('electron-log');
const logUtil = require('../init/log');
const path = require('path');
const { type, offlineUrl } = require('../common/urls');
const { sendToRenderMsg, getHostName, getAppConfig, debounce } = require('../utils');
const { reportLog } = require('../init/habo');
const store = require('../common/store');
const createStateWindow = require('../common/createStateWindow');
const { registerGlobalShortcut } = require('../common/shortcut');
const { getScreenScaleFactor } = require('../common/screenAdapter');
const { getLoadUrlAsync, getLoadUrl } = require('../common/loadUrl');
const { setRecordInfo } = require('../common/recordActivityInfo');
const { stopJava } = require('../common/inject');
const initUpdater = require('../event/updater');
const remoteMain = require('@electron/remote/main');

// 初始化remote模块（允许渲染进程调用主进程API）
remoteMain.initialize();

const isDev = !app.isPackaged;
const appConfig = getAppConfig();
let updateFlag = false;  // 是否因更新而关闭
const searchReg = /^\?s=\S{12}$/;

module.exports = () => {
    // ═══════════════════════════════════════════════════════════════════
    // 1. 注册全局快捷键
    // ═══════════════════════════════════════════════════════════════════
    registerGlobalShortcut();
    
    // ═══════════════════════════════════════════════════════════════════
    // 2. 创建主窗口（带状态保存功能）
    // ═══════════════════════════════════════════════════════════════════
    const mainWindow = app.mainWindow = createStateWindow({
        width: 1060,
        height: 680,
        minWidth: 1060,
        minHeight: 680,
        webPreferences: {
            sandbox: false,                                    // 禁用沙箱（允许访问Node.js）
            preload: path.resolve(__dirname, '../../extraResources/load/inject.js'),  // 预加载脚本
            nodeIntegration: true,                            // 启用Node.js集成
            webviewTag: true,                                 // 允许使用webview标签
            allowRunningInsecureContent: true,                // 允许不安全内容
            enableRemoteModule: true,                         // 启用remote模块
            contextIsolation: false,                          // 禁用上下文隔离
        },
        show: false,                                          // 初始不显示（等加载完成）
        icon: path.resolve(__dirname, '../../extraResources/icon.ico'),
    });
    
    // 为窗口启用remote模块
    remoteMain.enable(mainWindow.webContents);
    
    // 移除默认菜单
    mainWindow.removeMenu();
    
    // ═══════════════════════════════════════════════════════════════════
    // 3. 加载页面URL
    // ═══════════════════════════════════════════════════════════════════
    getLoadUrlAsync().then(url => {
        const { mainWindow } = app;
        log.info('LOAD FIRST TIME', url);
        mainWindow.loadURL(url);
        
        // 监听CAS登录成功后的重定向
        session.defaultSession.webRequest.onBeforeRedirect({ urls: [] }, details => {
            const { url, redirectURL } = details;
            const host = getHostName(url);
            const redirectHost = getHostName(redirectURL);
            log.info('SESSION onBeforeRedirect', host, redirectHost);
            
            // 检测到登录成功（重定向到公司域名）
            if (['www.baijiahulian.com', 'www.baijia.com'].includes(redirectHost)) {
                const nextUrl = getLoadUrl();
                log.info('LOAD AFTER SUCCESS', nextUrl);
                app.mainWindow.loadURL(nextUrl);
            }
        });
        
        // 开发/测试环境打开DevTools
        if (isDev || type === 'vt' || type === 'test') {
            mainWindow.webContents.openDevTools({
                mode: 'undocked',    // 独立窗口
                activate: false,     // 不抢焦点
            });
        }
    });
    
    // ═══════════════════════════════════════════════════════════════════
    // 4. 网络错误处理
    // ═══════════════════════════════════════════════════════════════════
    session.defaultSession.webRequest.onErrorOccurred({ urls: [] }, details => {
        const { resourceType, url, error } = details;
        if (resourceType === 'mainFrame' && !searchReg.test(new URL(url).search)) {
            log.info('SESSION onErrorOccurred', url, error);
        }
    });
    
    // ═══════════════════════════════════════════════════════════════════
    // 5. 右键菜单配置
    // ═══════════════════════════════════════════════════════════════════
    const ctxMenu = new Menu();
    ctxMenu.append(new MenuItem({
        label: '粘贴',
        role: 'paste',
        click() {
            log.info('MENU PASTE CLICKED');
        },
    }));
    
    mainWindow.webContents.on('context-menu', (e, params) => {
        ctxMenu.popup(mainWindow, params.x, params.y);
    });
    
    // ═══════════════════════════════════════════════════════════════════
    // 6. DevTools状态同步
    // ═══════════════════════════════════════════════════════════════════
    mainWindow.webContents.on('devtools-opened', () => {
```

```javascript
        return await convertBase64ToFile(arg);

        app.relaunch();

        event.returnValue = store.getIsGray();

        }
    });

            app.mainWindow.reload();
        }
    });
};
```

### 5.2 IPC通信方式对比

| 方式 | 方法 | 特点 | 适用场景 |
|-----|-----|-----|---------|
| 同步 | `ipc.on` + `event.returnValue` | 阻塞渲染进程 | 获取配置等快速操作 |
| 异步单向 | `ipc.on` | 不需要返回值 | 启动服务、打开文件 |
| 异步Promise | `ipc.answerRenderer` | 返回Promise | 文件操作、网络请求 |
| 主进程推送 | `webContents.send` | 主动推送给渲染进程 | 状态更新、通知 |

```javascript
// src/common/monitor.js

const log = require('electron-log');

// ═══════════════════════════════════════════════════════════════════
// 崩溃错误上报
// ═══════════════════════════════════════════════════════════════════
function crashOrErrorReport(type, event) {
    const errorInfo = {
        name: type,
        event: JSON.stringify(event),
        timestamp: Date.now(),
        platform: process.platform,
        nodeVersion: process.version,
    };
    
    log.error(`[${type}]`, errorInfo);
    reportLog(errorInfo);
}

// ═══════════════════════════════════════════════════════════════════
// 进程错误监控
// ═══════════════════════════════════════════════════════════════════
function processErrorMonitor() {
    // 未捕获的同步异常
    process.on('uncaughtException', (error) => {
        log.error('Uncaught Exception:', error);
        crashOrErrorReport('UncaughtException', {
            name: error.name,
            message: error.message,
            stack: error.stack
        });
    });
    
    // 未处理的Promise拒绝
    process.on('unhandledRejection', (reason, promise) => {
        log.error('Unhandled Rejection:', reason);
        crashOrErrorReport('UnhandledRejection', {
            reason: String(reason),
            promise: String(promise)
        });
    });
    
    // 进程警告
    process.on('warning', (warning) => {
        log.warn('Process Warning:', warning);
    });
}

module.exports = {
    crashOrErrorReport,
    processErrorMonitor
```

```javascript
// src/init/log.js

const log = require('electron-log');
const slsLog = require('./slsLog');
const { reportLog } = require('./habo');

// 日志级别配置
log.transports.file.level = 'info';
log.transports.console.level = 'debug';

// 自定义日志格式
log.transports.file.format = '[{y}-{m}-{d} {h}:{i}:{s}.{ms}] [{level}] {text}';

const logUtil = {
    /**
     * 自定义日志输出
     * @param {string} message - 日志消息
     * @param {Object} options - 选项
     * @param {string} options.level - 日志级别 (info/warn/error)
     * @param {string} options.errorKey - 错误标识（用于上报）
     */
    customLog(message, options = {}) {
        const { level = 'info', errorKey } = options;
        
        const timestamp = new Date().toISOString();
        const logMessage = `[${timestamp}] ${message}`;
        
        // 控制台输出
        if (level === 'error') {
            log.error(logMessage);
        } else if (level === 'warn') {
            log.warn(logMessage);
        } else {
            log.info(logMessage);
        }
        
        // 错误级别上报到阿里云SLS
        if (level === 'error' || level === 'warn') {
            slsLog.pushLog(message, { level, errorKey });
        }
        
        // 带errorKey的错误上报到Habo
        if (errorKey) {
            reportLog({
                name: errorKey,
                message: message,
                level: level
            });
        }
    },
    
    // 便捷方法
    info(message) {
        this.customLog(message, { level: 'info' });
    },
    
    warn(message) {
        this.customLog(message, { level: 'warn' });
    },
    
    error(message, errorKey) {
        this.customLog(message, { level: 'error', errorKey });
    }
};
```

### 7.3 崩溃检测与恢复

```javascript
// src/common/recordActivityInfo.js

const fs = require('fs');
const path = require('path');
const { app } = require('electron');
const { reportLog } = require('../init/habo');

const recordPath = path.join(app.getPath('userData'), 'activity.json');

// ═══════════════════════════════════════════════════════════════════
// 记录活动信息
// ═══════════════════════════════════════════════════════════════════
function setRecordInfo(status) {
    // status: 0 = 运行中, 1 = 正常退出
    const record = {
        status,
        lastActiveTime: Date.now(),
        pid: process.pid
    };
    
    try {
        fs.writeFileSync(recordPath, JSON.stringify(record));
    } catch (error) {
        console.error('Write record error:', error);
    }
}

// ═══════════════════════════════════════════════════════════════════
// 检测上次是否崩溃并上报
// ═══════════════════════════════════════════════════════════════════
function judgeCrashAndReport() {
    try {
        if (!fs.existsSync(recordPath)) {
            setRecordInfo(0);  // 首次运行
            return;
        }
        
        const recordStr = fs.readFileSync(recordPath, 'utf-8');
        const record = JSON.parse(recordStr);
        
        // 如果上次状态不是正常退出，说明发生了崩溃
        if (record.status !== 1) {
            reportLog({
                name: 'CRASH_DETECTED',
                lastActiveTime: record.lastActiveTime,
                lastPid: record.pid
            });
        }
        
        // 更新为运行中状态
        setRecordInfo(0);
        
    } catch (error) {
        console.error('Judge crash error:', error);
        setRecordInfo(0);
    }
}

module.exports = {
    setRecordInfo,
    judgeCrashAndReport
};
```

### 8.2 关键节点耗时

| 节点 | 预期耗时 | 说明 |
|-----|---------|-----|
| 日志初始化 | <10ms | 同步操作 |
| 事件注册 | <20ms | 纯注册，不执行 |
| WebSocket启动 | ~50ms | 端口扫描 |
| IPC扫描启动 | <10ms | 启动定时器 |
| 窗口创建 | ~100ms | 包含预加载 |
| 页面加载 | 500-2000ms | 取决于网络 |
| 窗口显示 | <50ms | 渲染完成后 |

### 8.3 调试关键点

```javascript
// 添加启动时间点日志
console.time('bootstrap');
bootstrap();
console.timeEnd('bootstrap');

// 查看electron-log日志位置
// Windows: %USERPROFILE%\AppData\Roaming\<app name>\logs\
// macOS: ~/Library/Logs/<app name>/

// 关键日志关键字
"READY"              // 应用就绪
"LOAD FIRST TIME"    // 首次加载URL
"startSuccessCost"   // 启动耗时
"IPC-START"          // IPC扫描启动
"WebSocket server"   // WS服务启动
```
