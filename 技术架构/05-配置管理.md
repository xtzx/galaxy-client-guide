# 配置管理详解

> 多环境配置与云端动态配置

## 一、配置体系概述

### 1.1 配置分类

```
┌─────────────────────────────────────────────────────────────────────────┐
│                       配置体系架构                                       │
└─────────────────────────────────────────────────────────────────────────┘

    ┌─────────────────────────────────────────────────────────────┐
    │                    配置来源                                  │
    │                                                             │
    │  ┌─────────────┐  ┌─────────────┐  ┌─────────────────────┐ │
    │  │ 静态配置     │  │ 环境配置    │  │ 动态配置（Apollo）  │ │
    │  │             │  │             │  │                     │ │
    │  │ • package.json│ │ • 开发环境  │  │ • 任务失败等待时间  │ │
    │  │ • builder.yml │ │ • 测试环境  │  │ • 群发消息间隔     │ │
    │  │ • config/*.yml│ │ • 生产环境  │  │ • MQTT过期时间     │ │
    │  │             │  │             │  │                     │ │
    │  └──────┬──────┘  └──────┬──────┘  └──────────┬──────────┘ │
    │         │                │                    │            │
    │         └────────────────┼────────────────────┘            │
    │                          │                                 │
    │                          ▼                                 │
    │              ┌─────────────────────┐                       │
    │              │   ApplicationEnv    │                       │
    │              │   (统一配置入口)     │                       │
    │              └─────────────────────┘                       │
    │                                                             │
    └─────────────────────────────────────────────────────────────┘
```

### 1.2 配置优先级

```
优先级从高到低：

1. 云端 Apollo 配置（运行时动态获取）
2. 环境变量（ELECTRON_NODE_ENV）
3. 应用环境配置（applicationProd/Qa/Rd.js）
4. 基础配置（applicationBase.js）
5. 构建配置（package.json / builder.yml）
```

## 三、多环境配置

### 3.1 环境选择逻辑

```javascript
// src/msg-center/core/application-config/index.js

const applicationQa = require('./applicationQa');
const applicationRd = require('./applicationRd');
const applicationProd = require('./applicationProd');
const log = require('electron-log');
const { type } = require('../../../common/urls');

// 环境类型获取优先级：
// 1. 环境变量 ELECTRON_NODE_ENV
// 2. URL配置中的 type
let env = process.env.ELECTRON_NODE_ENV || type;

log.info(`ELECTRON_NODE_ENV: ${env}`);

let application;

if (env === 'dev') {
    // 开发环境
    application = { ...applicationRd };
}
else if (env === 'prod') {
    // 生产环境
    application = { ...applicationProd };
}
else {
    // 测试环境（默认）
    application = { ...applicationQa };
}

module.exports = application;
```

## 四、Apollo 云端配置

### 4.1 配置项说明

| 配置键 | 说明 | 默认值 | 用途 |
|-------|------|-------|-----|
| `client.wxzs.fail.task.time` | 任务失败等待时间 | 3000ms | 失败重试前等待 |
| `client.wxzs.wait.task.time` | 群发消息间隔 | 200ms | 防止发送过快 |
| `client.wxzs.mqtt.task.expire.hours` | MQTT任务过期时间 | 3小时 | 过期任务不执行 |

### 4.2 实现代码

```javascript
// src/msg-center/core/utils/getApolloConfig.js

const ApplicationEnv = require("../../core/application-config/index");
const logUtil = require("../../../init/log");
const { httpFetch } = require("../../../common/fetch");

const getApolloConfigUrl = ApplicationEnv["getApolloConfigUrl"];

// ═══════════════════════════════════════════════════════════════════
// Apollo配置对象（带默认值）
// ═══════════════════════════════════════════════════════════════════
const apolloConfig = {
    taskFailWaitTime: ApplicationEnv["fail.galaxy.task.status"],  // 默认30000ms
    taskDefaultWaitTime: 200,    // 默认200ms
    mqttTaskExpireHours: 3,      // 默认3小时
    initStatus: false            // 初始化状态
};

// ═══════════════════════════════════════════════════════════════════
// 获取任务失败等待时间
// ═══════════════════════════════════════════════════════════════════
const getTaskFailTime = async () => {
    try {
        const res = await httpFetch({
            url: getApolloConfigUrl,
            data: { key: "client.wxzs.fail.task.time" },
            method: 'get'
        });

        if (res) {
            apolloConfig.taskFailWaitTime = +JSON.parse(res)?.data || 3000;
        }
        apolloConfig.initStatus = true;

    } catch (e) {
        logUtil.customLog(
            `[client.wxzs.fail.task.time] 获取阿波罗配置异常: ${JSON.stringify(e)}`
        );
    }
};

// ═══════════════════════════════════════════════════════════════════
// 获取任务等待时间（群发间隔）
// ═══════════════════════════════════════════════════════════════════
const getTaskWaitTime = async () => {
    try {
        const res = await httpFetch({
            url: getApolloConfigUrl,
            data: { key: "client.wxzs.wait.task.time" },
            method: 'get'
        });

        if (res) {
            apolloConfig.taskDefaultWaitTime = +JSON.parse(res)?.data || 200;
        }
        apolloConfig.initStatus = true;

    } catch (e) {
        logUtil.customLog(
            `[client.wxzs.wait.task.time] 获取阿波罗配置异常: ${JSON.stringify(e)}`
        );
    }
};

// ═══════════════════════════════════════════════════════════════════
// 获取MQTT任务过期时间
// ═══════════════════════════════════════════════════════════════════
const getMqttTaskExpireHours = async () => {
    try {
        const res = await httpFetch({
            url: getApolloConfigUrl,
            data: { key: "client.wxzs.mqtt.task.expire.hours" },
            method: 'get'
        });

        if (res) {
            apolloConfig.mqttTaskExpireHours = +JSON.parse(res)?.data || 3;
        }
        apolloConfig.initStatus = true;

    } catch (e) {
        logUtil.customLog(
            `[client.wxzs.mqtt.task.expire.hours] 获取阿波罗配置异常: ${JSON.stringify(e)}`
        );
    }
};

// ═══════════════════════════════════════════════════════════════════
// 初始化配置（应用启动时调用）
// ═══════════════════════════════════════════════════════════════════
const initApolloConfig = async () => {
    if (apolloConfig.initStatus) {
        logUtil.customLog(`[apolloInitSuccess] ${JSON.stringify(apolloConfig)}`);
        return;
    }

    // 并行获取所有配置
    getTaskFailTime();
    getTaskWaitTime();
    getMqttTaskExpireHours();
};

// 模块加载时自动初始化
initApolloConfig();

module.exports = apolloConfig;
```

### 4.3 使用示例

```javascript
// 使用Apollo配置
const apolloConfig = require('../core/utils/getApolloConfig');

// 在MQTT消息接收中使用
const expireHours = apolloConfig.mqttTaskExpireHours || 3;
const expireTime = expireHours * 60 * 60 * 1000;

if (dataInfo.createTime * 1000 < Date.now() - expireTime) {
    logUtil.customLog(`任务过期（${expireHours}小时），跳过执行`);
    return;
}

// 在消息发送中使用
await sleep(apolloConfig.taskDefaultWaitTime);
```

### 5.1 配置项汇总

| 配置项 | 来源 | 说明 |
|-------|------|-----|
| `instanceId` | 环境配置 | MQTT实例ID |

| `robotCloudConfig` | 环境配置 | 机器人配置API |

| `sls.expiration.url` | 环境配置 | SLS STS凭证API |

| `taskDefaultWaitTime` | Apollo | 群发消息间隔 |
