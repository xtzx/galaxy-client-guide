| 主进程 ↔ 云端 | HTTP/HTTPS | 443 | API调用、登录注册 | JSON |
| 主进程 ↔ OSS | HTTPS | 443 | 文件上传、下载 | Binary |

| 主进程 ↔ 逆向服务 | Named Pipe | \\.\pipe\xxx | 任务执行、消息接收 | JSON |

## 二、MQTT通信

### 2.1 连接管理

```javascript
// src/msg-center/core/mq/mqttClientBase.js

const mqtt = require("mqtt");
const RegistryConfig = require("../registry-config/index");
const logUtil = require("../../../init/log");

/**
 * 初始化MQTT客户端
 * @param {string} wxId - 微信ID
 * @param {Object} params - 连接参数
 * @param {Function} onRecieveMsg - 消息回调
 */
const initMqttClient = (wxId, params, onRecieveMsg) => {
    logUtil.customLog(`[wxid-${wxId}] [mqttClientBase] 开始初始化MQTT`);
    
    if (!wxId) {
        return;
    }
    
    // ═══════════════════════════════════════════════════════════════════
    // 1. 检查现有连接
    // ═══════════════════════════════════════════════════════════════════
    const registry = RegistryConfig.getRegistryByKey(wxId, "wxid");
    if (!registry) {
        logUtil.customLog(`[wxid-${wxId}] 该微信已退出，跳过MQTT初始化`);
        return;
    }
    
    // 检查是否已有有效连接
    if (registry.mqttClient?.connected) {
        logUtil.customLog(`[wxid-${wxId}] 已存在有效MQTT连接`);
        return;
    }
    
    // 检查是否正在连接中
    if (registry.isMqttConnecting) {
        logUtil.customLog(`[wxid-${wxId}] MQTT正在连接中，跳过`);
        return;
    }
    
    // 清理无效连接
    if (registry.mqttClient && !registry.mqttClient.connected) {
        logUtil.customLog(`[wxid-${wxId}] 发现无效连接，清理后重新初始化`);
        forceClearMqtt(wxId, registry);
    }
    
    // ═══════════════════════════════════════════════════════════════════
    // 2. 解析连接参数
    // ═══════════════════════════════════════════════════════════════════
    const {
        endPoint,           // MQTT服务器地址
        clientId,           // 客户端ID
        accessKey,          // 访问密钥
        secretKey,          // 密钥
        filterTopics,       // 订阅主题
        instanceId,         // 实例ID
        connectionOptions,  // 连接选项
        qos,                // QoS级别
    } = params;
    
    // 禁用自动重连（手动控制）
    connectionOptions.reconnectPeriod = 0;
    
    // ═══════════════════════════════════════════════════════════════════
    // 3. 建立连接
    // ═══════════════════════════════════════════════════════════════════
    let client = null;
    try {
        registry.isMqttConnecting = true;
        client = mqtt.connect(`tcp://${endPoint}:1883`, connectionOptions);
    } catch (error) {
        logUtil.customLog(`[wxid-${wxId}] MQTT连接失败: ${error.message}`, { level: 'error' });
        registry.isMqttConnecting = false;
        return;
    }
    
    if (!client) {
        return;
    }
    
    // 注册到Registry
    registryMqtt(wxId, client);
    
    // ═══════════════════════════════════════════════════════════════════
    // 4. 事件监听
    // ═══════════════════════════════════════════════════════════════════
    
    // 连接成功
    client.on("connect", function () {
        logUtil.customLog(`[wxid-${wxId}] MQTT连接成功`);
        
        // 订阅主题
        client.subscribe(filterTopics, { qos: 1 }, function (err, granted) {
            if (!err) {
                logUtil.customLog(`[wxid-${wxId}] MQTT订阅成功: ${JSON.stringify(granted)}`);
            } else {
                logUtil.customLog(`[wxid-${wxId}] MQTT订阅失败: ${err}`, { level: 'error' });
            }
        });
        
        registry.isMqttConnecting = false;
    });
    
    // 接收消息
    client.on("message", async function (topic, message) {
        // 处理消息...（见下文）
    });
    
    // 连接错误
    client.on("error", (error) => {
        clearMqttConnection(wxId);
        logUtil.customLog(`[wxid-${wxId}] MQTT错误: ${error.message}`, { level: 'error' });
    });
    
    // 连接关闭
    client.on("close", function (e) {
        logUtil.customLog(`[wxid-${wxId}] MQTT连接关闭`, { level: 'error' });
        notify.onMqttClose(wxId);
        clearMqttByWxId(wxId);
    });
    
    // 重连
    client.on("reconnect", () => {
        logUtil.customLog(`[wxid-${wxId}] MQTT重连中...`);
    });
    
    // 离线
    client.on("offline", () => {
        logUtil.customLog(`[wxid-${wxId}] MQTT离线`);
    });
};
```

### 2.2 消息接收与去重

```javascript
// MQTT消息接收处理

// 任务去重锁
let mqttTaskMapLock = new Map();
let mqttCleanupTimer = null;

client.on("message", async function (topic, message) {
    // 处理大数字（避免JSON精度丢失）
    let messageUtf8 = replaceLargeNumbers(message.toString("utf8"));
    const dataInfo = JSON.parse(messageUtf8);
    
    logUtil.customLog(
        `[wxid-${wxId}] [接收mqtt任务] type=${dataInfo.type} taskId=${dataInfo.id}`
    );
    
    const receiveId = dataInfo?.id;
    
    // ═══════════════════════════════════════════════════════════════════
    // 1. 过期消息过滤
    // ═══════════════════════════════════════════════════════════════════
    const expireHours = apolloConfig.mqttTaskExpireHours || 3;  // 默认3小时
    const expireTime = expireHours * 60 * 60 * 1000;
    
    if (dataInfo.createTime * 1000 < Date.now() - expireTime) {
        logUtil.customLog(`[wxid-${wxId}] taskId=${receiveId} 消息过期（${expireHours}小时）`, {
            level: "warn"
        });
        return;
    }
    
    // ═══════════════════════════════════════════════════════════════════
    // 2. 去重锁检查
    // ═══════════════════════════════════════════════════════════════════
    if (receiveId && mqttTaskMapLock.has(receiveId)) {
        logUtil.customLog(`[wxid-${wxId}] taskId=${receiveId} 消息重复，跳过`, {
            level: "warn"
        });
        return;
    }
    
    // 添加去重锁
    if (receiveId) {
        mqttTaskMapLock.set(receiveId, {
            wxId: wxId,
            timestamp: Date.now(),
            processed: true
        });
        
        // 启动清理定时器
        startCleanupTimerIfNeeded();
    }
    
    // ═══════════════════════════════════════════════════════════════════
    // 3. 任务分发
    // ═══════════════════════════════════════════════════════════════════
    try {
        const serverTaskBO = { ...ServerTaskBO, ...dataInfo };
        
        // type=100 私聊消息需要排队执行
        if (dataInfo.type === 100) {
            executeWithDelay(wxId, () => {
                execute(wxId, serverTaskBO);
            });
        } else {
            // 随机延迟1秒内（避免同时发送）
            await sleep(Math.ceil(Math.random() * 1000));
            execute(wxId, serverTaskBO);
        }
        
    } catch (error) {
        logUtil.customLog(`[wxid-${wxId}] 任务处理错误: ${error.message}`, { level: 'error' });
    }
});

// 去重锁清理
function cleanupExpiredLocks() {
    const now = Date.now();
    const expireTime = 6000 * 1000;  // 100分钟过期
    let cleanedCount = 0;
    
    for (const [taskId, lockInfo] of mqttTaskMapLock.entries()) {
        if (typeof lockInfo === 'object' && lockInfo.timestamp) {
            if (now - lockInfo.timestamp > expireTime) {
                mqttTaskMapLock.delete(taskId);
                cleanedCount++;
            }
        }
    }
    
    if (cleanedCount > 0) {
        logUtil.customLog(`[mqttTaskMapLock] 清理过期锁 ${cleanedCount} 个`);
    }
}
```

```javascript
// src/msg-center/core/mq/mqttConfig.js

const MqttConfig = {
    // 连接参数
    getConnectionParams(wxId, mqttInfo) {
        const { endPoint, clientId, accessKey, secretKey, instanceId } = mqttInfo;
        
        return {
            endPoint: endPoint,                    // 服务器地址

            secretKey: secretKey,

            // QoS级别（1=至少一次）
            qos: 1,
            
            // 连接选项
            connectionOptions: {
                username: accessKey,
                password: this.generatePassword(accessKey, secretKey),
                keepalive: 60,                     // 心跳间隔（秒）
                clean: true,                       // 清除会话
                reconnectPeriod: 0,                // 禁用自动重连
                connectTimeout: 30000,             // 连接超时
            }
        };
    },
    
    // 生成密码（签名）
    generatePassword(accessKey, secretKey) {
        const CryptoJS = require('crypto-js');
        const timestamp = Date.now();
        const message = `${accessKey}${timestamp}`;
        return CryptoJS.HmacSHA256(message, secretKey).toString();
    }
};

module.exports = MqttConfig;
```

### 3.1 服务端实现

```javascript
// src/msg-center/core/websocket/index.js

const WebSocket = require('ws');
const frontConnection = require('../data-config/frontConnection');
const logUtil = require('../../../init/log');

// 端口范围
const portRange = [13323, 13423];
const Host = '0.0.0.0';

const WebSocketServer = {
    port: null,
    client: null,
    server: null,
    
    /**
     * 获取可用端口列表
     */
    getPortList() {
        return Array.from(
            { length: portRange[1] - portRange[0] + 1 }, 
            (_, i) => i + portRange[0]
        );
    },
    
    /**
     * 启动服务器
     */
    async startServer() {
        let port = null;
        let portIndex = 0;
        const portList = this.getPortList();
        
        // 尝试端口直到找到可用的
        while (!port && portIndex < portList.length) {
            const candidatePort = portList[portIndex];
            
            try {
                this.server = new WebSocket.Server({ 
                    host: Host, 
                    port: candidatePort 
                });
                
                this.server.on('connection', this.handleConnection.bind(this));
                
                // 等待服务器启动
                await new Promise(resolve => this.server.once('listening', resolve));
                
                port = candidatePort;
                logUtil.customLog(`WebSocket服务器启动成功: ${Host}:${port}`);
                
            } catch (error) {
                logUtil.customLog(`端口${candidatePort}被占用，尝试下一个`);
                portIndex++;
            }
        }
        
        this.port = port;
        global.port = port;  // 全局保存供前端使用
    },
    
    /**
     * 处理新连接
     */
    handleConnection(connection) {
        logUtil.customLog('WebSocket客户端连接');
        
        this.client = connection;
        frontConnection.channel = connection;
        
        // 发送欢迎消息
        connection.send('欢迎连接到websocket服务器');
        
        // 连接关闭
        connection.on('close', () => {
            this.client = null;
            frontConnection.channel = null;
            logUtil.customLog('WebSocket客户端断开');
        });
        
        // 连接错误
        connection.on('error', (error) => {
            logUtil.customLog(`WebSocket错误: ${error.message}`, { level: 'error' });
        });
    },
    
    /**
     * 停止服务器
     */
    async stopServer() {
        if (this.server) {
            await new Promise(resolve => this.server.close(resolve));
            logUtil.customLog('WebSocket服务器已停止');
        }
    }
};

module.exports = WebSocketServer;
```

```javascript
// src/msg-center/core/front/frontServer.js

const WebSocketServer = require('../websocket/index');
const frontConnection = require('../data-config/frontConnection');
const wsPushTest = require('./wsPushTest');
const logUtil = require('../../../init/log');
const frontFlowInBound = require('../../dispatch-center/dispatch/frontFlowInBound');

const FrontServer = {
    start() {
        // 启动WebSocket服务器
        WebSocketServer.startServer().then(() => {
            logUtil.customLog(`WebSocket服务启动完成，端口: ${WebSocketServer.port}`);
        });
        
        // 监听连接
        WebSocketServer.server?.on('connection', (connection) => {
            // 监听消息
            connection.on('message', message => {
                const msgObj = message ? JSON.parse(message) : null;
                
                logUtil.customLog(`[收到前端WS消息] cmdId=${msgObj?.cmdId}`);
                
                if (!msgObj) return;
                
                // ═══════════════════════════════════════════════════════
                // 1. 系统消息处理
                // ═══════════════════════════════════════════════════════
                if (msgObj.cmdId === 'system') {
                    if (msgObj.body === 'online') {
                        // 前端上线
                        frontConnection.channel = connection;
                        frontConnection.isLogin = true;
                        logUtil.customLog('前端上线');
                    } else if (msgObj.body === 'offline') {
                        // 前端下线
                        frontConnection.channel = null;
                        frontConnection.isLogin = false;
                        logUtil.customLog('前端下线');
                    }
                }
                // ═══════════════════════════════════════════════════════
                // 2. 测试消息
                // ═══════════════════════════════════════════════════════
                else if (msgObj.cmdId === 'webtest') {
                    wsPushTest.onMessage(msgObj);
                }
                // ═══════════════════════════════════════════════════════
                // 3. 业务消息分发
                // ═══════════════════════════════════════════════════════
                else {
                    frontFlowInBound(message);
                }
            });
        });
    },
    
    stop() {
        WebSocketServer.stopServer();
    }
};

module.exports = FrontServer;
```

## 四、IPC通信（逆向服务）

### 4.1 通信协议

```
┌─────────────────────────────────────────────────────────────────────────┐
│                       IPC通信协议                                        │
└─────────────────────────────────────────────────────────────────────────┘

通信方式：Windows Named Pipes（命名管道）

    Electron 主进程                              BasicService.exe
    ┌─────────────────┐                        ┌─────────────────┐
    │                 │                        │                 │
    │   clibrary.js   │◄────── Named Pipe ────►│  PipeCore.dll   │
    │   (ffi-napi)    │                        │                 │
    │                 │                        │                 │
    └─────────────────┘                        └─────────────────┘
    
API调用流程：

    1. IpcConnectServer(processId)
       └─► 建立连接，返回 pipeCode（管道句柄）
       
    2. IpcClientSendMessage(buffer, length, pipeCode)
       └─► 发送消息到逆向服务
       
    3. IpcSelectCltChannel(pipeCode)
       └─► 检查是否有待接收消息
       └─► 返回值: >0 有消息, 0 无消息, <0 错误
       
    4. IpcClientRecvMessage(buffer, length, pipeCode)
       └─► 接收消息
       
    5. IpcClientClose(pipeCode)
       └─► 关闭连接
```

### 4.3 连接扫描

```javascript
// src/msg-center/core/reverse/initIpcTask.js

const AsyncLock = require('async-lock');
const sleep = require('../../core/utils/sleep');
const RegistryConfig = require('../registry-config/index');
const Clibrary = require('./dll/clibrary');
const IpcConfig = require('./ipcConfig');
const IpcUtil = require('./ipcUtil');
const AsyncSelectTask = require('./asyncSelectTask');
const frontSend = require('../../dispatch-center/frontSend');
const logUtil = require('../../../init/log');
const pipeCodeCache = require('../cache/common/pipeCodeCache');

let lock = new AsyncLock();
const frontSendService = frontSend.getInstance();

const InitIpcTask = (function () {
    let instance;

    function createInstance() {
        return {
            /**
             * 批量检查进程是否退出
             */
            batchCheckAndExit(processIdList) {
                const exitList = [];
                
                for (let processId of processIdList) {
                    const existFlag = IpcUtil.isProcessExist(processId);
                    const registry = RegistryConfig.getRegistryByKey(processId, 'id');
                    
                    if (!existFlag) {
                        exitList.push(processId);
                        RegistryConfig.removePipe(processId);
                        frontSendService.sendGetAllConfig();
                        logUtil.customLog(`[initIpcTask] [wxid-${registry?.wxid}] 删除IPC连接: ${processId}`);
                    }
                }
                
                return exitList;
            },
            
            /**
             * 启动消息接收
             */
            startSelectMessage(registry) {
                AsyncSelectTask.run(registry.pipeLineWrapper);
            },
            
            /**
             * 主循环：每5秒扫描一次进程
             */
            async run() {
                logUtil.customLog('[initIpcTask] 启动IPC连接扫描');
                
                let processIdsAvaliableOld = [];
                let oldAllProcessIds = [];
                
                while (true) {
                    try {
                        // ═══════════════════════════════════════════════════
                        // 1. 获取当前进程列表
                        // ═══════════════════════════════════════════════════
                        let processIds = IpcUtil.getProcessIds();
                        
                        // 过滤不可用的进程
                        const processIdsTemp = [];
                        processIds.forEach(processId => {
                            const isAvaliable = Clibrary.isUseProcess(processId);
                            
                            if (!isAvaliable) {
                                // 管道不可用，清理缓存
                                const pipeCode = pipeCodeCache.pipeCodeMap.get(processId);
                                if (pipeCode != null) {
                                    try {
                                        RegistryConfig.removePipe(processId, pipeCode);
                                        frontSendService.sendGetAllConfig();
                                    } catch (error) {
                                        pipeCodeCache.deleteProcessIdFromCache(processId);
                                    }
                                }
                            } else {
                                processIdsTemp.push(processId);
                            }
                        });
                        
                        processIds = processIdsTemp;
                        processIdsAvaliableOld = JSON.parse(JSON.stringify(processIds));
                        
                        // ═══════════════════════════════════════════════════
                        // 2. 比较当前连接
                        // ═══════════════════════════════════════════════════
                        let currProcessIds = RegistryConfig.getCurrProcessIds();
                        
                        if (processIds.length === 0) {
                            // 没有扫描到进程
                            const exitList = this.batchCheckAndExit(currProcessIds);
                            logUtil.customLog(`[initIpcTask] 没有扫描到进程，退出: ${JSON.stringify(exitList)}`);
                            await sleep(5000);
                            continue;
                        }
                        
                        const isArrEqual = IpcUtil.arrayEquals(processIds, currProcessIds);
                        if (isArrEqual) {
                            await sleep(5000);
                            continue;
                        }
                        
                        // ═══════════════════════════════════════════════════
                        // 3. 建立新连接
                        // ═══════════════════════════════════════════════════
                        for (let i = 0; i < processIds.length; i++) {
                            const processId = processIds[i];
                            
                            if (currProcessIds.includes(processId)) {
                                currProcessIds = currProcessIds.filter(pId => pId !== processId);
                                continue;
                            }
                            
                            // 使用锁防止并发
                            lock.acquire(`ipcLock-${processId}`, (done) => {
                                if (!RegistryConfig.getCurrProcessIds().includes(processId)) {
                                    
                                    // 建立连接
                                    let pipeCode = Clibrary.IpcConnectServer(processId);
                                    
                                    if (pipeCode <= 0) {
                                        // 连接失败，尝试清理旧连接
                                        const oldPipeCode = pipeCodeCache.pipeCodeMap.get(processId);
                                        if (oldPipeCode != null) {
                                            try {
                                                RegistryConfig.removePipe(processId, oldPipeCode);
                                            } catch (error) {
                                                pipeCodeCache.deleteProcessIdFromCache(processId);
                                            }
                                        }
                                    }
                                    
                                    // 缓存pipeCode
                                    pipeCodeCache.updatePipeCodeCache(processId, pipeCode);
                                    
                                    // 判断是否企业微信
                                    const workWx = IpcUtil.processMap[processId] || false;
                                    
                                    // 创建管道包装
                                    const pipeLineWrapper = {
                                        pipeCode,
                                        id: processId,
                                        processId,
                                        available: IpcConfig.NOT_AVAILABLE,
                                        workWx,
                                        createTime: Date.now(),
                                        lastReportId: null,
                                        lastTimer: null,
                                    };
                                    
                                    // 创建注册信息
                                    const registry = {
                                        pipeLineWrapper,
                                        sendToCloudFlag: IpcConfig.NOT_SEND_LOGIN_TO_CLOUD,
                                        id: processId,
                                        scanTime: Date.now(),
                                        workWx,
                                    };
                                    
                                    RegistryConfig.add(registry);
                                    logUtil.customLog(`[initIpcTask] 建立IPC连接成功 processId:${processId} pipeCode:${pipeCode}`);
                                    
                                    frontSendService.sendGetAllConfig();
                                    this.startSelectMessage(registry);
                                }
                                done();
                            });
                        }
                        
                        // ═══════════════════════════════════════════════════
                        // 4. 清理断开的连接
                        // ═══════════════════════════════════════════════════
                        if (currProcessIds.length > 0) {
                            let exitProcessIdList = this.batchCheckAndExit(currProcessIds);
                            logUtil.customLog(`[initIpcTask] 清理断开连接: ${JSON.stringify(exitProcessIdList)}`);
                        }
                        
                        await sleep(5000);
                        
                    } catch (error) {
                        logUtil.customLog(`[initIpcTask] 错误: ${error.message}`, { level: 'error' });
                        await sleep(5000);
                    }
                }
            }
        };
    }

    return {
        getInstance() {
            if (!instance) {
                instance = createInstance();
            }
            return instance;
        },
    };
})();

module.exports = InitIpcTask;
```

### 4.4 消息接收循环

```javascript
// src/msg-center/core/reverse/asyncSelectTask.js

const Clibrary = require('./dll/clibrary');
const sleep = require('../../core/utils/sleep');
const logoutService = require('../../business/convert-service/logoutService');
const logUtil = require('../../../init/log');
const dispatchOutBound = require('../../dispatch-center/dispatchOutBound');

/**
 * 处理大数字（避免JSON精度丢失）
 */
function replaceLargeNumbers(json) {
    const regex = /(?<=:)\s*\d{17,}(?=[,\}])/g;
    return json.replace(regex, match => `"${match}"`);
}

const AsyncSelectTask = {
    /**
     * IPC消息接收循环
     */
    async loop(wrapper) {
        const createTime = Date.now();
        const pipeCode = wrapper.pipeCode;
        
        // 检查是否有消息
        const selectCode = Clibrary.IpcSelectCltChannel(pipeCode);
        
        // ═══════════════════════════════════════════════════════════════
        // 1. 无消息：等待后继续
        // ═══════════════════════════════════════════════════════════════
        if (selectCode === 0) {
            await sleep(200);  // 200ms轮询间隔
            this.loop(wrapper);
            return;
        }
        
        // ═══════════════════════════════════════════════════════════════
        // 2. 连接断开：触发登出
        // ═══════════════════════════════════════════════════════════════
        if (selectCode < 0) {
            this.closeIpcConnect(pipeCode, selectCode, wrapper);
            return;
        }
        
        // ═══════════════════════════════════════════════════════════════
        // 3. 有消息：接收处理
        // ═══════════════════════════════════════════════════════════════
        this.successIpcConnect(pipeCode, selectCode, createTime, wrapper);
        this.loop(wrapper);
    },
    
    /**
     * 启动消息接收
     */
    async run(wrapper) {
        logUtil.customLog(`[IPC-START] [wxid-${wrapper.wxid}] 开始接收消息`);
        this.loop(wrapper);
    },
    
    /**
     * 处理接收到的消息
     */
    successIpcConnect(pipeCode, selectCode, createTime, wrapper) {
        try {
            // 接收消息
            let message = Clibrary.IpcClientRecvMessage(pipeCode, selectCode, wrapper.wxid);
            
            const overTime = Date.now();
            wrapper.lastReadTime = Date.now();
            
            logUtil.customLog(
                `[wxid-${wrapper.wxid}] 接收IPC消息 ` +
                `pipeCode:${pipeCode} selectCode:${selectCode} ` +
                `耗时:${overTime - createTime}ms`
            );
            
            // 处理大数字
            message = replaceLargeNumbers(message);
            
            // 分发消息
            dispatchOutBound(message, wrapper);
            
        } catch (error) {
            logUtil.customLog(`[wxid-${wrapper.wxid}] 消息处理错误: ${error.message}`, { level: 'error' });
        }
    },
    
    /**
     * 处理连接关闭
     */
    closeIpcConnect(pipeCode, selectCode, wrapper) {
        logUtil.customLog(`[IPC-CLOSE] [wxid-${wrapper.wxid}] selectCode:${selectCode}`);
        logoutService.operate(null, wrapper);
    }
};

module.exports = AsyncSelectTask;
```

```
│                    Electron IPC通信方式                                  │

【1. 同步通信】

        ├──────────────────────────────────────────────►│
        │                                               │  执行操作
        │                                               │  不返回结果
        
    特点：发送后不等待返回
    适用：触发操作（启动服务、打开文件）

    渲染进程                                          主进程
        │                                               │

        ├──────────────────────────────────────────────►│
        │                                               │  异步处理
        │◄──────────────────────────────────────────────┤
        │  返回 Promise 结果                            │

    特点：返回Promise，可await
    适用：耗时操作（文件处理、网络请求）


【4. 主进程推送】
    主进程                                          渲染进程
        │                                               │
        │  webContents.send('get-all-config', data)     │
        ├──────────────────────────────────────────────►│
        │                                               │  接收并处理

    适用：状态更新、通知
```

```javascript
const axios = require('axios');

/**
 * HTTP请求封装
 * @param {Object} options - 请求选项
 * @returns {Promise<Object>}
 */
async function httpFetch(options) {
    const { 
        url, 
        data, 
        method = 'POST', 
        headers = {},
        timeout = 30000 
    } = options;
    
    try {
        const response = await axios({
            url,
            method,
            data,
            headers: {
                'Content-Type': 'application/json',
                ...headers
            },
            timeout
        });
        
        return {
            code: 0,
            data: response.data
        };
        
    } catch (error) {
        log.error('HTTP请求失败:', error.message);
        
        return {
            code: -1,
            error: error.message,
            response: error.response?.data
        };
    }
}

/**
 * GET请求
 */
async function httpGet(url, params = {}) {
    return httpFetch({
        url,
        method: 'GET',
        data: params
    });
}

/**
 * 带重试的请求
 */
async function httpFetchWithRetry(options, maxRetries = 3) {
    let lastError;
    
    for (let i = 0; i < maxRetries; i++) {
        try {
            const result = await httpFetch(options);
            if (result.code === 0) {
                return result;
            }
            lastError = result;
        } catch (error) {
            lastError = error;
        }
        
        // 等待后重试
        await new Promise(resolve => setTimeout(resolve, 1000 * (i + 1)));
    }
    
    return lastError;
}

module.exports = { httpFetch, httpGet, httpFetchWithRetry };
```

```javascript
// src/msg-center/core/utils/fileDownload.js

const axios = require('axios');
const fs = require('fs');
const path = require('path');
const os = require('os');
const logUtil = require('../../../init/log');

/**
 * 下载文件并返回文件信息
 * @param {string} url - 文件URL
 * @param {string} filename - 文件名
 * @returns {Promise<{filePath: string, filename: string}>}
 */
async function downloadAsyncReturnFileInfo(url, filename) {
    const tempDir = os.tmpdir();
    const filePath = path.join(tempDir, 'galaxy-client', filename);
    
    // 确保目录存在
    const dir = path.dirname(filePath);
    if (!fs.existsSync(dir)) {
        fs.mkdirSync(dir, { recursive: true });
    }
    
    logUtil.customLog(`[下载文件] 开始 url=${url}`);
    
    try {
        const response = await axios({
            url,
            method: 'GET',
            responseType: 'stream',
            timeout: 60000,
        });
        
        const writer = fs.createWriteStream(filePath);
        response.data.pipe(writer);
        
        await new Promise((resolve, reject) => {
            writer.on('finish', resolve);
            writer.on('error', reject);
        });
        
        logUtil.customLog(`[下载文件] 成功 path=${filePath}`);
        
        return { 
            filePath, 
            filename,
            size: fs.statSync(filePath).size
        };
        
    } catch (error) {
        logUtil.customLog(`[下载文件] 失败: ${error.message}`, { level: 'error' });
        throw error;
    }
}

/**
 * 检查文件是否存在
 */
function checkFileExist(filePath) {
    return fs.existsSync(filePath);
}

    downloadAsyncReturnFileInfo,
    checkFileExist
```

```javascript
 * 上传文件到OSS
 * @param {string} localPath - 本地文件路径
```
