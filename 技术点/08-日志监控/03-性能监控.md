# 性能监控

> 进程与系统性能监控

---

## 一、技术简介

### 1.1 为什么需要性能监控

Electron 应用运行在用户电脑上，需要监控：

- **内存使用**：防止内存泄漏
- **CPU 占用**：避免卡顿
- **进程状态**：确保服务正常运行

### 1.2 使用的库

| 库 | 用途 |
|---|---|
| `pidusage` | 监控特定进程的 CPU/内存 |
| `systeminformation` | 获取系统信息 |
| Node.js 内置 | `process.memoryUsage()` 等 |

### 2.2 监控实现

```javascript
// src/msg-center/timer/monitorTimer.js

const pidusage = require('pidusage');
const si = require('systeminformation');
const logUtil = require('../../init/log');
const slsLog = require('../../init/slsLog');

class MonitorTimer {
    constructor() {
        this.interval = null;
        this.checkInterval = 60000;  // 1分钟检查一次
    }

    /**
     * 启动监控
     */
    start() {
        this.interval = setInterval(() => {
            this.collectMetrics();
        }, this.checkInterval);
    }

    /**
     * 收集性能指标
     */
    async collectMetrics() {
        try {
            // 1. 当前进程信息
            const processMetrics = await this.getProcessMetrics();

            // 2. 系统信息
            const systemMetrics = await this.getSystemMetrics();

            // 3. 合并上报
            const metrics = {
                ...processMetrics,
                ...systemMetrics,
                timestamp: Date.now()
            };

            // 记录到本地日志
            logUtil.customLog('性能指标', { level: 'debug', ...metrics });

            // 检查是否超过阈值
            this.checkThresholds(metrics);

        } catch (error) {
            logUtil.error('监控采集失败', error);
        }
    }

    /**
     * 获取进程指标
     */
    async getProcessMetrics() {
        // Node.js 内置内存信息
        const memoryUsage = process.memoryUsage();

        // pidusage 获取 CPU 使用率
        const stats = await pidusage(process.pid);

        return {
            // 内存（MB）
            heapUsed: Math.round(memoryUsage.heapUsed / 1024 / 1024),
            heapTotal: Math.round(memoryUsage.heapTotal / 1024 / 1024),
            rss: Math.round(memoryUsage.rss / 1024 / 1024),
            external: Math.round(memoryUsage.external / 1024 / 1024),

            // CPU（百分比）
            cpu: Math.round(stats.cpu * 100) / 100,

            // 进程运行时间
            uptime: process.uptime()
        };
    }

    /**
     * 获取系统指标
     */
    async getSystemMetrics() {
        // 系统内存
        const mem = await si.mem();

        // CPU 负载
        const load = await si.currentLoad();

        return {
            // 系统内存（MB）
            systemMemTotal: Math.round(mem.total / 1024 / 1024),
            systemMemUsed: Math.round(mem.used / 1024 / 1024),
            systemMemFree: Math.round(mem.free / 1024 / 1024),

            // 系统 CPU
            systemCpu: Math.round(load.currentLoad * 100) / 100
        };
    }

    /**
     * 检查阈值并告警
     */
    checkThresholds(metrics) {
        // 内存超过 500MB
        if (metrics.heapUsed > 500) {
            logUtil.warn('内存使用过高', { heapUsed: metrics.heapUsed });
            slsLog.reportEvent('memory_warning', { heapUsed: metrics.heapUsed });
        }

        // CPU 超过 80%
        if (metrics.cpu > 80) {
            logUtil.warn('CPU 使用过高', { cpu: metrics.cpu });
            slsLog.reportEvent('cpu_warning', { cpu: metrics.cpu });
        }
    }

    /**
     * 停止监控
     */
    stop() {
        if (this.interval) {
            clearInterval(this.interval);
            this.interval = null;
        }
    }
}

module.exports = new MonitorTimer();
```

## 三、pidusage 详解

### 3.1 基本使用

```javascript
const pidusage = require('pidusage');

// 获取当前进程
const stats = await pidusage(process.pid);
console.log(stats);
// {
//   cpu: 10.5,      // CPU 使用率（百分比）
//   memory: 12345,  // 内存使用（字节）
//   ppid: 123,      // 父进程 ID
//   pid: 456,       // 进程 ID
//   ctime: 1000,    // CPU 时间
//   elapsed: 5000,  // 运行时间
//   timestamp: ...  // 时间戳
// }

// 获取其他进程
const stats = await pidusage(1234);  // 传入 PID

// 获取多个进程
const stats = await pidusage([pid1, pid2, pid3]);
```

### 3.2 监控子进程

```javascript
const { spawn } = require('child_process');
const pidusage = require('pidusage');

// 启动子进程
const child = spawn('node', ['worker.js']);

// 监控子进程
setInterval(async () => {
    try {
        const stats = await pidusage(child.pid);
        console.log(`子进程 CPU: ${stats.cpu}%, 内存: ${stats.memory}`);
    } catch (e) {
        console.log('子进程已退出');
    }
}, 5000);
```

### 3.3 清理历史数据

```javascript
// pidusage 会保留历史数据用于计算
// 定期清理避免内存泄漏
pidusage.clear();
```

## 四、systeminformation 详解

### 4.1 常用 API

```javascript
const si = require('systeminformation');

// 系统信息
const system = await si.system();
// { manufacturer, model, version, ... }

// CPU 信息
const cpu = await si.cpu();
// { brand, speed, cores, ... }

// 当前 CPU 负载
const load = await si.currentLoad();
// { currentLoad, currentLoadUser, currentLoadSystem, ... }

// 内存信息
const mem = await si.mem();
// { total, free, used, active, ... }

// 磁盘信息
const disk = await si.fsSize();
// [{ fs, type, size, used, available, ... }]

// 网络信息
const network = await si.networkStats();
// [{ iface, rx_bytes, tx_bytes, ... }]
```

### 4.2 批量获取

```javascript
// 一次性获取多种信息
const data = await si.get({
    cpu: 'brand,speed,cores',
    mem: 'total,used,free',
    currentLoad: 'currentLoad'
});

console.log(data.cpu);
console.log(data.mem);
console.log(data.currentLoad);
```

## 五、Node.js 内置监控

### 5.1 内存监控

```javascript
// process.memoryUsage()
const usage = process.memoryUsage();

console.log({
    rss: usage.rss,           // 常驻内存（进程总内存）
    heapTotal: usage.heapTotal, // V8 堆总大小
    heapUsed: usage.heapUsed,   // V8 堆已使用
    external: usage.external,   // C++ 对象绑定的内存
    arrayBuffers: usage.arrayBuffers  // ArrayBuffer 内存
});
```

### 5.2 CPU 使用

```javascript
// process.cpuUsage()
const startUsage = process.cpuUsage();

// ... 执行一些操作 ...

const endUsage = process.cpuUsage(startUsage);
console.log({
    user: endUsage.user / 1000,   // 用户态 CPU 时间（毫秒）
    system: endUsage.system / 1000 // 系统态 CPU 时间（毫秒）
});
```

### 5.3 其他信息

```javascript
// 进程运行时间
console.log(process.uptime());  // 秒

// 进程 ID
console.log(process.pid);

// 平台信息
console.log(process.platform);  // 'win32', 'darwin', 'linux'
console.log(process.arch);      // 'x64', 'arm64'
```

## 六、监控数据上报

### 6.1 定期上报

```javascript
// 每分钟采集一次，每5分钟上报一次
class MetricsReporter {
    constructor() {
        this.buffer = [];
    }

    collect() {
        this.buffer.push({
            timestamp: Date.now(),
            ...process.memoryUsage()
        });

        // 缓冲区满了上报
        if (this.buffer.length >= 5) {
            this.report();
        }
    }

    report() {
        slsLog.reportPerformance({
            metrics: this.buffer,
            gid: store.getGid()
        });
        this.buffer = [];
    }
}
```

### 6.2 异常时立即上报

```javascript
// 检测到问题立即上报
function checkAndReport(metrics) {
    if (metrics.heapUsed > 500 * 1024 * 1024) {  // 500MB
        slsLog.reportEvent('memory_high', {
            heapUsed: metrics.heapUsed,
            immediate: true
        });
    }
}
```
