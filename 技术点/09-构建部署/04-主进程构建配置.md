# 主进程构建配置

> Electron 主进程和 Preload 脚本的 TypeScript 编译与打包配置

---

## 一、概述

### 1.1 主进程代码特点

```
┌─────────────────────────────────────────────────────────────────┐
│                    主进程构建特点                                │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  运行环境：Node.js（不是浏览器）                                 │
│  ├── 可使用所有 Node.js API                                    │
│  ├── 可使用 require/import                                     │
│  ├── 可使用 __dirname / __filename                             │
│  └── 可使用 Electron 特有 API                                  │
│                                                                 │
│  构建目标：                                                      │
│  ├── 输出 CommonJS 或 ESM                                      │
│  ├── 不需要 polyfill                                           │
│  ├── 外部依赖不打包（external）                                 │
│  └── 保留 Node.js 语义                                         │
│                                                                 │
│  入口文件：                                                      │
│  ├── main.js / main.ts        主进程入口                       │
│  ├── preload.js / preload.ts  预加载脚本                       │
│  └── worker.js                可选的 Worker 脚本               │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

### 1.2 与 Renderer 构建对比

| 方面 | Main/Preload | Renderer |
|------|--------------|----------|
| 运行环境 | Node.js | Chromium |
| 构建目标 | node/electron | chrome/browser |
| 模块格式 | CJS/ESM | ESM |
| 外部依赖 | external | bundle |
| __dirname | 保留 | 不存在 |
| Polyfill | 不需要 | 可能需要 |

## 二、TypeScript 编译

### 2.1 编译工具选择

| 工具 | 速度 | TypeScript 检查 | 特点 |
|------|------|----------------|------|
| `tsc` | 慢 | ✓ 完整检查 | 官方工具，最可靠 |
| `esbuild` | 极快 | ✗ 不检查 | 最快，需配合 tsc 检查 |
| `swc` | 很快 | ✗ 不检查 | Rust 实现，接近 esbuild |
| `tsup` | 快 | 可选 | esbuild 封装，易用 |

**推荐组合**：
- 类型检查：`tsc --noEmit`
- 编译输出：`esbuild` 或 `tsup`

### 2.2 tsconfig 配置

```json
// tsconfig.main.json - 主进程专用配置
{
  "compilerOptions": {
    // 输出目标
    "target": "ES2022",
    "module": "CommonJS",
    "moduleResolution": "Node",
    
    // 输出配置
    "outDir": "./dist/main",
    "rootDir": "./src/main",
    
    // 严格模式
    "strict": true,
    "esModuleInterop": true,
    "skipLibCheck": true,
    "forceConsistentCasingInFileNames": true,
    
    // 重要：保留这些
    "resolveJsonModule": true,
    "declaration": false,
    
    // Source Map
    "sourceMap": true,
    
    // 类型定义
    "types": ["node"]
  },
  "include": ["src/main/**/*"],
  "exclude": ["node_modules"]
}
```

```json
// tsconfig.preload.json - Preload 专用配置
{
  "extends": "./tsconfig.main.json",
  "compilerOptions": {
    "outDir": "./dist/preload",
    "rootDir": "./src/preload"
  },
  "include": ["src/preload/**/*"]
}
```

### 2.3 关键配置说明

```
┌─────────────────────────────────────────────────────────────────┐
│                    tsconfig 关键配置                             │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  target: "ES2022"                                               │
│  ├── Electron 内置的 Node.js 支持现代语法                       │
│  └── 无需转译 async/await、可选链等                             │
│                                                                 │
│  module: "CommonJS"                                             │
│  ├── Electron 主进程默认使用 CommonJS                           │
│  └── 也可以用 "ESNext"，但需要 package.json 配置               │
│                                                                 │
│  moduleResolution: "Node"                                       │
│  ├── 使用 Node.js 模块解析策略                                  │
│  └── 不要用 "Bundler"（那是给打包工具用的）                     │
│                                                                 │
│  esModuleInterop: true                                          │
│  ├── 允许 import x from 'x' 导入 CommonJS 模块                  │
│  └── Electron 相关包通常需要这个                                │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

## 三、使用 esbuild

### 3.1 基础配置

```javascript
// scripts/build-main.js
const esbuild = require('esbuild')
const path = require('path')

const isDev = process.env.NODE_ENV !== 'production'

async function build() {
  await esbuild.build({
    entryPoints: [
      'src/main/index.ts',
      'src/preload/index.ts'
    ],
    bundle: true,
    platform: 'node',
    target: 'node18',
    outdir: 'dist',
    
    // 外部依赖不打包
    external: [
      'electron',
      // 所有 node_modules 都外部化
      ...Object.keys(require('../package.json').dependencies || {})
    ],
    
    // Source Map
    sourcemap: isDev ? 'inline' : false,
    
    // 压缩
    minify: !isDev,
    
    // 保持文件名
    entryNames: '[dir]/[name]',
    
    // 元数据
    metafile: true
  })
  
  console.log('Main process build complete')
}

build().catch(() => process.exit(1))
```

### 3.2 处理 __dirname

esbuild 默认会替换 `__dirname`，需要特殊处理：

```javascript
// esbuild 配置
{
  define: {
    // 不替换 __dirname
  },
  banner: {
    js: `
      const __filename = require('url').fileURLToPath(import.meta.url);
      const __dirname = require('path').dirname(__filename);
    `
  },
  // 或者使用 inject
  inject: ['./scripts/dirname-shim.js']
}
```

```javascript
// scripts/dirname-shim.js
// 对于 ESM 输出
import { fileURLToPath } from 'url'
import { dirname } from 'path'
export const __filename = fileURLToPath(import.meta.url)
export const __dirname = dirname(__filename)
```

更简单的方式 - 保持 CommonJS：

```javascript
// esbuild 配置
{
  format: 'cjs',  // CommonJS 格式
  // __dirname 自然可用
}
```

### 3.3 开发模式（Watch）

```javascript
const esbuild = require('esbuild')

async function watch() {
  const ctx = await esbuild.context({
    entryPoints: ['src/main/index.ts'],
    bundle: true,
    platform: 'node',
    target: 'node18',
    outdir: 'dist/main',
    external: ['electron'],
    sourcemap: 'inline',
    format: 'cjs'
  })
  
  await ctx.watch()
  console.log('Watching for changes...')
  
  // 文件变化时通知 Electron 重启
  // 可以配合 electron-reload 或自定义逻辑
}

watch()
```

## 四、使用 tsup

tsup 是 esbuild 的封装，配置更简洁：

### 4.1 安装

```bash
npm install --save-dev tsup
```

### 4.2 配置文件

```typescript
// tsup.config.ts
import { defineConfig } from 'tsup'

export default defineConfig({
  entry: {
    main: 'src/main/index.ts',
    preload: 'src/preload/index.ts'
  },
  
  // 输出格式
  format: ['cjs'],
  
  // 目标平台
  platform: 'node',
  target: 'node18',
  
  // 输出目录
  outDir: 'dist',
  
  // 清理输出目录
  clean: true,
  
  // Source Map
  sourcemap: process.env.NODE_ENV !== 'production',
  
  // 压缩
  minify: process.env.NODE_ENV === 'production',
  
  // 外部依赖
  external: ['electron'],
  
  // 不打包依赖
  noExternal: [],
  
  // 类型定义
  dts: false,
  
  // 分割代码
  splitting: false
})
```

### 4.3 package.json 脚本

```json
{
  "scripts": {
    "build:main": "tsup",
    "build:main:watch": "tsup --watch",
    "typecheck": "tsc --noEmit -p tsconfig.main.json"
  }
}
```

## 六、external 依赖处理

### 6.1 为什么需要 external

```
┌─────────────────────────────────────────────────────────────────┐
│                    为什么使用 external                          │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  主进程依赖不应该被打包：                                        │
│                                                                 │
│  1. electron 模块                                               │
│     ├── 由 Electron 运行时提供                                  │
│     └── 打包进去会导致找不到                                    │
│                                                                 │
│  2. Node.js 内置模块                                            │
│     ├── fs, path, crypto 等                                    │
│     └── 由 Node.js 运行时提供                                  │
│                                                                 │
│  3. Native 模块                                                 │
│     ├── .node 文件无法被打包                                    │
│     └── 需要从 node_modules 加载                               │
│                                                                 │
│  4. 大型依赖                                                    │
│     ├── 避免重复打包                                            │
│     └── 保持更新灵活性                                          │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

### 6.2 配置策略

```javascript
// 方式一：全部外部化
const pkg = require('./package.json')
const external = [
  'electron',
  ...Object.keys(pkg.dependencies || {}),
  ...Object.keys(pkg.optionalDependencies || {})
]

// 方式二：只外部化 native 模块和 electron
const external = [
  'electron',
  'better-sqlite3',
  'sharp',
  'node-pty',
  'ffi-napi'
]

// 方式三：使用插件自动检测
// esbuild-plugin-external-global
```

### 6.3 特殊模块处理

```javascript
// 某些模块需要特殊处理
{
  external: ['electron'],
  
  // 内联某些纯 JS 模块以避免路径问题
  noExternal: ['lodash-es', 'dayjs'],
  
  // 或使用 esbuild 插件
  plugins: [
    {
      name: 'native-modules',
      setup(build) {
        // 所有 .node 文件标记为外部
        build.onResolve({ filter: /\.node$/ }, args => ({
          external: true
        }))
      }
    }
  ]
}
```

## 八、完整构建脚本

### 8.1 综合构建脚本

```javascript
// scripts/build.js
const esbuild = require('esbuild')
const path = require('path')
const fs = require('fs-extra')

const pkg = require('../package.json')
const isDev = process.env.NODE_ENV !== 'production'

// 公共配置
const commonConfig = {
  bundle: true,
  platform: 'node',
  target: 'node18',
  format: 'cjs',
  external: [
    'electron',
    ...Object.keys(pkg.dependencies || {})
  ],
  sourcemap: isDev ? 'inline' : false,
  minify: !isDev,
  logLevel: 'info'
}

async function buildMain() {
  await esbuild.build({
    ...commonConfig,
    entryPoints: ['src/main/index.ts'],
    outfile: 'dist/main/index.js',
    define: {
      'process.env.NODE_ENV': JSON.stringify(process.env.NODE_ENV || 'development')
    }
  })
  console.log('✓ Main process built')
}

async function buildPreload() {
  await esbuild.build({
    ...commonConfig,
    entryPoints: ['src/preload/index.ts'],
    outfile: 'dist/preload/index.js',
    // Preload 脚本可能需要 Electron 的部分 API
    external: ['electron']
  })
  console.log('✓ Preload script built')
}

async function copyAssets() {
  // 复制静态资源
  if (fs.existsSync('src/main/assets')) {
    await fs.copy('src/main/assets', 'dist/main/assets')
    console.log('✓ Assets copied')
  }
}

async function main() {
  const startTime = Date.now()
  
  // 清理
  await fs.remove('dist/main')
  await fs.remove('dist/preload')
  
  // 并行构建
  await Promise.all([
    buildMain(),
    buildPreload(),
    copyAssets()
  ])
  
  console.log(`\n✓ Build completed in ${Date.now() - startTime}ms`)
}

main().catch(err => {
  console.error(err)
  process.exit(1)
})
```

### 8.2 开发模式脚本

```javascript
// scripts/dev.js
const esbuild = require('esbuild')
const { spawn } = require('child_process')
const path = require('path')

let electronProcess = null

async function startElectron() {
  if (electronProcess) {
    electronProcess.kill()
  }
  
  electronProcess = spawn('electron', ['.'], {
    stdio: 'inherit',
    env: {
      ...process.env,
      NODE_ENV: 'development'
    }
  })
}

async function watch() {
  const ctx = await esbuild.context({
    entryPoints: [
      'src/main/index.ts',
      'src/preload/index.ts'
    ],
    bundle: true,
    platform: 'node',
    target: 'node18',
    format: 'cjs',
    outdir: 'dist',
    external: ['electron'],
    sourcemap: 'inline',
    logLevel: 'info',
    
    plugins: [{
      name: 'restart-electron',
      setup(build) {
        build.onEnd(result => {
          if (result.errors.length === 0) {
            console.log('Build succeeded, restarting Electron...')
            startElectron()
          }
        })
      }
    }]
  })
  
  await ctx.watch()
  console.log('Watching for changes...')
}

watch().catch(console.error)
```
