# 体积与性能优化

> Electron 应用的打包体积优化与构建性能提升

## 一、Electron 应用体积分析

### 1.1 典型体积构成

```
┌─────────────────────────────────────────────────────────────────┐
│                    Electron 应用体积构成                         │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  总体积（约 150-250MB 安装包）                                   │
│  ├── Electron 运行时     ~120MB (50-60%)                       │
│  │   ├── Chromium 内核                                         │
│  │   ├── Node.js 运行时                                        │
│  │   └── V8 引擎                                               │
│  │                                                              │
│  ├── node_modules        ~50-100MB (20-40%)                    │
│  │   ├── 生产依赖                                               │
│  │   └── 可能被错误打包的开发依赖                                │
│  │                                                              │
│  ├── 应用代码            ~1-10MB (1-5%)                        │
│  │   ├── 主进程代码                                             │
│  │   ├── 渲染进程代码                                           │
│  │   └── 静态资源                                               │
│  │                                                              │
│  └── 额外资源            变化大                                 │
│      ├── 图片/字体                                              │
│      └── 其他二进制文件                                          │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

### 1.2 分析工具

```bash
# 查看 asar 内容
asar list dist/*/resources/app.asar

# 统计文件大小
du -sh dist/win-unpacked/*
du -sh dist/win-unpacked/resources/*

# 分析 node_modules
npx npm-why <package>  # 查看为什么安装了某个包
npx depcheck           # 检查未使用的依赖
```

## 二、打包体积优化

### 2.1 排除不必要文件

```yaml
# electron-builder.yml
files:
  # 只包含必要文件
  - "dist/**/*"
  - "package.json"
  
  # 排除源码
  - "!**/*.ts"
  - "!**/*.tsx"
  - "!**/*.map"
  - "!**/*.md"
  - "!**/*.markdown"
  
  # 排除配置文件
  - "!**/tsconfig*.json"
  - "!**/.eslintrc*"
  - "!**/.prettierrc*"
  - "!**/jest.config.*"
  
  # 排除 node_modules 中的无用文件
  - "!**/node_modules/**/*.md"
  - "!**/node_modules/**/*.ts"
  - "!**/node_modules/**/*.map"
  - "!**/node_modules/**/LICENSE*"
  - "!**/node_modules/**/CHANGELOG*"
  - "!**/node_modules/**/README*"
  - "!**/node_modules/**/{test,tests,__tests__}/**"
  - "!**/node_modules/**/{example,examples}/**"
  - "!**/node_modules/**/{doc,docs}/**"
  - "!**/node_modules/**/.bin/**"
  - "!**/node_modules/**/.github/**"
  - "!**/node_modules/**/.git/**"
```

### 2.2 生产依赖瘦身

```json
// package.json - 确保开发依赖正确分类
{
  "dependencies": {
    // 只包含运行时必需的依赖
    "electron-updater": "^6.0.0"
  },
  "devDependencies": {
    // 构建工具放这里
    "electron": "^28.0.0",
    "electron-builder": "^24.0.0",
    "typescript": "^5.0.0",
    "vite": "^5.0.0"
  }
}
```

```bash
# 检查是否有不必要的生产依赖
npm ls --production

# 清理并重装
rm -rf node_modules
npm ci --production  # 只安装生产依赖
```

### 2.3 Tree Shaking

```javascript
// vite.config.js - 渲染进程
export default {
  build: {
    rollupOptions: {
      treeshake: true
    }
  }
}

// esbuild - 主进程
{
  treeShaking: true,
  minify: true
}
```

### 2.4 压缩配置

```yaml
# electron-builder.yml
compression: maximum  # store | normal | maximum

# store:   不压缩，构建快但体积大
# normal:  默认，平衡
# maximum: 最大压缩，体积最小
```

### 2.5 单架构构建

```yaml
# 只构建需要的架构
win:
  target:
    - target: nsis
      arch: [x64]  # 不要同时构建 ia32

mac:
  target:
    - target: dmg
      arch: [arm64]  # 或 x64，不要 universal（体积翻倍）
```

```bash

# 但通常不推荐，维护成本高
```

## 三、构建速度优化

### 3.1 缓存利用

```yaml
# CI 缓存配置
- name: Cache node_modules
  uses: actions/cache@v4
  with:
    path: node_modules
    key: node-${{ runner.os }}-${{ hashFiles('**/package-lock.json') }}

- name: Cache Electron
  uses: actions/cache@v4
  with:
    path: |
      ~/.cache/electron
      ~/.cache/electron-builder
    key: electron-${{ runner.os }}-${{ hashFiles('**/package-lock.json') }}
```

### 3.2 并行构建

```bash
# esbuild 天然并行

# tsup 并行
# tsup.config.ts
export default {
  entry: ['src/main/index.ts', 'src/preload/index.ts'],
  // 多入口并行构建
}
```

### 3.3 增量构建

```javascript
// esbuild watch 模式
const ctx = await esbuild.context(config)
await ctx.watch()

// 只重新构建变化的文件
```

### 3.4 开发时跳过优化

```yaml
# 开发/测试构建
compression: store
asar: false  # 调试时可以禁用
```

### 3.5 CI 优化

```yaml
# 使用更快的包管理器
- run: npm ci --prefer-offline

# 或使用 pnpm
- uses: pnpm/action-setup@v2
  with:
    version: 8
- run: pnpm install --frozen-lockfile
```

## 四、启动性能优化

### 4.1 懒加载模块

```javascript
// 不好：启动时加载所有模块
const { app, BrowserWindow, Menu, Tray, ... } = require('electron')

// 好：按需加载
const { app, BrowserWindow } = require('electron')

function createTray() {
  const { Tray, Menu } = require('electron')
  // ...
}
```

### 4.2 延迟初始化

```javascript
// main.js
app.whenReady().then(async () => {
  // 先显示窗口
  createWindow()
  
  // 延迟初始化非关键功能
  setTimeout(() => {
    initTray()
    initAutoUpdater()
    initAnalytics()
  }, 1000)
})
```

### 4.3 Preload 优化

```javascript
// preload.js - 保持精简
const { contextBridge, ipcRenderer } = require('electron')

// 只暴露必要的 API
contextBridge.exposeInMainWorld('api', {
  // 最小化暴露的功能
})

// 不要在 preload 中做重计算
```

### 4.4 渲染进程优化

```javascript
// 代码分割
const HeavyComponent = React.lazy(() => import('./HeavyComponent'))

// 首屏优化
// 先显示骨架屏/加载状态
```

### 4.5 V8 快照

```javascript
// Electron 已经使用 V8 快照
// 可以通过 --v8-cache-options 进一步优化

// package.json
{
  "main": "dist/main/index.js",
  // Electron 启动参数
}
```

## 五、运行时性能

### 5.1 减少 IPC 调用

```javascript
// 不好：频繁的小消息
for (const item of items) {
  await ipcRenderer.invoke('process-item', item)
}

// 好：批量处理
await ipcRenderer.invoke('process-items', items)
```

### 5.2 使用 Worker

```javascript
// 主进程中使用 Worker 处理 CPU 密集任务
const { Worker } = require('worker_threads')

const worker = new Worker('./heavy-task.js')
worker.on('message', (result) => {
  // 处理结果
})
worker.postMessage(data)
```

### 5.3 内存管理

```javascript
// 监控内存使用
const usage = process.memoryUsage()
console.log('Memory:', {
  heapUsed: Math.round(usage.heapUsed / 1024 / 1024) + 'MB'
})

// 及时清理大对象
largeData = null
if (global.gc) global.gc()

// 窗口关闭时清理
win.on('closed', () => {
  win = null
})
```

## 七、完整优化配置

```yaml
# electron-builder.yml

appId: com.example.myapp
productName: My App

# 压缩
compression: maximum

# 文件配置
files:
  - "dist/**/*"
  - "package.json"
  - "!**/*.{ts,tsx,map,md}"
  - "!**/node_modules/**/*.{md,ts,map}"
  - "!**/node_modules/**/{test,tests,__tests__,docs}/**"
  - "!**/node_modules/.bin/**"

# asar
asar: true
asarUnpack:
  - "**/*.node"  # 只 unpack 必要的

# 只构建需要的目标
win:
  target:
    - target: nsis
      arch: [x64]

mac:
  target:
    - target: dmg
      arch: [arm64]
    - target: zip
      arch: [arm64]

linux:
  target:
    - target: AppImage
      arch: [x64]
```

```javascript
// vite.config.js - 渲染进程优化
export default {
  build: {
    minify: 'terser',
    terserOptions: {
      compress: {
        drop_console: true,  // 移除 console
        drop_debugger: true
      }
    },
    rollupOptions: {
      output: {
        manualChunks: {
          vendor: ['react', 'react-dom']
        }
      }
    }
  }
}
```
