# 自动更新实现

> Electron 应用的自动更新机制与最佳实践

## 一、概述

### 1.1 更新方式对比

```
┌─────────────────────────────────────────────────────────────────┐
│                    Electron 更新方式                             │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  1. electron-updater（推荐）                                    │
│     ├── 与 electron-builder 配套                               │
│     ├── 支持多平台                                              │
│     ├── 支持增量更新                                            │
│     └── 最成熟的方案                                            │
│                                                                 │
│  2. Electron autoUpdater（内置）                                │
│     ├── 需要搭配 Squirrel                                      │
│     ├── macOS 使用 Squirrel.Mac                                │
│     ├── Windows 使用 Squirrel.Windows                          │
│     └── 配置较复杂                                              │
│                                                                 │
│  3. 手动更新                                                    │
│     ├── 自己实现下载逻辑                                        │
│     ├── 完全控制更新流程                                        │
│     └── 工作量较大                                              │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

## 三、主进程实现

### 3.1 基础实现

```javascript
// src/main/updater.js
const { autoUpdater } = require('electron-updater')
const { app, dialog, BrowserWindow } = require('electron')
const log = require('electron-log')

// 配置日志
autoUpdater.logger = log
autoUpdater.logger.transports.file.level = 'info'

// 禁用自动下载
autoUpdater.autoDownload = false

// 允许降级（开发时可能需要）
autoUpdater.allowDowngrade = false

// 检查更新
async function checkForUpdates() {
  try {
    const result = await autoUpdater.checkForUpdates()
    return result
  } catch (error) {
    log.error('Update check failed:', error)
    return null
  }
}

// 事件监听
autoUpdater.on('checking-for-update', () => {
  log.info('Checking for update...')
})

autoUpdater.on('update-available', (info) => {
  log.info('Update available:', info.version)
  
  // 通知渲染进程
  const win = BrowserWindow.getFocusedWindow()
  if (win) {
    win.webContents.send('update-available', info)
  }
})

autoUpdater.on('update-not-available', (info) => {
  log.info('Update not available')
})

autoUpdater.on('download-progress', (progress) => {
  log.info(`Download progress: ${progress.percent.toFixed(2)}%`)
  
  // 通知渲染进程
  const win = BrowserWindow.getFocusedWindow()
  if (win) {
    win.webContents.send('download-progress', progress)
  }
})

autoUpdater.on('update-downloaded', (info) => {
  log.info('Update downloaded')
  
  // 通知渲染进程
  const win = BrowserWindow.getFocusedWindow()
  if (win) {
    win.webContents.send('update-downloaded', info)
  }
})

autoUpdater.on('error', (error) => {
  log.error('Update error:', error)
})

// 下载更新
async function downloadUpdate() {
  return autoUpdater.downloadUpdate()
}

// 安装更新
function quitAndInstall() {
  autoUpdater.quitAndInstall(false, true)
}

module.exports = {
  checkForUpdates,
  downloadUpdate,
  quitAndInstall,
  autoUpdater
}
```

### 3.2 IPC 集成

```javascript
// src/main/index.js
const { ipcMain, app } = require('electron')
const { checkForUpdates, downloadUpdate, quitAndInstall } = require('./updater')

// 应用启动后检查更新
app.whenReady().then(() => {
  // 延迟检查，避免影响启动
  setTimeout(() => {
    checkForUpdates()
  }, 3000)
})

// IPC 处理
ipcMain.handle('updater:check', async () => {
  const result = await checkForUpdates()
  return result?.updateInfo || null
})

ipcMain.handle('updater:download', async () => {
  await downloadUpdate()
})

ipcMain.handle('updater:install', () => {
  quitAndInstall()
})
```

### 3.3 Preload 暴露

```javascript
// src/preload/index.js
const { contextBridge, ipcRenderer } = require('electron')

contextBridge.exposeInMainWorld('updater', {
  // 检查更新
  check: () => ipcRenderer.invoke('updater:check'),
  
  // 下载更新
  download: () => ipcRenderer.invoke('updater:download'),
  
  // 安装更新
  install: () => ipcRenderer.invoke('updater:install'),
  
  // 监听事件
  onUpdateAvailable: (callback) => {
    ipcRenderer.on('update-available', (_, info) => callback(info))
  },
  
  onDownloadProgress: (callback) => {
    ipcRenderer.on('download-progress', (_, progress) => callback(progress))
  },
  
  onUpdateDownloaded: (callback) => {
    ipcRenderer.on('update-downloaded', (_, info) => callback(info))
  }
})
```

## 四、渲染进程 UI

### 4.1 React 示例

```tsx
// src/renderer/components/UpdateNotification.tsx
import { useState, useEffect } from 'react'

interface UpdateInfo {
  version: string
  releaseDate: string
  releaseNotes?: string
}

interface DownloadProgress {
  percent: number
  bytesPerSecond: number
  transferred: number
  total: number
}

export function UpdateNotification() {
  const [updateInfo, setUpdateInfo] = useState<UpdateInfo | null>(null)
  const [progress, setProgress] = useState<DownloadProgress | null>(null)
  const [downloaded, setDownloaded] = useState(false)
  const [checking, setChecking] = useState(false)
  
  useEffect(() => {
    // 监听更新事件
    window.updater.onUpdateAvailable((info) => {
      setUpdateInfo(info)
    })
    
    window.updater.onDownloadProgress((prog) => {
      setProgress(prog)
    })
    
    window.updater.onUpdateDownloaded(() => {
      setDownloaded(true)
      setProgress(null)
    })
  }, [])
  
  const handleCheck = async () => {
    setChecking(true)
    await window.updater.check()
    setChecking(false)
  }
  
  const handleDownload = async () => {
    await window.updater.download()
  }
  
  const handleInstall = () => {
    window.updater.install()
  }
  
  // 无更新
  if (!updateInfo) {
    return (
      <button onClick={handleCheck} disabled={checking}>
        {checking ? '检查中...' : '检查更新'}
      </button>
    )
  }
  
  // 下载完成
  if (downloaded) {
    return (
      <div className="update-notification">
        <p>更新 v{updateInfo.version} 已下载完成</p>
        <button onClick={handleInstall}>立即重启安装</button>
      </div>
    )
  }
  
  // 下载中
  if (progress) {
    return (
      <div className="update-notification">
        <p>正在下载 v{updateInfo.version}</p>
        <progress value={progress.percent} max={100} />
        <span>{progress.percent.toFixed(1)}%</span>
      </div>
    )
  }
  
  // 有更新可用
  return (
    <div className="update-notification">
      <p>发现新版本 v{updateInfo.version}</p>
      <button onClick={handleDownload}>下载更新</button>
    </div>
  )
}
```

## 五、更新策略

### 5.1 全量更新 vs 增量更新

```
┌─────────────────────────────────────────────────────────────────┐
│                    更新方式对比                                  │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  全量更新：                                                      │
│  ├── 下载完整安装包                                             │
│  ├── 简单可靠                                                   │
│  └── 带宽消耗大（80-150MB）                                     │
│                                                                 │
│  增量更新（Differential Update）：                               │
│  ├── 使用 blockmap 对比差异                                     │
│  ├── 只下载变化的部分                                           │
│  ├── 节省 70-90% 带宽                                           │
│  └── 需要 differentialPackage: true                            │
│                                                                 │
│  blockmap 工作原理：                                             │
│  ┌─────────────────────────────────────────┐                   │
│  │  旧版本    │ Block1 │ Block2 │ Block3 │  │                   │
│  │  新版本    │ Block1 │ Block2'│ Block4 │  │                   │
│  │  下载      │        │ Block2'│ Block4 │  │ ← 只下载差异      │
│  └─────────────────────────────────────────┘                   │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

### 5.2 强制更新 vs 可选更新

```javascript
// src/main/updater.js

autoUpdater.on('update-available', (info) => {
  const currentVersion = app.getVersion()
  const newVersion = info.version
  
  // 解析版本号
  const [curMajor] = currentVersion.split('.')
  const [newMajor] = newVersion.split('.')
  
  // 主版本号变化 = 强制更新
  const isForced = newMajor > curMajor
  
  // 或者通过更新清单中的字段判断
  // const isForced = info.forceUpdate === true
  
  if (isForced) {
    // 强制更新：自动下载并提示
    autoUpdater.downloadUpdate()
  } else {
    // 可选更新：通知用户
    notifyUser(info)
  }
})
```

### 5.3 静默更新

```javascript
// 后台静默下载，下次启动时安装
autoUpdater.autoDownload = true
autoUpdater.autoInstallOnAppQuit = true

autoUpdater.on('update-downloaded', () => {
  // 不提示用户，下次启动自动安装
  log.info('Update downloaded, will install on next restart')
})
```

### 5.4 灰度发布

```javascript
// electron-builder.yml 中配置 staged rollout
publish:
  - provider: generic
    url: https://releases.example.com/

// 或自己实现
autoUpdater.on('update-available', async (info) => {
  const userId = getUserId()
  const rolloutPercentage = info.stagingPercentage || 100
  
  // 基于用户 ID 计算是否在灰度范围
  const hash = hashCode(userId)
  const inRollout = (hash % 100) < rolloutPercentage
  
  if (inRollout) {
    autoUpdater.downloadUpdate()
  }
})
```

## 七、自定义更新服务器

### 7.1 Node.js 服务器示例

```javascript
// server.js
const express = require('express')
const path = require('path')
const fs = require('fs')

const app = express()
const RELEASES_DIR = './releases'

// 静态文件服务
app.use('/releases', express.static(RELEASES_DIR))

// 更新检查接口（可选，用于自定义逻辑）
app.get('/api/check-update', (req, res) => {
  const { version, platform } = req.query
  
  // 读取最新版本信息
  const latestFile = platform === 'darwin' 
    ? 'latest-mac.yml' 
    : 'latest.yml'
  
  const latestPath = path.join(RELEASES_DIR, latestFile)
  
  if (!fs.existsSync(latestPath)) {
    return res.json({ hasUpdate: false })
  }
  
  const yaml = require('yaml')
  const latestInfo = yaml.parse(fs.readFileSync(latestPath, 'utf8'))
  
  const hasUpdate = compareVersions(latestInfo.version, version) > 0
  
  res.json({
    hasUpdate,
    version: latestInfo.version,
    releaseDate: latestInfo.releaseDate,
    downloadUrl: `/releases/${latestInfo.path}`
  })
})

app.listen(3000, () => {
  console.log('Update server running on port 3000')
})
```

### 7.2 Nginx 配置

```nginx
server {
    listen 443 ssl;
    server_name releases.example.com;
    
    root /var/www/releases;
    
    # YAML 文件
    location ~ \.yml$ {
        add_header Content-Type text/yaml;
    }
    
    # 下载文件
    location / {
        try_files $uri =404;
        
        # 大文件优化
        sendfile on;
        tcp_nopush on;
    }
    
    # 支持 Range 请求（增量更新需要）
    location ~ \.(exe|dmg|zip|AppImage)$ {
        add_header Accept-Ranges bytes;
    }
}
```

### 9.2 回滚机制

```javascript
// 记录版本历史
const Store = require('electron-store')
const store = new Store()

function recordVersion() {
  const versions = store.get('versionHistory', [])
  const currentVersion = app.getVersion()
  
  if (versions[0] !== currentVersion) {
    versions.unshift(currentVersion)
    store.set('versionHistory', versions.slice(0, 5)) // 保留最近5个
  }
}

// 检测启动问题
let startupAttempts = store.get('startupAttempts', 0)

if (startupAttempts > 3) {
  // 多次启动失败，可能是更新导致
  log.error('Multiple startup failures detected')
  // 这里可以实现回滚逻辑
}

store.set('startupAttempts', startupAttempts + 1)

// 应用正常运行后重置计数
setTimeout(() => {
  store.set('startupAttempts', 0)
}, 10000)
```

```javascript
// src/main/updater.js - 完整实现
const { autoUpdater } = require('electron-updater')
const { app, dialog, ipcMain, BrowserWindow } = require('electron')
const log = require('electron-log')

class Updater {
  constructor() {
    this.updateInfo = null
    this.downloadProgress = null
    this.isDownloaded = false
    
    this.init()
  }
  
  init() {
    autoUpdater.logger = log
    autoUpdater.autoDownload = false
    autoUpdater.autoInstallOnAppQuit = true
    
    this.bindEvents()
    this.bindIPC()
  }
  
  bindEvents() {
    autoUpdater.on('checking-for-update', () => {
      log.info('Checking for update...')
      this.sendToRenderer('update-status', 'checking')
    })
    
    autoUpdater.on('update-available', (info) => {
      log.info('Update available:', info.version)
      this.updateInfo = info
      this.sendToRenderer('update-available', info)
    })
    
    autoUpdater.on('update-not-available', () => {
      log.info('Update not available')
      this.sendToRenderer('update-status', 'not-available')
    })
    
    autoUpdater.on('download-progress', (progress) => {
      this.downloadProgress = progress
      this.sendToRenderer('download-progress', progress)
    })
    
    autoUpdater.on('update-downloaded', (info) => {
      log.info('Update downloaded')
      this.isDownloaded = true
      this.sendToRenderer('update-downloaded', info)
    })
    
    autoUpdater.on('error', (error) => {
      log.error('Update error:', error)
      this.sendToRenderer('update-error', error.message)
    })
  }
  
  bindIPC() {
    ipcMain.handle('updater:check', () => this.checkForUpdates())
    ipcMain.handle('updater:download', () => this.downloadUpdate())
    ipcMain.handle('updater:install', () => this.quitAndInstall())
    ipcMain.handle('updater:getInfo', () => ({
      updateInfo: this.updateInfo,
      downloadProgress: this.downloadProgress,
      isDownloaded: this.isDownloaded
    }))
  }
  
  async checkForUpdates() {
    try {
      const result = await autoUpdater.checkForUpdates()
      return result?.updateInfo || null
    } catch (error) {
      log.error('Check update failed:', error)
      return null
    }
  }
  
  async downloadUpdate() {
    return autoUpdater.downloadUpdate()
  }
  
  quitAndInstall() {
    autoUpdater.quitAndInstall(false, true)
  }
  
  sendToRenderer(channel, data) {
    const windows = BrowserWindow.getAllWindows()
    windows.forEach(win => {
      if (!win.isDestroyed()) {
        win.webContents.send(channel, data)
      }
    })
  }
}

module.exports = new Updater()
```
